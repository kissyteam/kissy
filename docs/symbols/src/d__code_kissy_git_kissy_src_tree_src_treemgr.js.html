1 /**
  2  * @fileOverview tree management utils
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("tree/treemgr", function(S, Event) {
  6 
  7     function TreeMgr() {
  8     }
  9 
 10     TreeMgr.ATTRS = {
 11         // 默认 true
 12         // 是否显示根节点
 13         showRootNode:{
 14             view:true
 15         },
 16         selectedItem:{},
 17         tree:{
 18             valueFn:function() {
 19                 return this;
 20             }
 21         },
 22         focusable:{
 23             value:true
 24         }
 25     };
 26 
 27     S.augment(TreeMgr, {
 28         /*
 29          加快从事件代理获取原事件节点
 30          */
 31         __getAllNodes:function() {
 32             var self=this;
 33             if (!self._allNodes) {
 34                 self._allNodes = {};
 35             }
 36             return self._allNodes;
 37         },
 38 
 39         __renderUI:function() {
 40              var self=this;
 41             // add 过那么一定调用过 checkIcon 了
 42             if (!self.get("children").length) {
 43                 self._computeClass("root_renderUI");
 44             }
 45         },
 46 
 47         _register:function(c) {
 48             this.__getAllNodes()[c.get("id")] = c;
 49         },
 50 
 51         _unregister:function(c) {
 52             delete this.__getAllNodes()[c.get("id")];
 53         },
 54 
 55         _handleKeyEventInternal:function(e) {
 56             var current = this.get("selectedItem");
 57             if (e.keyCode == Event.KeyCodes.ENTER) {
 58                 // 传递给真正的单个子节点
 59                 return current._performInternal(e);
 60             }
 61             return current._keyNav(e);
 62         },
 63 
 64         // 重写 delegatechildren ，缓存加快从节点获取对象速度
 65         getOwnerControl:function(node) {
 66             var self = this,
 67                 n,
 68                 allNodes = self.__getAllNodes(),
 69                 elem = self.get("el")[0];
 70             while (node && node !== elem) {
 71                 if (n = allNodes[node.id]) {
 72                     return n;
 73                 }
 74                 node = node.parentNode;
 75             }
 76             // 最终自己处理
 77             return self;
 78         },
 79 
 80         // 单选
 81         _uiSetSelectedItem:function(n, ev) {
 82             if (ev.prevVal) {
 83                 ev.prevVal.set("selected", false);
 84             }
 85             n.set("selected", true);
 86         },
 87 
 88 
 89         _uiSetFocused:function(v) {
 90             var self = this;
 91             self.constructor.superclass._uiSetFocused.call(self, v);
 92             // 得到焦点时没有选择节点
 93             // 默认选择自己
 94             if (v && !self.get("selectedItem")) {
 95                 self.select();
 96             }
 97         }
 98     });
 99 
100     return TreeMgr;
101 }, {
102     requires:['event']
103 });
