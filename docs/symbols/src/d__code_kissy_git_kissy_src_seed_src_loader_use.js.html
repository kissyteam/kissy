1 /**
  2  * @fileOverview use and attach mod
  3  * @author  yiminghe@gmail.com,lifesinger@gmail.com
  4  */
  5 (function (S, loader, utils, data) {
  6 
  7     if ("require" in this) {
  8         return;
  9     }
 10 
 11     var LOADED = data.LOADED,
 12         ATTACHED = data.ATTACHED;
 13 
 14     S.mix(loader,
 15         /**
 16          * @lends KISSY
 17          */
 18         {
 19             /**
 20              * Start load specific mods, and fire callback when these mods and requires are attached.
 21              * @example
 22              * <code>
 23              * S.use('mod-name', callback, config);
 24              * S.use('mod1,mod2', callback, config);
 25              * </code>
 26              * @param {String|String[]} modNames names of mods to be loaded,if string then separated by space
 27              * @param {Function} callback callback when modNames are all loaded,
 28              *                   with KISSY as first argument and mod's value as the following argumwnts
 29              * @param {Object} cfg special config for this use
 30              */
 31             use:function (modNames, callback, cfg) {
 32                 if (S.isString(modNames)) {
 33                     modNames = modNames.replace(/\s+/g, "").split(',');
 34                 }
 35                 utils.indexMapping(modNames);
 36                 cfg = cfg || {};
 37 
 38                 var self = this,
 39                     fired;
 40 
 41                 // 已经全部 attached, 直接执行回调即可
 42                 if (self.__isAttached(modNames)) {
 43                     var mods = self.__getModules(modNames);
 44                     callback && callback.apply(self, mods);
 45                     return;
 46                 }
 47 
 48                 // 有尚未 attached 的模块
 49                 S.each(modNames, function (modName) {
 50                     // 从 name 开始调用，防止不存在模块
 51                     self.__attachModByName(modName, function () {
 52                         if (!fired &&
 53                             self.__isAttached(modNames)) {
 54                             fired = true;
 55                             var mods = self.__getModules(modNames);
 56                             callback && callback.apply(self, mods);
 57                         }
 58                     }, cfg);
 59                 });
 60 
 61                 return self;
 62             },
 63 
 64             __getModules:function (modNames) {
 65                 var self = this,
 66                     mods = [self];
 67 
 68                 S.each(modNames, function (modName) {
 69                     if (!utils.isCss(modName)) {
 70                         mods.push(self.require(modName));
 71                     }
 72                 });
 73                 return mods;
 74             },
 75 
 76             /**
 77              * get module's value defined by define function
 78              * @param {string} moduleName
 79              * @private
 80              */
 81             require:function (moduleName) {
 82                 var self = this,
 83                     mods = self.Env.mods,
 84                     mod = mods[moduleName],
 85                     re = self['onRequire'] && self['onRequire'](mod);
 86                 if (re !== undefined) {
 87                     return re;
 88                 }
 89                 return mod && mod.value;
 90             },
 91 
 92             // 加载指定模块名模块，如果不存在定义默认定义为内部模块
 93             __attachModByName:function (modName, callback, cfg) {
 94                 var self = this,
 95                     mods = self.Env.mods;
 96 
 97                 var mod = mods[modName];
 98                 //没有模块定义
 99                 if (!mod) {
100                     // 默认 js/css 名字
101                     // 不指定 .js 默认为 js
102                     // 指定为 css 载入 .css
103                     var componentJsName = self.Config['componentJsName'] ||
104                         function (m) {
105                             var suffix = "js", match;
106                             if (match = m.match(/(.+)\.(js|css)$/i)) {
107                                 suffix = match[2];
108                                 m = match[1];
109                             }
110                             return m + '-min.' + suffix;
111                         },
112                         path = componentJsName(modName);
113                     mod = {
114                         path:path,
115                         charset:'utf-8'
116                     };
117                     //添加模块定义
118                     mods[modName] = mod;
119                 }
120 
121                 mod.name = modName;
122 
123                 if (mod && mod.status === ATTACHED) {
124                     callback();
125                     return;
126                 }
127 
128                 // 先从 global 里取
129                 if (cfg.global) {
130                     self.__mixMod(modName, cfg.global);
131                 }
132 
133                 self.__attach(mod, callback, cfg);
134             },
135 
136             /**
137              * Attach a module and all required modules.
138              */
139             __attach:function (mod, callback, cfg) {
140                 var self = this,
141                     r,
142                     rMod,
143                     i,
144                     attached = 0,
145                     mods = self.Env.mods,
146                     //复制一份当前的依赖项出来，防止 add 后修改！
147                     requires = (mod['requires'] || []).concat();
148 
149                 mod['requires'] = requires;
150 
151                 /**
152                  * check cyclic dependency between mods
153                  * @private
154                  */
155                 function cyclicCheck() {
156                     var __allRequires,
157                         myName = mod.name,
158                         r, r2, rmod,
159                         r__allRequires,
160                         requires = mod.requires;
161                     // one mod's all requires mods to run its callback
162                     __allRequires = mod.__allRequires = mod.__allRequires || {};
163                     for (var i = 0; i < requires.length; i++) {
164                         r = requires[i];
165                         rmod = mods[r];
166                         __allRequires[r] = 1;
167                         if (rmod && (r__allRequires = rmod.__allRequires)) {
168                             for (r2 in r__allRequires) {
169                                 if (r__allRequires.hasOwnProperty(r2)) {
170                                     __allRequires[r2] = 1;
171                                 }
172                             }
173                         }
174                     }
175                     if (__allRequires[myName]) {
176                         var t = [];
177                         for (r in __allRequires) {
178                             if (__allRequires.hasOwnProperty(r)) {
179                                 t.push(r);
180                             }
181                         }
182                         S.error("find cyclic dependency by mod " + myName + " between mods : " + t.join(","));
183                     }
184                 }
185 
186                 if (S.Config.debug) {
187                     cyclicCheck();
188                 }
189 
190                 // attach all required modules
191                 for (i = 0; i < requires.length; i++) {
192                     r = requires[i] = utils.normalDepModuleName(mod.name, requires[i]);
193                     rMod = mods[r];
194                     if (rMod && rMod.status === ATTACHED) {
195                         //no need
196                     } else {
197                         self.__attachModByName(r, fn, cfg);
198                     }
199                 }
200 
201                 // load and attach this module
202                 self.__buildPath(mod, self.__getPackagePath(mod));
203 
204                 self.__load(mod, function () {
205 
206                     // add 可能改了 config，这里重新取下
207                     mod['requires'] = mod['requires'] || [];
208 
209                     var newRequires = mod['requires'],
210                         needToLoad = [];
211 
212                     //本模块下载成功后串行下载 require
213                     for (i = 0; i < newRequires.length; i++) {
214                         r = newRequires[i] = utils.normalDepModuleName(mod.name, newRequires[i]);
215                         var rMod = mods[r],
216                             inA = S.inArray(r, requires);
217                         //已经处理过了或将要处理
218                         if (rMod &&
219                             rMod.status === ATTACHED
220                             //已经正在处理了
221                             || inA) {
222                             //no need
223                         } else {
224                             //新增的依赖项
225                             needToLoad.push(r);
226                         }
227                     }
228 
229                     if (needToLoad.length) {
230                         for (i = 0; i < needToLoad.length; i++) {
231                             self.__attachModByName(needToLoad[i], fn, cfg);
232                         }
233                     } else {
234                         fn();
235                     }
236                 }, cfg);
237 
238                 function fn() {
239                     if (!attached &&
240                         self.__isAttached(mod['requires'])) {
241 
242                         if (mod.status === LOADED) {
243                             self.__attachMod(mod);
244                         }
245                         if (mod.status === ATTACHED) {
246                             attached = 1;
247                             callback();
248                         }
249                     }
250                 }
251             },
252 
253             __attachMod:function (mod) {
254                 var self = this,
255                     fns = mod.fns;
256 
257                 if (fns) {
258                     S.each(fns, function (fn) {
259                         var value;
260                         if (S.isFunction(fn)) {
261                             value = fn.apply(self, self.__getModules(mod['requires']));
262                         } else {
263                             value = fn;
264                         }
265                         mod.value = mod.value || value;
266                     });
267                 }
268 
269                 mod.status = ATTACHED;
270             }
271         });
272 })(KISSY, KISSY.__loader, KISSY.__loaderUtils, KISSY.__loaderData);
