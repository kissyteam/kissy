 12     function Paste(editor) {
 13         var self = this;
 14         self.editor = editor;
 15         self._init();
 16     }
 17 
 18     S.augment(Paste, {
 19         _init:function () {
 20             var self = this,
 21                 editor = self.editor;
 22             // Event.on(editor.document.body, UA['ie'] ? "beforepaste" : "keydown", self._paste, self);
 23             // beforepaste not fire on webkit and firefox
 24             // paste fire too later in ie ,cause error
 25             // 奇怪哦
 26             // http://help.dottoro.com/ljxqbxkf.php
 27             // refer : http://stackoverflow.com/questions/2176861/javascript-get-clipboard-data-on-paste-event-cross-browser
 28             Event.on(editor.get("document")[0].body,
 29                 UA['webkit'] ? 'paste' : (UA.gecko ? 'paste' : 'beforepaste'),
 30                 self._paste, self);
 31 
 32             // Dismiss the (wrong) 'beforepaste' event fired on context menu open. (#7953)
 33             Event.on(editor.get("document")[0].body, 'contextmenu', function () {
 34                 depressBeforeEvent = 1;
 35                 setTimeout(function () {
 36                     depressBeforeEvent = 0;
 37                 }, 10);
 38             });
 39             editor.addCommand("copy", new cutCopyCmd("copy"));
 40             editor.addCommand("cut", new cutCopyCmd("cut"));
 41             editor.addCommand("paste", new cutCopyCmd("paste"));
 42 
 43         },
 44         _paste:function (ev) {
 45 
 46             if (depressBeforeEvent) {
 47                 return;
 48             }
 49 
 50             // ie beforepaste 会触发两次，第一次 pastebin 为锚点内容，奇怪
 51             // chrome keydown 也会两次
 52             S.log(ev.type + " : " + " paste event happen");
 53 
 54             var self = this,
 55                 editor = self.editor,
 56                 doc = editor.get("document")[0];
 57 
 58             // Avoid recursions on 'paste' event or consequent paste too fast. (#5730)
 59             if (doc.getElementById('ke_pastebin')) {
 60                 // ie beforepaste 会重复触发
 61                 // chrome keydown 也会重复触发
 62                 // 第一次 bms 是对的，但是 pasterbin 内容是错的
 63                 // 第二次 bms 是错的，但是内容是对的
 64                 // 这样返回刚好，用同一个 pastebin 得到最后的正确内容
 65                 // bms 第一次时创建成功
 66                 S.log(ev.type + " : trigger more than once ...");
 67                 return;
 68             }
 69 
 70             var sel = editor.getSelection(),
 71                 range = new KERange(doc);
 72 
 73             // Create container to paste into
 74             var pastebin = $(UA['webkit'] ? '<body></body>' : '<div></div>', null, doc);
 75             pastebin.attr('id', 'ke_pastebin');
 76             // Safari requires a filler node inside the div to have the content pasted into it. (#4882)
 77             UA['webkit'] && pastebin[0].appendChild(doc.createTextNode('\xa0'));
 78             doc.body.appendChild(pastebin[0]);
 79 
 80             pastebin.css({
 81                 position:'absolute',
 82                 // Position the bin exactly at the position of the selected element
 83                 // to avoid any subsequent document scroll.
 84                 top:sel.getStartElement().offset().top + 'px',
 85                 width:'1px',
 86                 height:'1px',
 87                 overflow:'hidden'
 88             });
 89 
 90             // It's definitely a better user experience if we make the paste-bin pretty unnoticed
 91             // by pulling it off the screen.
 92             pastebin.css('left', '-1000px');
 93 
 94             var bms = sel.createBookmarks();
 95 
 96             // Turn off design mode temporarily before give focus to the paste bin.
 97             range.setStartAt(pastebin, KER.POSITION_AFTER_START);
 98             range.setEndAt(pastebin, KER.POSITION_BEFORE_END);
 99             range.select(true);
100             //self._running = true;
101             // Wait a while and grab the pasted contents
102             setTimeout(function () {
103 
104                 //self._running = false;
105                 pastebin.remove();
106 
107                 // Grab the HTML contents.
108                 // We need to look for a apple style wrapper on webkit it also adds
109                 // a div wrapper if you copy/paste the body of the editor.
110                 // Remove hidden div and restore selection.
111                 var bogusSpan;
112 
113                 pastebin = ( UA['webkit']
114                     && ( bogusSpan = pastebin.first() )
115                     && (bogusSpan.hasClass('Apple-style-span') ) ?
116                     bogusSpan : pastebin );
117 
118                 sel.selectBookmarks(bms);
119 
120                 var html = pastebin.html();
121 
122                 //S.log("paster " + html);
123 
124                 //莫名其妙会有这个东西！，不知道
125                 //去掉
126                 if (!( html = S.trim(html.replace(/<span[^>]+_ke_bookmark[^<]*?<\/span>( )*/ig, '')) )) {
127                     // ie 第2次触发 beforepaste 会报错！
128                     // 第一次 bms 是对的，但是 pasterbin 内容是错的
129                     // 第二次 bms 是错的，但是内容是对的
130                     return;
131                 }
132 
133                 S.log("paster " + html);
134 
135                 var re = editor.fire("paste", {
136                     html:html,
137                     holder:pastebin
138                 });
139 
140                 if (re !== undefined) {
141                     html = re;
142                 }
143 
144                 var dataFilter = null;
145 
146                 // MS-WORD format sniffing.
147                 if (/(class="?Mso|style="[^"]*\bmso\-|w:WordDocument)/.test(html)) {
148                     dataFilter = editor.htmlDataProcessor.wordFilter;
149                 }
150 
151                 editor.insertHtml(html, dataFilter);
152 
153             }, 0);
154         }
155     });

157     // Tries to execute any of the paste, cut or copy commands in IE. Returns a
158     // boolean indicating that the operation succeeded.
159     var execIECommand = function (editor, command) {
160         var doc = editor.get("document")[0],
161             body = $(doc.body);
162 
163         var enabled = false;
164         var onExec = function () {
165             enabled = true;
166         };
167 
168         // The following seems to be the only reliable way to detect that
169         // clipboard commands are enabled in IE. It will fire the
170         // onpaste/oncut/oncopy events only if the security settings allowed
171         // the command to execute.
172         body.on(command, onExec);
173 
174         // IE6/7: document.execCommand has problem to paste into positioned element.
175         ( UA['ie'] > 7 ? doc : doc.selection.createRange() ) [ 'execCommand' ](command);
176 
177         body.detach(command, onExec);
178 
179         return enabled;
180     };
181 
182     // Attempts to execute the Cut and Copy operations.
183     var tryToCutCopy = UA['ie'] ?
184         function (editor, type) {
185             return execIECommand(editor, type);
186         }
187         : // !IE.
188         function (editor, type) {
189             try {
190                 // Other browsers throw an error if the command is disabled.
191                 return editor.get("document")[0].execCommand(type);
192             }
193             catch (e) {
194                 return false;
195             }
196         };

198     var error_types = {
199         cut:"您的浏览器安全设置不允许编辑器自动执行剪切操作，请使用键盘快捷键(Ctrl/Cmd+X)来完成",
200         copy:"您的浏览器安全设置不允许编辑器自动执行复制操作，请使用键盘快捷键(Ctrl/Cmd+C)来完成",
201         paste:"您的浏览器安全设置不允许编辑器自动执行粘贴操作，请使用键盘快捷键(Ctrl/Cmd+V)来完成"
202     };

204     // A class that represents one of the cut or copy commands.
205     var cutCopyCmd = function (type) {
206         this.type = type;
207     };
208 
209     cutCopyCmd.prototype = {
210         exec:function (editor) {
211             this.type == 'cut' && fixCut(editor);
212 
213             var success = tryToCutCopy(editor, this.type);
214 
215             if (!success)
216                 alert(error_types[this.type]);		// Show cutError or copyError.
217 
218             return success;
219         }
220     };
221     var KES = Editor.Selection;
222     // Cutting off control type element in IE standards breaks the selection entirely. (#4881)
223     function fixCut(editor) {
224         if (!UA['ie'] || editor.get("document")[0].compatMode == 'BackCompat')
225             return;
226 
227         var sel = editor.getSelection();
228         var control;
229         if (( sel.getType() == KES.SELECTION_ELEMENT ) && ( control = sel.getSelectedElement() )) {
230             var range = sel.getRanges()[ 0 ];
231             var dummy = $(editor.get("document")[0].createTextNode(''));
232             dummy.insertBefore(control);
233             range.setStartBefore(dummy);
234             range.setEndAfter(control);
235             sel.selectRanges([ range ]);
236 
237             // Clear up the fix if the paste wasn't succeeded.
238             setTimeout(function () {
239                 // Element still online?
240                 if (control.parent()) {
241                     dummy.remove();
242                     sel.selectElement(control);
243                 }
244             }, 0);
245         }
246     }

263 
264             /**
265              * 给所有右键都加入复制粘贴
266              */
267             editor.on("contextmenu", function (ev) {
268                 var contextmenu = ev.contextmenu;
269 
270                 if (contextmenu.__copy_fix) {
271                     return;
272                 }
273 
274                 contextmenu.__copy_fix = 1;
275 
276                 for (var i in pastes) {
277                     if (pastes.hasOwnProperty(i)) {
278                         contextmenu.addChild({
279                             xclass:'menuitem',
280                             content:lang[i],
281                             value:i
282                         });
283                     }
284                 }
285 
286                 contextmenu.on('click', function (e) {
287                     var value = e.target.get("value");
288                     if (pastes[value]) {
289                         this.hide();
290                         // 给 ie 一点 hide() 中的事件触发 handler 运行机会，
291                         // 原编辑器获得焦点后再进行下步操作
292                         setTimeout(function () {
293                             editor.execCommand(value);
294                         }, 30);
295                     }
296                 });
297             });
298         }
