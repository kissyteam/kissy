1 /**
  2  * @fileOverview submenu render for kissy ,extend menuitem render with arrow
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/submenurender", function(S, UIBase, MenuItemRender) {
  6         var SubMenuRender;
  7         var ARROW_TMPL = '<span class="{prefixCls}submenu-arrow">â–º<' + '/span>';
  8         SubMenuRender = UIBase.create(MenuItemRender, {
  9             renderUI:function() {
 10                 var self = this,
 11                     el = self.get("el"),
 12                     contentEl = self.get("contentEl");
 13                 el.attr("aria-haspopup", "true");
 14                 contentEl.append(S.substitute(ARROW_TMPL, {
 15                     prefixCls:this.get("prefixCls")
 16                 }));
 17             },
 18             _uiSetContent:function(v) {
 19                 var self = this;
 20                 SubMenuRender.superclass._uiSetContent.call(self, v);
 21                 self.get("contentEl").append(S.substitute(ARROW_TMPL, {
 22                     prefixCls:this.get("prefixCls")
 23                 }));
 24             }
 25 
 26         });
 27         return SubMenuRender;
 28     },
 29     {
 30         requires:['uibase','./menuitemrender']
 31     });
