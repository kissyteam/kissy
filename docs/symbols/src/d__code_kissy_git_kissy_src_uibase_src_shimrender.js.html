  5 KISSY.add("uibase/shimrender", function(S, Node) {
  6 
  7     function Shim() {
  8     }
  9 
 10     Shim.ATTRS = {
 11         shim:{
 12             value:true
 13         }
 14     };
 15 
 16     Shim.prototype = {
 17 
 18         _uiSetShim:function(v) {
 19             var self = this,el = self.get("el");
 20             if (v && !self.__shimEl) {
 21                 self.__shimEl = new Node("<" + "iframe style='position: absolute;" +
 22                     "border: none;" +
 23                     "width: expression(this.parentNode.offsetWidth);" +
 24                     "top: 0;" +
 25                     "opacity: 0;" +
 26                     "filter: alpha(opacity=0);" +
 27                     "left: 0;" +
 28                     "z-index: -1;" +
 29                     "height: expression(this.parentNode.offsetHeight);" + "'/>");
 30                 el.prepend(self.__shimEl);
 31             } else if (!v && self.__shimEl) {
 32                 self.__shimEl.remove();
 33                 delete self.__shimEl;
 34             }
 35         }
 36     };
 37 
 38     return Shim;
 39 }, {
 40     requires:['node']
 41 });
