1 /**
  2  * Common btn for list.
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/listUtils/btn", function (S, Editor, Button) {
  6 
  7     function onClick() {
  8         var editor = this.get("editor");
  9         var cmd = this.get("cmdType");
 10         editor.execCommand(cmd);
 11         editor.focus();
 12     }
 13 
 14     return Button.Toggle.extend({
 15         initializer:function () {
 16             var self = this;
 17             self.on("click", onClick, self);
 18             var editor = self.get("editor");
 19             editor.on("selectionChange", function (e) {
 20                 var cmd = Editor.Utils.getQueryCmd(self.get("cmdType"));
 21                 if (editor.execCommand(cmd, e.path)) {
 22                     self.set("checked", true);
 23                 } else {
 24                     self.set("checked", false);
 25                 }
 26             })
 27         }
 28     }, {
 29         ATTRS:{
 30             mode:{
 31                 value:Editor.WYSIWYG_MODE
 32             }
 33         }
 34     });
 35 }, {
 36     requires:['editor', '../button/']
 37 });
