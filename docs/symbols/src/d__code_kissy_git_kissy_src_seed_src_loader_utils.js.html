 28     function indexMap(s) {
 29         if (S.isArray(s)) {
 30             var ret = [];
 31             S.each(s, function (si) {
 32                 ret.push(indexMap(si));
 33             });
 34             return ret;
 35         }
 36         return indexMapStr(s);
 37     }

 39     function indexMapStr(s) {
 40         if (/(.+\/)(\?t=.+)?$/.test(s)) {
 41             return RegExp.$1 + "index" + RegExp.$2;
 42         } else {
 43             return s
 44         }
 45     }

 47     function removeSuffixAndTagFromModName(modName) {
 48         var tag = undefined,
 49             m,
 50             withTagReg = /([^?]+)(?:\?t=(.+))/;
 51 
 52         if (m = modName.match(withTagReg)) {
 53             modName = m[1];
 54             tag = m[2];
 55         }
 56 
 57         // js do not need suffix
 58         modName = modName.replace(/\.js$/i, "");
 59 
 60         return {
 61             modName:modName,
 62             tag:tag
 63         };
 64     }

 67     function getPackageInfo(self, mod) {
 68         if (mod.packageInfo) {
 69             return mod.packageInfo;
 70         }
 71 
 72         var modName = mod.name,
 73             Config = self.Config,
 74             packages = Config.packages || {},
 75             pName = "",
 76             packageDesc;
 77 
 78         for (var p in packages) {
 79             if (packages.hasOwnProperty(p)) {
 80                 if (S.startsWith(modName, p) &&
 81                     p.length > pName.length) {
 82                     pName = p;
 83                 }
 84             }
 85         }
 86 
 87         packageDesc = packages[pName] || {
 88             // 无包，kissy 自身模块
 89             __kissy:1
 90         };
 91 
 92         S.mix(packageDesc, {
 93             name:pName,
 94             tag:encodeURIComponent(Config.tag),
 95             path:Config.base,
 96             debug:Config.debug,
 97             charset:"utf-8"
 98         }, false);
 99 
100         mod.packageInfo = packageDesc;
101 
102         return packageDesc;
103     }

127         /**
128          * resolve relative part of path
129          * x/../y/z -> y/z
130          * x/./y/z -> x/y/z
131          * @param path uri path
132          * @return {string} resolved path
133          * @description similar to path.normalize in nodejs
134          */
135         normalizePath:function (path) {
136             var paths = path.split("/"),
137                 re = [],
138                 p;
139             for (var i = 0; i < paths.length; i++) {
140                 p = paths[i];
141                 if (p == ".") {
142                 } else if (p == "..") {
143                     re.pop();
144                 } else {
145                     re.push(p);
146                 }
147             }
148             return re.join("/");
149         },

151         /**
152          * 根据当前模块以及依赖模块的相对路径，得到依赖模块的绝对路径
153          * @param moduleName 当前模块
154          * @param depName 依赖模块
155          * @return {string|Array} 依赖模块的绝对路径
156          * @description similar to path.resolve in nodejs
157          */
158         normalDepModuleName:function (moduleName, depName) {
159             if (!depName) {
160                 return depName;
161             }
162             if (S.isArray(depName)) {
163                 for (var i = 0; i < depName.length; i++) {
164                     depName[i] = utils.normalDepModuleName(moduleName, depName[i]);
165                 }
166                 return depName;
167             }
168             if (startsWith(depName, "../") || startsWith(depName, "./")) {
169                 var anchor = "", index;
170                 // x/y/z -> x/y/
171                 if ((index = moduleName.lastIndexOf("/")) != -1) {
172                     anchor = moduleName.substring(0, index + 1);
173                 }
174                 return normalizePath(anchor + depName);
175             } else if (depName.indexOf("./") != -1
176                 || depName.indexOf("../") != -1) {
177                 return normalizePath(depName);
178             } else {
179                 return depName;
180             }
181         },

188         /**
189          * 路径正则化，不能是相对地址
190          * 相对地址则转换成相对页面的绝对地址
191          * 用途:
192          * package path 相对地址则相对于当前页面获取绝对地址
193          */
194         normalBasePath:function (path) {
195             path = S.trim(path);
196 
197             // path 为空时，不能变成 "/"
198             if (path &&
199                 path.charAt(path.length - 1) != '/') {
200                 path += "/";
201             }
202 
203             /**
204              * 一定要正则化，防止出现 ../ 等相对路径
205              * 考虑本地路径
206              */
207             if (!path.match(/^(http(s)?)|(file):/i) &&
208                 !startsWith(path, "/")) {
209                 path = __pagePath + path;
210             }
211 
212             if (startsWith(path, "/")) {
213                 var loc = win.location;
214                 path = loc.protocol + "//" + loc.host + path;
215             }
216 
217             return normalizePath(path);
218         },

327         /**
328          * Three effects:
329          * 1. add index : / => /index
330          * 2. unalias : core => dom,event,ua
331          * 3. relative to absolute : ./x => y/x
332          * 4. create module info with tag : core.js?t=xx => core , tag=xx
333          * @param {KISSY} self Global KISSY instance
334          * @param {String|String[]} modNames Array of module names or module names string separated by comma
335          */
336         normalizeModNames:function (self, modNames, refModName, keepAlias) {
337             var ret = [],
338                 mods = self['Env'].mods;
339             S.each(modNames, function (name) {
340                 var alias, m;
341                 // 1. index map
342                 name = indexMap(name);
343                 // 2. un alias
344                 if (!keepAlias && (m = mods[name]) && (alias = m.alias)) {
345                     ret.push.apply(ret, indexMap(alias));
346                 } else {
347                     ret.push(name);
348                 }
349             });
350             // 3. relative to absolute (optional)
351             if (refModName) {
352                 ret = utils.normalDepModuleName(refModName, ret);
353             }
354             // 4. create module info with tag
355             S.each(ret, function (name, i) {
356                 ret[i] = utils.createModuleInfo(self, name).name;
357             });
358             return ret;
359         },
360 
361         normalizeModNamesWithAlias:function (self, modNames, refModName) {
362             return utils.normalizeModNames(self, modNames, refModName, 1);
363         },

367             var mods = self.Env.mods,
368                 mod = mods[name];

372                 return;
373             }

376 
377             mod = mods[name];
378 
379             // 注意：通过 S.add(name[, fn[, config]]) 注册的代码，无论是页面中的代码，
380             // 还是 js 文件里的代码，add 执行时，都意味着该模块已经 LOADED
381             S.mix(mod, { name:name, status:data.LOADED });
382 
383 
384             mod.fn = fn;
385 
386 
387             if (config && config.requires) {
388                 config.requires = utils.normalizeModNames(self, config.requires, name);
389             }
390 
391             S.mix((mods[name] = mod), config);
392 
393             S.log(name + " is loaded");

396         /**
397          * 只用来指定模块依赖信息. 注意：需要在 package 声明后 add ！
398          * @param self
399          * @param name
400          * @param fn
401          * @param config
402          * @example
403          * <code>
404          *
405          * KISSY.config({
406          *  packages:[
407          *      {
408          *          name:"biz1",
409          *          path:"haha"
410          *      }
411          *  ]
412          * });
413          *
414          * KISSY.add({
415          *   "biz1/main" : {
416          *      requires:[ "biz1/part1" , "biz1/part2" ]
417          *   }
418          * });
419          *
420          * </code>
421          */
422         normAdd:function (self, name, fn, config) {
423             var mods = self.Env.mods,
424                 o;
425 
426             // S.add(name, config) => S.add( { name: config } )
427             if (S.isString(name) &&
428                 !config &&
429                 S.isPlainObject(fn)) {
430                 o = {};
431                 o[name] = fn;
432                 name = o;
433             }
434 
435             // S.add( { name: config } )
436             if (S.isPlainObject(name)) {
437                 S.each(name, function (modCfg, modName) {
438                     modName = utils.indexMapStr(modName);
439                     if (modCfg.requires) {
440                         modCfg.requires =
441                             utils.normalizeModNames(self, modCfg.requires, modName);
442                     }
443                     utils.createModuleInfo(self, modName);
444                     S.mix(mods[modName], modCfg);
445                 });
446                 return true;
447             }
448         },

483     function defaultComponentJsName(m, packageInfo) {
484         var suffix = ".js",
485             match;
486         if (match = m.match(/(.+)(\.css)$/i)) {
487             suffix = match[2];
488             m = match[1];
489         }
490         var min = "-min";
491         if (packageInfo.debug) {
492             min = "";
493         }
494         return m + min + suffix;
495     }

497     function isStatus(self, modNames, status) {
498         var mods = self.Env.mods,
499             i;
500         modNames = S.makeArray(modNames);
501         for (i = 0; i < modNames.length; i++) {
502             var mod = mods[modNames[i]];
503             if (!mod || mod.status !== status) {
504                 return false;
505             }
506         }
507         return true;
508     }
