1 /**
  2  * @fileOverview simple split button ,common usecase :button + menubutton
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("button/split", function(S) {
  6 
  7     var handles = {
  8         content:function(e) {
  9             var first = this,t = e.target;
 10             first.__set("content", t.get("content"));
 11             first.__set("value", t.get("value"));
 12         },
 13         value:function(e) {
 14             var first = this,t = e.target;
 15             first.__set("value", t.get("value"));
 16         }
 17     };
 18 
 19     function Split() {
 20         Split.superclass.constructor.apply(this, arguments);
 21     }
 22 
 23     Split.ATTRS = {
 24         // 第一个组件按钮
 25         first:{},
 26         // 第二个组件
 27         second:{},
 28         // 第二个组件的见ring事件
 29         eventType:{
 30             value:"click"
 31         },
 32         eventHandler:{
 33             // 或者 value
 34             value:"content"
 35         }
 36     };
 37 
 38     S.extend(Split, S.Base, {
 39         render:function() {
 40             var self = this,
 41                 eventType = self.get("eventType"),
 42                 eventHandler = handles[self.get("eventHandler")],
 43                 first = self.get("first"),
 44                 second = self.get("second");
 45             first.__set("collapseSide", "right");
 46             second.__set("collapseSide", "left");
 47             first.render();
 48             second.render();
 49             if (eventType && eventHandler) {
 50                 second.on(eventType, eventHandler, first);
 51             }
 52         }
 53     });
 54 
 55     return Split;
 56 
 57 }, {
 58     requires:['base']
 59 });
