469         selectElement:function (element) {
470             var range,
471                 self = this,
472                 doc = self.document;
473             if (OLD_IE) {
474                 //do not use empty()，编辑器内滚动条重置了
475                 //选择的 img 内容前后莫名被清除
476                 //self.getNative().empty();
477                 try {
478                     // Try to select the node as a control.
479                     range = doc.body['createControlRange']();
480                     range['addElement'](element[0]);
481                     range.select();
482                 } catch (e) {
483                     // If failed, select it as a text range.
484                     range = doc.body.createTextRange();
485                     range.moveToElementText(element[0]);
486                     range.select();
487                 } finally {
488                     //this.document.fire('selectionchange');
489                 }
490                 self.reset();
491             } else {
492                 // Create the range for the element.
493                 range = doc.createRange();
494                 range.selectNode(element[0]);
495                 // Select the range.
496                 var sel = self.getNative();
497                 sel.removeAllRanges();
498                 sel.addRange(range);
499                 self.reset();
500             }

503         selectRanges:function (ranges) {
504             var self = this;
505             if (OLD_IE) {
506                 if (ranges.length > 1) {
507                     // IE doesn't accept multiple ranges selection, so we join all into one.
508                     var last = ranges[ ranges.length - 1 ];
509                     ranges[ 0 ].setEnd(last.endContainer, last.endOffset);
510                     ranges.length = 1;
511                 }
512 
513                 // IE doesn't accept multiple ranges selection, so we just
514                 // select the first one.
515                 if (ranges[ 0 ])
516                     ranges[ 0 ].select();
517 
518                 self.reset();
519             }
520             else {
521                 var sel = self.getNative();
522                 if (!sel) {
523                     return;
524                 }
525                 sel.removeAllRanges();
526                 for (var i = 0; i < ranges.length; i++) {
527                     var range = ranges[ i ],
528                         nativeRange = self.document.createRange(),
529                         startContainer = range.startContainer;
530 
531                     // In FF2, if we have a collapsed range, inside an empty
532                     // element, we must add something to it otherwise the caret
533                     // will not be visible.
534                     // opera move out of this element
535                     if (range.collapsed &&
536                         (( UA.gecko && UA.gecko < 1.0900 ) || UA.opera || UA['webkit']) &&
537                         startContainer[0].nodeType == DOM.ELEMENT_NODE &&
538                         !startContainer[0].childNodes.length) {
539                         // webkit 光标停留不到在空元素内，要fill char，之后范围定在 fillchar 之后
540                         startContainer[0].appendChild(self.document.createTextNode(UA['webkit'] ? "\u200b" : ""));
541                         range.startOffset++;
542                         range.endOffset++;
543                     }
544 
545                     nativeRange.setStart(startContainer[0], range.startOffset);
546                     nativeRange.setEnd(range.endContainer[0], range.endOffset);
547                     // Select the range.
548                     sel.addRange(nativeRange);
549                 }
550                 self.reset();
551             }

708                         ieRangeEnd.moveToElementText(endNode[0]);
709                         // Move the end boundary of the main range to match the tool range.
710                         ieRange.setEndPoint('EndToEnd', ieRangeEnd);
711                         ieRange.moveEnd('character', -1);
712                     }
713                     else {
714                         // The isStartMarkerAlone logic comes from V2. It guarantees that the lines
715                         // will expand and that the cursor will be blinking on the right place.
716                         // Actually, we are using this flag just to avoid using this hack in all
717                         // situations, but just on those needed.
718                         var next = startNode[0].nextSibling;
719                         while (next && !notWhitespaces(next)) {
720                             next = next.nextSibling;
721                         }
722                         isStartMarkerAlone =
723                             (
724                                 !( next && next.nodeValue && next.nodeValue.match(fillerTextRegex) )     // already a filler there?
725                                     && ( forceExpand
726                                     ||
727                                     !startNode[0].previousSibling
728                                     ||
729                                     (
730                                         startNode[0].previousSibling &&
731                                             DOM.nodeName(startNode[0].previousSibling) == 'br'
732                                         )
733                                     )
734                                 );
735 
736                         // Append a temporary <span>﻿</span> before the selection.
737                         // This is needed to avoid IE destroying selections inside empty
738                         // inline elements, like <b></b> (#253).
739                         // It is also needed when placing the selection right after an inline
740                         // element to avoid the selection moving inside of it.
741                         dummySpan = new Node(self.document.createElement('span'));
742                         dummySpan.html('﻿');	// Zero Width No-Break Space (U+FEFF). See #1359.
743                         dummySpan.insertBefore(startNode);
744                         if (isStartMarkerAlone) {
745                             // To expand empty blocks or line spaces after <br>, we need
746                             // instead to have any char, which will be later deleted using the
747                             // selection.
748                             // \ufeff = Zero Width No-Break Space (U+FEFF). (#1359)
749                             DOM.insertBefore(self.document.createTextNode('\ufeff'), startNode[0] || startNode);
750                         }
751                     }
752 
753                     // Remove the markers (reset the position, because of the changes in the DOM tree).
754                     self.setStartBefore(startNode);
755                     startNode._4e_remove();
756 
757                     if (collapsed) {
758                         if (isStartMarkerAlone) {
759                             // Move the selection start to include the temporary \ufeff.
760                             ieRange.moveStart('character', -1);
761                             ieRange.select();
762                             // Remove our temporary stuff.
763                             self.document.selection.clear();
764                         } else
765                             ieRange.select();
766                         if (dummySpan) {
767                             self.moveToPosition(dummySpan, KER.POSITION_BEFORE_START);
768                             dummySpan._4e_remove();
769                         }
770                     }
771                     else {
772                         self.setEndBefore(endNode);
773                         endNode._4e_remove();
774                         ieRange.select();
775                     }
776                     // this.document.fire('selectionchange');
777                 };

782         return ( !sel || sel.isInvalid ) ? NULL : sel;
783     }
