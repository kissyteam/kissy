1 /**
  2  * @fileOverview positionable and not focusable menu
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/popupmenu", function (S, Component, Menu, PopupMenuRender) {
  6 
  7     function getParentMenu(self) {
  8         var subMenuItem = self.get("parent"),
  9             parentMenu;
 10         if (subMenuItem && subMenuItem.get("menu") == self) {
 11             parentMenu = subMenuItem.get("parent");
 12         }
 13         return parentMenu;
 14     }
 15 
 16     function getAutoHideParentMenu(self) {
 17         var parentMenu = getParentMenu(self);
 18         if (parentMenu && parentMenu.get(autoHideOnMouseLeave)) {
 19             return parentMenu;
 20         }
 21         return 0;
 22     }
 23 
 24     var autoHideOnMouseLeave = "autoHideOnMouseLeave";
 25 
 26     function clearOwn(self) {
 27         var l;
 28         if (l = self._leaveHideTimer) {
 29             clearTimeout(l);
 30             self._leaveHideTimer = 0;
 31         }
 32     }
 33 
 34 
 35     var UIBase=Component.UIBase;
 36 
 37     /**
 38      * Popup Menu
 39      * @name PopupMenu
 40      * @memberOf Menu
 41      * @constructor
 42      * @extends Menu
 43      * @extends Component.UIBase.Position
 44      * @extends Component.UIBase.Align
 45      */
 46     var PopupMenu = Component.define(Menu, [
 47         UIBase.ContentBox,
 48         UIBase.Position,
 49         UIBase.Align
 50     ],
 51         /**
 52          * @lends Menu.PopupMenu#
 53          */
 54         {
 55             _clearLeaveHideTimers:function () {
 56                 var self = this, i, item, menu;
 57                 if (!self.get(autoHideOnMouseLeave)) {
 58                     return;
 59                 }
 60                 // 清除自身
 61                 clearOwn(self);
 62                 var cs = self.get("children");
 63                 for (i = 0; i < cs.length; i++) {
 64                     item = cs[i];
 65                     // 递归清除子菜单
 66                     if ((menu = item.get("menu")) &&
 67                         // 不是懒加载函数
 68                         !S.isFunction(menu) &&
 69                         menu.get(autoHideOnMouseLeave)) {
 70                         menu._clearLeaveHideTimers();
 71                     }
 72                 }
 73             },
 74 
 75             /**
 76              * Handle mouseleave event.Make parent subMenu item unHighlighted.
 77              * Protected, should only be overridden by subclasses.
 78              * @protected
 79              * @override
 80              */
 81             handleMouseLeave:function () {
 82                 var self = this,
 83                     parent;
 84                 if (!self.get(autoHideOnMouseLeave)) {
 85                     return;
 86                 }
 87                 self._leaveHideTimer = setTimeout(function () {
 88                     self.hide();
 89                     var subMenuItem = self.get("parent"), m;
 90                     if (subMenuItem) {
 91                         m = getParentMenu(self);
 92                         // 过段时间仍然是父亲 submenu 仍然是他的兄弟中高亮，证明已经离开
 93                         // 否则
 94                         // 1.鼠标移到 submenu 的话，submenu mouseenter clearTimers,
 95                         //   这个 timer 执行不了！
 96                         //
 97                         // 2.鼠标移到了 submenu 并列的其他 menuitem，
 98                         //   导致其他 menuitem highlighted
 99                         //   那么 当前所属 submenu unhighlighted
100                         //   执行 clearTimers ，这个 timer 仍然不执行
101 
102                         // 那么只剩下一种情况，移除了整个嵌套的 menu，
103                         // 那么执行该 timer
104                         // menu hide 并且将其所属的 submenu 高亮去掉！
105                         if (m && m.get("highlightedItem") === subMenuItem) {
106                             m.set("highlightedItem", null);
107                         }
108                     }
109                 }, self.get("autoHideDelay"));
110                 parent = getAutoHideParentMenu(self);
111                 if (parent) {
112                     parent.handleMouseLeave();
113                 }
114             },
115 
116             /**
117              * Handle mouseenter event.Make parent subMenu item highlighted.
118              * Protected, should only be overridden by subclasses.
119              * @protected
120              * @override
121              */
122             handleMouseEnter:function () {
123                 var self = this;
124                 if (!self.get(autoHideOnMouseLeave)) {
125                     return;
126                 }
127                 var parent = getAutoHideParentMenu(self);
128                 if (parent) {
129                     parent.handleMouseEnter();
130                 }
131                 self._clearLeaveHideTimers();
132             },
133 
134 
135             /**
136              * Suppose it has focus (as a context menu), then it must hide when lose focus.
137              * Protected, should only be overridden by subclasses.
138              * @protected
139              * @override
140              */
141             handleBlur:function () {
142                 var self = this;
143                 PopupMenu.superclass.handleBlur.apply(self, arguments);
144                 self.hide();
145             }
146         }, {
147             ATTRS:/**
148              * @lends Menu.PopupMenu#
149              */
150             {
151                 // 弹出菜单一般不可聚焦，焦点在使它弹出的元素上
152                 /**
153                  * Whether the popup menu is focuable.
154                  * Default : false.
155                  * @type Boolean
156                  */
157                 focusable:{
158                     value:false
159                 },
160                 visibleMode:{
161                     value:"visibility"
162                 },
163                 /**
164                  * Whether the popup menu hides when mouseleave.
165                  * Default : false.
166                  * @type Boolean
167                  */
168                 autoHideOnMouseLeave:{},
169                 /**
170                  * After how much time the popup menu hides when mouseleave.
171                  * Unit : second.
172                  * Default : 0.1.
173                  * @type Number
174                  */
175                 autoHideDelay:{
176                     value:0.1
177                 }
178             },
179             DefaultRender:PopupMenuRender
180         }, "Menu_PopupMenu");
181 
182     Component.UIStore.setUIConstructorByCssClass("popupmenu", {
183         priority:Component.UIStore.PRIORITY.LEVEL2,
184         ui:PopupMenu
185     });
186 
187     return PopupMenu;
188 
189 }, {
190     requires:['component', './base', './popupmenuRender']
191 });
