167     function findAWithImg(img) {
168         var ret = img.parent();
169         while (ret) {
170             var name = ret.nodeName();
171             if (name == "a") {
172                 return ret;
173             }
174             if (dtd.$block[name] || dtd.$blockLimit[name]) {
175                 return null;
176             }
177             ret = ret.parent();
178         }
179         return null;
180     }

425         _insert:function () {

517         _update:function (_selectedEl) {
518             var self = this,

521             self.selectedEl = _selectedEl;
522             if (self.selectedEl && self.imageCfg['remote'] !== false) {

524                 var w = self.selectedEl.width(),
525                     h = self.selectedEl.height();
526                 valInput(self.imgHeight, h);
527                 valInput(self.imgWidth, w);
528                 self.imgAlign.set("value", self.selectedEl.style("float") || "none");
529                 var margin = parseInt(self.selectedEl.style("margin"))
530                     || 0;
531                 self.imgMargin.val(margin);
532                 self.imgRatio[0].disabled = false;
533                 self.imgRatioValue = w / h;
534                 var link = findAWithImg(self.selectedEl);
535                 if (link) {
536                     valInput(self.imgLink, link.attr("_ke_saved_href") || link.attr("href"));
537                     self.imgLinkBlank.attr("checked", link.attr("target") == "_blank");
538                 } else {
539                     resetInput(self.imgLink);
540                     self.imgLinkBlank.attr("checked", true);
541                 }
542             } else {
543                 var defaultMargin = self.imageCfg['defaultMargin'];
544                 if (defaultMargin == undefined) {
545                     defaultMargin = MARGIN_DEFAULT;
546                 }
547                 if (self.tab['panels'].length == 2) {
548                     active = 1;
549                 }
550                 self.imgLinkBlank.attr("checked", true);
551                 resetInput(self.imgUrl);
552                 resetInput(self.imgLink);
553                 resetInput(self.imgHeight);
554                 resetInput(self.imgWidth);
555                 self.imgAlign.set("value", "none");
556                 self.imgMargin.val(defaultMargin);
557                 self.imgRatio[0].disabled = true;
558                 self.imgRatioValue = null;
559             }
560             self.uploadForm[0].reset();
561             self.imgLocalUrl.val(warning);
562             self.tab['switchTo'](active);
563         },

573         }
