1 /**
  2  * @fileOverview decorate its children from one element
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/decoratechild", function (S, DecorateChildren) {
  6     function DecorateChild() {
  7 
  8     }
  9 
 10     S.augment(DecorateChild, DecorateChildren, {
 11         decorateInternal:function (element) {
 12             var self = this;
 13             // 不用 __set , 通知 view 更新
 14             self.set("el", element);
 15             var ui = self.get("decorateChildCls"),
 16                 prefixCls = self.get("prefixCls"),
 17                 child = element.one("." + self.getCls(ui));
 18             // 可以装饰?
 19             if (child) {
 20                 var UI = self._findUIByClass(child);
 21                 if (UI) {
 22                     // 可以直接装饰
 23                     self.decorateChildrenInternal(UI, child, prefixCls);
 24                 } else {
 25                     // 装饰其子节点集合
 26                     self.decorateChildren(child);
 27                 }
 28             }
 29         }
 30     });
 31 
 32     return DecorateChild;
 33 }, {
 34     requires:['./decoratechildren']
 35 });
