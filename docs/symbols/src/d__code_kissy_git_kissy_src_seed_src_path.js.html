  6 (function (S) {

  8     /**
  9      * @namespace
 10      * Path Utils For KISSY from nodejs
 11      * @name Path
 12      * @memberOf KISSY
 13      */
 14     var Path = {},
 15     // [root, dir, basename, ext]
 16         splitPathRe = /^(\/?)([\s\S]+\/(?!$)|\/)?((?:\.{1,2}$|[\s\S]+?)?(\.[^.\/]*)?)$/;
 17 
 18     KISSY.Path = Path;
 19 
 20     /**
 21      * Remove .. and . in path array
 22      * @param parts
 23      * @param allowAboveRoot
 24      * @return {*}
 25      */
 26     function normalizeArray(parts, allowAboveRoot) {
 27         // level above root
 28         var up = 0;
 29         for (var i = parts.length - 1; i >= 0; i--) {
 30             var last = parts[i];
 31             if (last == ".") {
 32                 parts.splice(i, 1);
 33             } else if (last === "..") {
 34                 parts.splice(i, 1);
 35                 up++;
 36             } else if (up) {
 37                 parts.splice(i, 1);
 38                 up--;
 39             }
 40         }
 41 
 42         // if allow above root, has to add ..
 43         if (allowAboveRoot) {
 44             for (; up--; up) {
 45                 parts.unshift("..");
 46             }
 47         }
 48 
 49         return parts;
 50     }

 52     S.mix(Path,
 53         /**
 54          * @lends KISSY.Path
 55          */
 56         {
 57 
 58             /**
 59              * resolve([from ...], to)
 60              * @return {String} Resolved path.
 61              */
 62             resolve:function () {
 63 
 64                 var resolvedPath = "",
 65                     resolvedPathStr,
 66                     i,
 67                     args = S.makeArray(arguments),
 68                     path,
 69                     absolute = 0;
 70 
 71                 for (i = args.length - 1; i >= 0 && !absolute; i--) {
 72                     path = args[i];
 73                     if (typeof path != "string" || !path) {
 74                         continue;
 75                     }
 76                     resolvedPath = path + "/" + resolvedPath;
 77                     absolute = path.charAt(0) == "/";
 78                 }
 79 
 80                 resolvedPathStr = normalizeArray(S.filter(resolvedPath.split("/"), function (p) {
 81                     return !!p;
 82                 }), !absolute).join("/");
 83 
 84                 return (absolute ? "/" : "" + resolvedPathStr) || ".";
 85             },
 86 
 87             /**
 88              * normalize .. and . in path
 89              * @param {String} path Path tobe normalized
 90              * @example
 91              * <code>
 92              * "x/y/../z" => "x/z"
 93              * "x/y/z/../" => "x/y/"
 94              * </code>
 95              * @return {String}
 96              */
 97             normalize:function (path) {
 98                 var absolute = path.charAt(0) == "/",
 99                     trailingSlash = path.slice(-1) == "/";
100 
101                 path = normalizeArray(S.filter(path.split("/"), function (p) {
102                     return !!p;
103                 }), !absolute).join("/");
104 
105                 if (!path && !absolute) {
106                     path = ".";
107                 }
108 
109                 if (path && trailingSlash) {
110                     path += "/";
111                 }
112 
113 
114                 return (absolute ? "/" : "") + path;
115             },
116 
117             /**
118              * join([path ...]) and normalize
119              * @return {String}
120              */
121             join:function () {
122                 var args = S.makeArray(arguments);
123                 return Path.normalize(S.filter(args,function (p) {
124                     return p && (typeof p == "string");
125                 }).join("/"));
126             },
127 
128             /**
129              * Get string which is to relative to from
130              * @param {String} from
131              * @param {String} to
132              * @example
133              * <code>
134              * relative("x/","x/y/z") => "y/z"
135              * relative("x/t/z","x/") => "../../"
136              * </code>
137              * @return {String}
138              */
139             relative:function (from, to) {
140                 from = Path.resolve(from);
141                 to = Path.resolve(to);
142 
143                 var fromParts = S.filter(from.split("/"), function (p) {
144                         return !!p;
145                     }),
146                     path = [],
147                     sameIndex,
148                     sameIndex2,
149                     toParts = S.filter(to.split("/"), function (p) {
150                         return !!p;
151                     }), commonLength = Math.min(fromParts.length, toParts.length);
152 
153                 for (sameIndex = 0; sameIndex < commonLength; sameIndex++) {
154                     if (fromParts[sameIndex] != toParts[sameIndex]) {
155                         break;
156                     }
157                 }
158 
159                 sameIndex2 = sameIndex;
160 
161                 while (sameIndex < fromParts.length) {
162                     path.push("..");
163                     sameIndex++;
164                 }
165 
166                 path = path.concat(toParts.slice(sameIndex2));
167 
168                 path = path.join("/");
169 
170                 return path;
171             },
172 
173             /**
174              * Get base name of path
175              * @param {String} path
176              * @param {String} [ext] ext to be stripped from result returned.
177              * @return {String}
178              */
179             basename:function (path, ext) {
180                 var result = path.match(splitPathRe) || [];
181                 result = result[3] || "";
182                 if (ext && result && result.slice(-1 * ext.length) == ext) {
183                     result = result.slice(0, -1 * ext.length);
184                 }
185                 return result;
186             },
187 
188             /**
189              * Get dirname of path
190              * @return {String}
191              */
192             dirname:function (path) {
193                 var result = path.match(splitPathRe) || [],
194                     root = result[1] || "",
195                     dir = result[2] || "";
196 
197                 if (!root && !dir) {
198                     // No dirname
199                     return '.';
200                 }
201 
202                 if (dir) {
203                     // It has a dirname, strip trailing slash
204                     dir = dir.substring(0, dir.length - 1);
205                 }
206 
207                 return root + dir;
208             },
209 
210             /**
211              * Get extension name of file in path
212              * @param {String} path
213              * @return {String}
214              */
215             extname:function (path) {
216                 return (path.match(splitPathRe) || [])[4] || "";
217             }
218 
219         });
