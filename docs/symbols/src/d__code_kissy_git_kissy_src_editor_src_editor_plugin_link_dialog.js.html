1 /**
  2  * link dialog
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/link/dialog", function (S, Editor, Overlay4E, Utils) {
  6 
  7     var Dialog = Overlay4E.Dialog,
  8         _ke_saved_href = Utils._ke_saved_href,
  9         MIDDLE = "vertical-align:middle;",
 10         bodyHtml = "<div style='padding:20px 20px 0 20px'>" +
 11             "<p>" +
 12             "<label>" +
 13             "链接网址： " +
 14             "<input " +
 15             " data-verify='^(https?://[^\\s]+)|(#.+)$' " +
 16             " data-warning='请输入合适的网址格式' " +
 17             "class='ke-link-url ke-input' " +
 18             "style='width:390px;" +
 19             MIDDLE +
 20             "'" +
 21             " />" +
 22             "</label>" +
 23             "</p>" +
 24             "<p " +
 25             "style='margin: 15px 0 10px 0px;'>" +
 26             "<label>" +
 27             "链接名称： " +
 28             "<input class='ke-link-title ke-input' style='width:100px;" +
 29             MIDDLE + "'>" +
 30             "</label> " +
 31             "<label>" +
 32             "<input " +
 33             "class='ke-link-blank' " +
 34             "style='vertical-align: middle; margin-left: 21px;' " +
 35             "type='checkbox'/>" +
 36             "   在新窗口打开链接" +
 37             "</label>" +
 38             "</p>" +
 39             "</div>",
 40         footHtml = "<div style='padding:5px 20px 20px;'>" +
 41             "<a " +
 42             "href='javascript:void(\'确定\')' " +
 43             "class='ke-link-ok ke-button' " +
 44             "style='margin-left:65px;margin-right:20px;'>确定</a> " +
 45             "<a " +
 46             "href='javascript:void(\'取消\')' " +
 47             "class='ke-link-cancel ke-button'>取消</a>" +
 48             "</div>";
 49 
 50     function LinkDialog(editor) {
 51         var self = this;
 52         self.editor = editor;
 53         Editor.Utils.lazyRun(self, "_prepareShow", "_real");
 54     }
 55 
 56     S.augment(LinkDialog, {
 57         _prepareShow:function () {
 58             var self = this,
 59                 d = new Dialog({
 60                     autoRender:true,
 61                     width:500,
 62                     headerContent:"链接",
 63                     bodyContent:bodyHtml,
 64                     footerContent:footHtml,
 65                     mask:true
 66                 });
 67             self.dialog = d;
 68             var body = d.get("body"),
 69                 foot = d.get("footer");
 70             d.urlEl = body.one(".ke-link-url");
 71             d.urlTitle = body.one(".ke-link-title");
 72             d.targetEl = body.one(".ke-link-blank");
 73             var cancel = foot.one(".ke-link-cancel"),
 74                 ok = foot.one(".ke-link-ok");
 75             ok.on("click", self._link, self);
 76             cancel.on("click", function (ev) {
 77                 ev && ev.halt();
 78                 d.hide();
 79             });
 80             Editor.Utils.placeholder(d.urlEl, "http://");
 81         },
 82 
 83         _link:function (ev) {
 84             ev.halt();
 85             var self = this,
 86                 d = self.dialog,
 87                 url = d.urlEl.val();
 88             if (!Editor.Utils.verifyInputs(d.get("el").all("input"))) {
 89                 return;
 90             }
 91             d.hide();
 92             var attr = {
 93                 href:url,
 94                 target:d.targetEl[0].checked ? "_blank" : "_self",
 95                 title:S.trim(d.urlTitle.val())
 96             };
 97             // ie9 focus 不同步，hide后等会才能恢复焦点
 98             setTimeout(function () {
 99                 Utils.applyLink(self.editor, attr, self._selectedEl);
100             }, 0);
101         },
102 
103         _real:function () {
104             var self = this,
105                 cfg = self.editor.get("pluginConfig")["link"] || {},
106                 d = self.dialog,
107                 _selectedEl = self._selectedEl;
108             //是修改行为
109             if (_selectedEl) {
110                 var url = _selectedEl.attr(_ke_saved_href) || _selectedEl.attr("href");
111                 Editor.Utils.valInput(d.urlEl, url);
112                 d.urlTitle.val(_selectedEl.attr("title") || "");
113                 d.targetEl[0].checked = (_selectedEl.attr("target") == "_blank");
114             } else {
115                 Editor.Utils.resetInput(d.urlEl);
116                 d.urlTitle.val("");
117                 if (cfg.target) {
118                     d.targetEl[0].checked = true;
119                 }
120             }
121             d.show();
122         },
123         show:function (_selectedEl) {
124             var self = this;
125             self._selectedEl = _selectedEl;
126             self._prepareShow();
127         }
128     });
129     return LinkDialog;
130 }, {
131     requires:['editor', '../overlay/', './utils']
132 });
