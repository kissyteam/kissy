1 /**
  2  * link editor support for kissy editor ,innovation from google doc and ckeditor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/link/index", function (S, Editor, Bubble, Utils, DialogLoader) {
  6 
  7     var $ = S.all,
  8         tipHtml = '<a ' +
  9             'href="" '
 10             + ' target="_blank" ' +
 11             'class="ks-editor-bubble-url">' +
 12             '在新窗口查看' +
 13             '</a>  –  '
 14             + ' <span ' +
 15             'class="ks-editor-bubble-link ks-editor-bubble-change">' +
 16             '编辑' +
 17             '</span>   |   '
 18             + ' <span ' +
 19             'class="ks-editor-bubble-link ks-editor-bubble-remove">' +
 20             '去除' +
 21             '</span>';
 22 
 23     function checkLink(lastElement) {
 24         lastElement = $(lastElement);
 25         return lastElement.closest('a', undefined);
 26     }
 27 
 28     function LinkPlugin() {
 29 
 30     }
 31 
 32     S.augment(LinkPlugin, {
 33         renderUI:function (editor) {
 34             editor.addButton("link", {
 35                 tooltip:"插入链接",
 36                 listeners:{
 37                     click:function () {
 38                         showLinkEditDialog();
 39 
 40                     }
 41                 },
 42                 mode:Editor.WYSIWYG_MODE
 43             });
 44 
 45             function showLinkEditDialog(selectedEl) {
 46                 DialogLoader.useDialog(editor, "link", selectedEl);
 47             }
 48 
 49             editor.addBubble("link", checkLink, {
 50                 listeners:{
 51                     afterRenderUI:function () {
 52                         var bubble = this,
 53                             el = bubble.get("contentEl");
 54 
 55                         el.html(tipHtml);
 56 
 57                         var tipUrl = el.one(".ks-editor-bubble-url"),
 58                             tipChange = el.one(".ks-editor-bubble-change"),
 59                             tipRemove = el.one(".ks-editor-bubble-remove");
 60 
 61                         //ie focus not lose
 62                         Editor.Utils.preventFocus(el);
 63 
 64                         tipChange.on("click", function (ev) {
 65                             showLinkEditDialog(bubble.get("editorSelectedEl"));
 66                             ev.halt();
 67                         });
 68 
 69                         tipRemove.on("click", function (ev) {
 70                             Utils.removeLink(editor, bubble.get("editorSelectedEl"));
 71                             ev.halt();
 72                         });
 73 
 74                         bubble.on("show", function () {
 75                             var a = bubble.get("editorSelectedEl");
 76                             if (!a) {
 77                                 return;
 78                             }
 79                             var href = a.attr(Utils._ke_saved_href) ||
 80                                 a.attr("href");
 81                             tipUrl.html(href);
 82                             tipUrl.attr("href", href);
 83                         });
 84                     }
 85 
 86                 }
 87             });
 88         }
 89     });
 90 
 91     return LinkPlugin;
 92 }, {
 93     requires:['editor', '../bubble/',
 94         './utils', '../dialogLoader/', '../button/']
 95 });
