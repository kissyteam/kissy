1 /**
  2  * @fileOverview drag extension for position
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/drag", function (S) {
  6 
  7     /**
  8      * Drag extension class.
  9      * Make element draggable.
 10      * @class
 11      * @memberOf UIBase
 12      */
 13     function Drag() {
 14     }
 15 
 16     Drag.ATTRS =
 17 
 18     /**
 19      * @lends UIBase.Drag
 20      */
 21     {
 22         /**
 23          * Current draggable element's handlers.
 24          * See {@link DD.Draggable#handlers}
 25          */
 26         handlers:{
 27             value:[]
 28         },
 29         /**
 30          * Whether current element is draggable.
 31          * @type Boolean
 32          */
 33         draggable:{value:true}
 34     };
 35 
 36     function dragExtAction(ev) {
 37         this.set("xy", [ev.left, ev.top]);
 38     }
 39 
 40     Drag.prototype = {
 41 
 42         _uiSetHandlers:function (v) {
 43             var d;
 44             if (v && v.length > 0 && (d = this.__drag)) {
 45                 d.set("handlers", v);
 46             }
 47         },
 48 
 49         __bindUI:function () {
 50             var DD = S.require("dd") || {},
 51                 Draggable = DD.Draggable,
 52                 d,
 53                 self = this,
 54                 dragCfg = self.get("draggable"),
 55                 el = self.get("el");
 56             if (dragCfg && Draggable) {
 57                 if (dragCfg === true) {
 58                     dragCfg = {};
 59                 }
 60                 d = self.__drag = new Draggable({
 61                     node:el,
 62                     move:dragCfg.proxy
 63                 });
 64 
 65                 if (dragCfg.proxy) {
 66                     dragCfg.proxy.moveOnEnd = false;
 67                     d.on("dragend", function () {
 68                         var proxyOffset = p.get("proxyNode").offset();
 69                         el.css("visibility", "");
 70                         self.set("x", proxyOffset.left);
 71                         self.set("y", proxyOffset.top);
 72                     });
 73                     var p = self.__proxy = new DD.Proxy(dragCfg.proxy);
 74                     p.attachDrag(d);
 75                 } else {
 76                     d.on("drag", dragExtAction, self);
 77                 }
 78 
 79                 if (dragCfg.scroll) {
 80                     var s = self.__scroll = new DD.Scroll(dragCfg.scroll);
 81                     s.attachDrag(d);
 82                 }
 83 
 84             }
 85         },
 86 
 87         _uiSetDraggable:function (v) {
 88             var d = this.__drag;
 89             d && d.set("disabled", !v);
 90         },
 91 
 92         __destructor:function () {
 93             var self = this,
 94                 p = self.__proxy,
 95                 s = self.__scroll,
 96                 d = self.__drag;
 97             d && d.destroy();
 98             s && s.destroy();
 99             p && p.destroy();
100         }
101 
102     };
103     return Drag;
104 
105 });
