  8     function wrapperViewSetter(attrName) {
  9         return function (ev) {
 10             var value = ev.newVal,
 11                 self = this,
 12                 view = self.get("view");
 13             view && view.set(attrName, value);
 14         };
 15     }
 16 
 17     function wrapperViewGetter(attrName) {
 18         return function (v) {
 19             var self = this,
 20                 view = self.get("view");
 21             return v === undefined ? view && view.get(attrName) : v;
 22         };
 23     }

 26     function initChild(self, c, elBefore) {
 27         // If this (parent) component doesn't have a DOM yet, call createDom now
 28         // to make sure we render the child component's element into the correct
 29         // parent element (otherwise render_ with a null first argument would
 30         // render the child into the document body, which is almost certainly not
 31         // what we want).
 32         self.create();
 33         var contentEl = self.getContentElement();
 34         c.__set("parent", self);
 35         // set 通知 view 也更新对应属性
 36         c.set("render", contentEl);
 37         c.set("elBefore", elBefore);
 38         // 如果 parent 已经渲染好了子组件也要立即渲染，就 创建 dom ，绑定事件
 39         if (self.get("rendered")) {
 40             c.render();
 41         }
 42         // 如果 parent 也没渲染，子组件 create 出来和 parent 节点关联
 43         // 子组件和 parent 组件一起渲染
 44         else {
 45             // 之前设好属性，view ，logic 同步还没 bind ,create 不是 render ，还没有 bindUI
 46             c.create();
 47             // 设置好，render 时插入到对应位置，这里不需要了
 48             // contentEl[0].insertBefore(c.get("el")[0], elBefore && elBefore[0] || null);
 49         }
 50     }

 52     /**
 53      * 不使用 valueFn
 54      * 只有 render 时需要找到默认，其他时候不需要，防止莫名其妙初始化
 55      */
 56     function getDefaultView() {
 57         // 逐层找默认渲染器
 58         var self = this,
 59             c = self.constructor,
 60             DefaultRender;
 61 
 62         while (c && !DefaultRender) {
 63             DefaultRender = c['DefaultRender'];
 64             c = c.superclass && c.superclass.constructor;
 65         }
 66         if (DefaultRender) {
 67             /**
 68              * 将渲染层初始化所需要的属性，直接构造器设置过去
 69              */
 70             var attrs = self['__attrs'] || {},
 71                 cfg = {};
 72             for (var attrName in attrs) {
 73                 if (attrs.hasOwnProperty(attrName)) {
 74                     var attrCfg = attrs[attrName], v;
 75                     if (attrCfg.view) {
 76                         // 只设置用户设置的值
 77                         // 考虑 c 上的默认值
 78                         if (( v = self.get(attrName) ) !== undefined) {
 79                             cfg[attrName] = v;
 80                         }
 81                     }
 82                 }
 83             }
 84             return new DefaultRender(cfg);
 85         }
 86         return 0;
 87     }

 89     function getClsByHierarchy(self) {
 90         if (self.__componentClasses) {
 91             return self.__componentClasses;
 92         }
 93         var constructor = self.constructor, re = [];
 94         while (constructor && constructor != ModelControl) {
 95             var cls = UIStore.getClsByUI(constructor);
 96             if (cls) {
 97                 re.push(cls);
 98             }
 99             constructor = constructor.superclass && constructor.superclass.constructor;
100         }
101         return self.__componentClasses = re.join(" ");
102     }

126                 var self = this,

141                     }

145             /**
146              * control 层的渲染 ui 就是 render view
147              * finally，不能被 override

149             renderUI:function () {
150                 var self = this, i, child;
151                 self.get("view").render();
152                 //then render my children
153                 var children = self.get("children");
154                 for (i = 0; i < children.length; i++) {
155                     child = children[i];
156                     // 不在 Base 初始化设置属性时运行，防止和其他初始化属性冲突
157                     initChild(self, child);
158                     child.render();
159                 }

162             /**
163              * 控制层的 createDom 实际上就是调用 view 层的 create 来创建真正的节点
164              */
165             createDom:function () {
166                 var self = this;
167                 var view = self.get("view") || getDefaultView.call(self);
168                 if (!view) {
169                     S.error("no view for");
170                     S.error(self.constructor);
171                     return;
172                 }
173                 view.create();
174                 view._renderCls(getClsByHierarchy(self));
175                 if (!self.get("allowTextSelection_")) {
176                     view.get("el").unselectable();
177                 }
178                 self.__set("view", view);
179             },

239             _uiSetHandleMouseEvents:function (v) {
240                 var self = this,
241                     el = self.get("el");
242                 if (v) {
243                     el.on("mouseenter", self._handleMouseEnter, self);
244                     el.on("mouseleave", self._handleMouseLeave, self);
245                     el.on("mousedown", self._handleMouseDown, self);
246                     el.on("mouseup", self._handleMouseUp, self);
247                     el.on("dblclick", self._handleDblClick, self);
248                 } else {
249                     el.detach("mouseenter", self._handleMouseEnter, self);
250                     el.detach("mouseleave", self._handleMouseLeave, self);
251                     el.detach("mousedown", self._handleMouseDown, self);
252                     el.detach("mouseup", self._handleMouseUp, self);
253                     el.detach("dblclick", self._handleDblClick, self);
254                 }

264             _isMouseEventWithinElement:function (e, elem) {
265                 var relatedTarget = e.relatedTarget;
266                 if (!relatedTarget) {
267                     return false;
268                 }
269                 // 在里面或等于自身都不算 mouseenter/leave
270                 if (relatedTarget === elem[0]
271                     || elem.contains(relatedTarget)) {
272                     return true;
273                 }
274             },

276             _handleMouseOver:function (e) {
277                 var self = this,
278                     el = self.get("el");
279                 if (self.get("disabled")) {
280                     return true;
281                 }
282                 if (!self._isMouseEventWithinElement(e, el)) {
283                     self._handleMouseEnter(e);
284                 }
285             },

289                 var self = this,
290                     el = self.get("el");
291                 if (self.get("disabled")) {
292                     return true;
293                 }
294                 if (!self._isMouseEventWithinElement(e, el)) {
295                     self._handleMouseLeave(e);
296                 }

320                     return true;
321                 }
322                 if (1 > 2) {

329             /**
330              * root element handler for mouse down
331              * @param ev
332              */
333             _handleMouseDown:function (ev) {
334                 var self = this, el;
335                 if (self.get("disabled")) {
336                     return true;
337                 }
338                 if (ev.which == 1 && self.get("activeable")) {
339                     self.set("active", true);
340                 }
341                 el = self.getKeyEventTarget();
342                 // 左键，否则 unselectable 在 ie 下鼠标点击获得不到焦点
343                 if (ev.which == 1 && el.attr("tabindex") >= 0) {
344                     el[0].focus();
345                 }
346                 // Cancel the default action unless the control
347                 // allows text selection.
348                 if (ev.which == 1 && !self.get("allowTextSelection_")) {
349                     // firefox 不会引起焦点转移
350                     ev.preventDefault();
351                 }
352             },

354             /**
355              * whether component can receive focus
356              */
357             _uiSetFocusable:function (v) {
358                 var self = this,
359                     el = self.getKeyEventTarget();
360                 if (v) {
361                     el.on("focus", self._handleFocus, self);
362                     el.on("blur", self._handleBlur, self);
363                     el.on("keydown", self._handleKeydown, self);
364                 } else {
365                     el.detach("focus", self._handleFocus, self);
366                     el.detach("blur", self._handleBlur, self);
367                     el.detach("keydown", self._handleKeydown, self);
368                 }
369             },
