1 /**
  2  * Maximize plugin
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/maximize/index", function (S, Editor, maximizeCmd) {
  6     var MAXIMIZE_CLASS = "ke-toolbar-maximize",
  7         RESTORE_CLASS = "ke-toolbar-restore",
  8         MAXIMIZE_TIP = "全屏",
  9         RESTORE_TIP = "取消全屏";
 10 
 11     return {
 12         init:function (editor) {
 13             maximizeCmd.init(editor);
 14             editor.addButton({
 15                 title:MAXIMIZE_TIP,
 16                 contentCls:MAXIMIZE_CLASS
 17             }, {
 18                 onClick:function () {
 19                     var self = this;
 20                     editor.execCommand("restoreWindow");
 21                     self.boff();
 22                     self.set("title", MAXIMIZE_TIP);
 23                     self.set("contentCls", MAXIMIZE_CLASS);
 24                 },
 25                 offClick:function () {
 26                     var self = this;
 27                     editor.execCommand("maximizeWindow");
 28                     self.bon();
 29                     self.set("title", RESTORE_TIP);
 30                     self.set("contentCls", RESTORE_CLASS);
 31                 }
 32             });
 33         }
 34     };
 35 }, {
 36     requires:['editor', './cmd']
 37 });
