 21     function iterate(rtl, breakOnFalseRetFalse) {
 22         var self = this;
 23         // Return NULL if we have reached the end.
 24         if (self._.end) {
 25             return NULL;
 26         }
 27         var node,
 28             range = self.range,
 29             guard,
 30             userGuard = self.guard,
 31             type = self.type,
 32             getSourceNodeFn = ( rtl ? '_4e_previousSourceNode' :
 33                 '_4e_nextSourceNode' );
 34 
 35         // This is the first call. Initialize it.
 36         if (!self._.start) {
 37             self._.start = 1;
 38 
 39             // Trim text nodes and optimize the range boundaries. DOM changes
 40             // may happen at this point.
 41             range.trim();
 42 
 43             // A collapsed range must return NULL at first call.
 44             if (range.collapsed) {
 45                 self.end();
 46                 return NULL;
 47             }
 48         }
 49 
 50         // Create the LTR guard function, if necessary.
 51         if (!rtl && !self._.guardLTR) {
 52             // Gets the node that stops the walker when going LTR.
 53             var limitLTR = range.endContainer[0],
 54                 blockerLTR = limitLTR.childNodes[range.endOffset];
 55             // 从左到右保证在 range 区间内获取 nextSourceNode
 56             this._.guardLTR = function (node, movingOut) {
 57                 // 从endContainer移出去，失败返回false
 58                 if (movingOut && (limitLTR == node || DOM.nodeName(node) == "body")) {
 59                     return false;
 60                 }
 61                 // 达到边界的下一个节点,注意 null 的情况
 62                 // node 永远不能为 null
 63                 return node != blockerLTR;
 64 
 65             };
 66         }
 67 
 68         // Create the RTL guard function, if necessary.
 69         if (rtl && !self._.guardRTL) {
 70             // Gets the node that stops the walker when going LTR.
 71             var limitRTL = range.startContainer[0],
 72                 blockerRTL = ( range.startOffset > 0 ) && limitRTL.childNodes[range.startOffset - 1] || null;
 73 
 74             self._.guardRTL = function (node, movingOut) {
 75                 // 从endContainer移出去，失败返回false
 76                 if (movingOut && (limitRTL == node || DOM.nodeName(node) == "body")) {
 77                     return false;
 78                 }
 79                 // 达到边界的下一个节点,注意 null 的情况
 80                 // node 永远不能为 null
 81                 return node != blockerRTL;
 82             };
 83         }
 84 
 85         // Define which guard function to use.
 86         var stopGuard = rtl ? self._.guardRTL : self._.guardLTR;
 87 
 88         // Make the user defined guard function participate in the process,
 89         // otherwise simply use the boundary guard.
 90         if (userGuard) {
 91             guard = function (node, movingOut) {
 92                 if (stopGuard(node, movingOut) === FALSE) {
 93                     return FALSE;
 94                 }
 95                 return userGuard(node, movingOut);
 96             };
 97         }
 98         else {
 99             guard = stopGuard;
100         }
101 
102         if (self.current) {
103             node = this.current[ getSourceNodeFn ](FALSE, type, guard);
104         } else {
105             // Get the first node to be returned.
106 
107             if (rtl) {
108                 node = range.endContainer;
109                 if (range.endOffset > 0) {
110                     node = new Node(node[0].childNodes[range.endOffset - 1]);
111                     if (guard(node) === FALSE) {
112                         node = NULL;
113                     }
114                 } else {
115                     node = ( guard(node, TRUE) === FALSE ) ?
116                         NULL : node._4e_previousSourceNode(TRUE, type, guard, undefined);
117                 }
118             }
119             else {
120                 node = range.startContainer;
121                 node = new Node(node[0].childNodes[range.startOffset]);
122 
123                 if (node.length) {
124                     if (guard(node) === FALSE) {
125                         node = NULL;
126                     }
127                 } else {
128                     node = ( guard(range.startContainer, TRUE) === FALSE ) ?
129                         NULL : range.startContainer._4e_nextSourceNode(TRUE, type, guard, undefined);
130                 }
131             }
132         }
133 
134         while (node && !self._.end) {
135             self.current = node;
136             if (!self.evaluator || self.evaluator(node[0]) !== FALSE) {
137                 if (!breakOnFalseRetFalse) {
138                     return node;
139                 }
140             } else if (breakOnFalseRetFalse && self.evaluator) {
141                 return FALSE;
142             }
143             node = node[ getSourceNodeFn ](FALSE, type, guard);
144         }
145 
146         self.end();
147         return self.current = NULL;
148     }

150     function iterateToLast(rtl) {
151         var node,
152             last = NULL;
153         while (node = iterate.call(this, rtl)) {
154             last = node;
155         }
156         return last;
157     }

159     /**
160      * Walker for DOM.
161      * @param {Editor.Range} range
162      * @class
163      * @memberOf Editor
164      */
165     function Walker(range) {
166         this.range = range;
167 
168         /**
169          * A function executed for every matched node, to check whether
170          * it's to be considered into the walk or not. If not provided, all
171          * matched nodes are considered good.
172          * If the function returns "FALSE" the node is ignored.
173          * @type Function
174          * @memberOf Editor.Walker#
175          */
176         this.evaluator = NULL;// 当前 range 范围内深度遍历的元素调用
177 
178         /**
179          * A function executed for every node the walk pass by to check
180          * whether the walk is to be finished. It's called when both
181          * entering and exiting nodes, as well as for the matched nodes.
182          * If this function returns "FALSE", the walking ends and no more
183          * nodes are evaluated.
184          * @type Function
185          * @memberOf Editor.Walker#
186          */
187         this.guard = NULL;// 人为缩小当前 range 范围
188 
189 
190         /** @private */
191         this._ = {};
192     }

195     S.augment(Walker,
196         /**
197          * @lends Editor.Walker#
198          */
199         {
200             /**
201              * Stop walking. No more nodes are retrieved if this function gets
202              * called.
203              */
204             end:function () {
205                 this._.end = 1;
206             },
207 
208             /**
209              * Retrieves the next node (at right).
210              * @returns {Boolean} The next node or NULL if no more
211              *        nodes are available.
212              */
213             next:function () {
214                 return iterate.call(this);
215             },
216 
217             /**
218              * Retrieves the previous node (at left).
219              * @returns {Boolean} The previous node or NULL if no more
220              *        nodes are available.
221              */
222             previous:function () {
223                 return iterate.call(this, TRUE);
224             },
225 
226             /**
227              * Check all nodes at right, executing the evaluation function.
228              * @returns {Boolean} "FALSE" if the evaluator function returned
229              *        "FALSE" for any of the matched nodes. Otherwise "TRUE".
230              */
231             checkForward:function () {
232                 return iterate.call(this, FALSE, TRUE) !== FALSE;
233             },
234 
235             /**
236              * Check all nodes at left, executing the evaluation function.
237              * 是不是 (不能后退了)
238              * @returns {Boolean} "FALSE" if the evaluator function returned
239              *        "FALSE" for any of the matched nodes. Otherwise "TRUE".
240              */
241             checkBackward:function () {
242                 // 在当前 range 范围内不会出现 evaluator 返回 false 的情况
243                 return iterate.call(this, TRUE, TRUE) !== FALSE;
244             },
245 
246             /**
247              * Executes a full walk forward (to the right), until no more nodes
248              * are available, returning the last valid node.
249              * @returns {Boolean} The last node at the right or NULL
250              *        if no valid nodes are available.
251              */
252             lastForward:function () {
253                 return iterateToLast.call(this);
254             },
255 
256             /**
257              * Executes a full walk backwards (to the left), until no more nodes
258              * are available, returning the last valid node.
259              * @returns {Boolean} The last node at the left or NULL
260              *        if no valid nodes are available.
261              */
262             lastBackward:function () {
263                 return iterateToLast.call(this, TRUE);
264             },
265 
266             reset:function () {
267                 delete this.current;
268                 this._ = {};
269             },
270 
271             // for unit test
272             _iterator:iterate
273 
274         });

277     S.mix(Walker,
278         /**
279          * @lends Editor.Walker
280          */
281         {
282             /**
283              * Whether the to-be-evaluated node is not a block node and does not match given node name map.
284              * @param {Object} customNodeNames Given node name map.
285              * @return {Function} Function for evaluation.
286              */
287             blockBoundary:function (customNodeNames) {
288                 return function (node) {
289                     return !(node.nodeType == DOM.ELEMENT_NODE &&
290                         DOM._4e_isBlockBoundary(node, customNodeNames) );
291                 };
292             },
293 
294             /**
295              * Whether the to-be-evaluated node is a bookmark node OR bookmark node
296              * inner contents.
297              * @param {Boolean} [contentOnly] Whether only test againt the text content of
298              * bookmark node instead of the element itself(default).
299              * @param {Boolean} [isReject] Whether should return 'FALSE' for the bookmark
300              * node instead of 'TRUE'(default).
301              * @return {Function} Function for evaluation.
302              */
303             bookmark:function (contentOnly, isReject) {
304                 function isBookmarkNode(node) {
305                     return  DOM.nodeName(node) == 'span' &&
306                         DOM.attr(node, '_ke_bookmark');
307                 }
308 
309                 return function (node) {
310                     var isBookmark, parent;
311                     // Is bookmark inner text node?
312                     isBookmark = ( node.nodeType == DOM.TEXT_NODE &&
313                         ( parent = node.parentNode ) &&
314                         isBookmarkNode(parent) );
315                     // Is bookmark node?
316                     isBookmark = contentOnly ? isBookmark : isBookmark || isBookmarkNode(node);
317                     // !! 2012-05-15
318                     // evaluator check ===false, must turn it to boolean false
319                     return !!(isReject ^ isBookmark);
320                 };
321             },
322 
323             /**
324              * Whether the node is a text node containing only whitespaces characters.
325              * @param {Boolean} [isReject]
326              */
327             whitespaces:function (isReject) {
328                 return function (node) {
329                     var isWhitespace = node.nodeType == DOM.TEXT_NODE &&
330                         !S.trim(node.nodeValue);
331                     return !!(isReject ^ isWhitespace);
332                 };
333             },
334             /**
335              * Whether the node is invisible in wysiwyg mode.
336              * @param isReject
337              */
338             invisible:function (isReject) {
339                 var whitespace = Walker.whitespaces();
340                 return function (node) {
341                     // Nodes that take no spaces in wysiwyg:
342                     // 1. White-spaces but not including NBSP;
343                     // 2. Empty inline elements, e.g. <b></b> we're checking here
344                     // 'offsetHeight' instead of 'offsetWidth' for properly excluding
345                     // all sorts of empty paragraph, e.g. <br />.
346                     var isInvisible = whitespace(node) ||
347                         node.nodeType == DOM.ELEMENT_NODE && !node.offsetHeight;
348                     return !!(isReject ^ isInvisible);
349                 };
350             }
351         });

353     var tailNbspRegex = /^[\t\r\n ]*(?: |\xa0)$/,
354         isWhitespaces = Walker.whitespaces(),
355         isBookmark = Walker.bookmark(),
356         toSkip = function (node) {
357             var name = DOM.nodeName(node);
358             return isBookmark(node) ||
359                 isWhitespaces(node) ||
360                 node.nodeType == 1 &&
361                     name in dtd.$inline &&
362                     !( name in dtd.$empty );
363         };

365     // Check if there's a filler node at the end of an element, and return it.
366     function getBogus(tail) {
367         // Bogus are not always at the end, e.g. <p><a>text<br /></a></p>
368         do {
369             tail = tail._4e_previousSourceNode();
370         } while (tail && toSkip(tail[0]));
371 
372         if (tail && ( !UA.ie ? tail.nodeName() == "br"
373             : tail[0].nodeType == 3 && tailNbspRegex.test(tail.text()) )) {
374             return tail[0];
375         }
376         return false;
377     }

380         _4e_getBogus:function (el) {
381             return getBogus(new Node(el));
382         }
