  5 KISSY.add('node/attach', function (S, DOM, Event, NodeList, undefined) {
  6 
  7     var NLP = NodeList.prototype,
  8         makeArray = S.makeArray,
  9         // DOM 添加到 NP 上的方法
 10         // if DOM methods return undefined , Node methods need to transform result to itself
 11         DOM_INCLUDES_NORM = [
 12             "equals",
 13             "contains",
 14             "scrollTop",
 15             "scrollLeft",
 16             "height",
 17             "width",
 18             "innerHeight",
 19             "innerWidth",
 20             "outerHeight",
 21             "outerWidth",
 22             "addStyleSheet",
 23             // "append" will be overridden
 24             "appendTo",
 25             // "prepend" will be overridden
 26             "prependTo",
 27             "insertBefore",
 28             "before",
 29             "after",
 30             "insertAfter",
 31             "test",
 32             "hasClass",
 33             "addClass",
 34             "removeClass",
 35             "replaceClass",
 36             "toggleClass",
 37             "removeAttr",
 38             "hasAttr",
 39             "hasProp",
 40             // anim override
 41 //            "show",
 42 //            "hide",
 43 //            "toggle",
 44             "scrollIntoView",
 45             "remove",
 46             "empty",
 47             "removeData",
 48             "hasData",
 49             "unselectable",
 50 
 51             "wrap",
 52             "wrapAll",
 53             "replaceWith",
 54             "wrapInner",
 55             "unwrap"
 56         ],
 57         // if return array ,need transform to nodelist
 58         DOM_INCLUDES_NORM_NODE_LIST = [
 59             "filter",
 60             "first",
 61             "last",
 62             "parent",
 63             "closest",
 64             "next",
 65             "prev",
 66             "clone",
 67             "siblings",
 68             "contents",
 69             "children"
 70         ],
 71         // if set return this else if get return true value ,no nodelist transform
 72         DOM_INCLUDES_NORM_IF = {
 73             // dom method : set parameter index
 74             attr:1,
 75             text:0,
 76             css:1,
 77             style:1,
 78             val:0,
 79             prop:1,
 80             offset:0,
 81             html:0,
 82             data:1
 83         },
 84         // Event 添加到 NP 上的方法
 85         EVENT_INCLUDES = [
 86             "on",
 87             "detach",
 88             "fire",
 89             "fireHandler",
 90             "delegate",
 91             "undelegate"
 92         ];
 93 
 94 
 95     function accessNorm(fn, self, args) {
 96         args.unshift(self);
 97         var ret = DOM[fn].apply(DOM, args);
 98         if (ret === undefined) {
 99             return self;
100         }
101         return ret;
102     }
103 
104     function accessNormList(fn, self, args) {
105         args.unshift(self);
106         var ret = DOM[fn].apply(DOM, args);
107         if (ret === undefined) {
108             return self;
109         }
110         else if (ret === null) {
111             return null;
112         }
113         return new NodeList(ret);
114     }
115 
116     function accessNormIf(fn, self, index, args) {
117 
118         // get
119         if (args[index] === undefined
120             // 并且第一个参数不是对象，否则可能是批量设置写
121             && !S.isObject(args[0])) {
122             args.unshift(self);
123             return DOM[fn].apply(DOM, args);
124         }
125         // set
126         return accessNorm(fn, self, args);
127     }
128 
129     S.each(DOM_INCLUDES_NORM, function (k) {
130         NLP[k] = function () {
131             var args = makeArray(arguments);
132             return accessNorm(k, this, args);
133         };
134     });
135 
136     S.each(DOM_INCLUDES_NORM_NODE_LIST, function (k) {
137         NLP[k] = function () {
138             var args = makeArray(arguments);
139             return accessNormList(k, this, args);
140         };
141     });
142 
143     S.each(DOM_INCLUDES_NORM_IF, function (index, k) {
144         NLP[k] = function () {
145             var args = makeArray(arguments);
146             return accessNormIf(k, this, index, args);
147         };
148     });
149 
150     S.each(EVENT_INCLUDES, function (k) {
151         NLP[k] = function () {
152             var self = this,
153                 args = makeArray(arguments);
154             args.unshift(self);
155             Event[k].apply(Event, args);
156             return self;
157         }
158     });
159 
160 }, {
161     requires:["dom", "event", "./base"]
162 });
