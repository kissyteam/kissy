  9 KISSY.add("editor/core/domIterator", function (S) {
 10     var TRUE = true,
 11         FALSE = false,
 12         NULL = null,
 13         Editor = S.Editor,
 14         UA = S.UA,
 15         Walker = Editor.Walker,
 16         KERange = Editor.Range,
 17         KER = Editor.RANGE,
 18         KEN = Editor.NODE,
 19         ElementPath = Editor.ElementPath,
 20         Node = S.Node,
 21         DOM = S.DOM;
 22 
 23     /**
 24      * @constructor
 25      * @param range {KISSY.Editor.Range}
 26      */
 27     function Iterator(range) {
 28         if (arguments.length < 1)
 29             return;
 30         var self = this;
 31         self.range = range;
 32         self.forceBrBreak = FALSE;
 33 
 34         // Whether include <br>s into the enlarged range.(#3730).
 35         self.enlargeBr = TRUE;
 36         self.enforceRealBlocks = FALSE;
 37 
 38         self._ || ( self._ = {} );
 39     }
 40 
 41     var beginWhitespaceRegex = /^[\r\n\t ]*$/;///^[\r\n\t ]+$/,//+:*??不匹配空串
 42 
 43     S.augment(Iterator, {
 44         //奇怪点：
 45         //<ul>
 46         // <li>
 47         // x
 48         // </li>
 49         // <li>
 50         // y
 51         // </li>
 52         // </ul>
 53         //会返回两次 li,li,而不是一次 ul ，
 54         // 可能只是返回包含文字的段落概念？
 55         getNextParagraph:function (blockTag) {
 56             // The block element to be returned.
 57             var block, self = this;
 58 
 59             // The range object used to identify the paragraph contents.
 60             var range;
 61 
 62             // Indicats that the current element in the loop is the last one.
 63             var isLast;
 64 
 65             // Instructs to cleanup remaining BRs.
 66             var removePreviousBr, removeLastBr;
 67 
 68             // self is the first iteration. Let's initialize it.
 69             if (!self._.lastNode) {
 70                 range = self.range.clone();
 71 
 72                 // 2010-09-30 shrink
 73                 // 3.4.2 新增，
 74                 // Shrink the range to exclude harmful "noises" (#4087, #4450, #5435).
 75                 range.shrink(KER.SHRINK_ELEMENT, TRUE);
 76 
 77                 range.enlarge(self.forceBrBreak || !self.enlargeBr ?
 78                     KER.ENLARGE_LIST_ITEM_CONTENTS : KER.ENLARGE_BLOCK_CONTENTS);
 79 
 80                 var walker = new Walker(range),
 81                     ignoreBookmarkTextEvaluator = Walker.bookmark(TRUE, TRUE);
 82                 // Avoid anchor inside bookmark inner text.
 83                 walker.evaluator = ignoreBookmarkTextEvaluator;
 84                 self._.nextNode = walker.next();
 85                 // TODO: It's better to have walker.reset() used here.
 86                 walker = new Walker(range);
 87                 walker.evaluator = ignoreBookmarkTextEvaluator;
 88                 var lastNode = walker.previous();
 89                 self._.lastNode = lastNode._4e_nextSourceNode(TRUE);
 90 
 91                 // We may have an empty text node at the end of block due to [3770].
 92                 // If that node is the lastNode, it would cause our logic to leak to the
 93                 // next block.(#3887)
 94                 if (self._.lastNode &&
 95                     self._.lastNode[0].nodeType == KEN.NODE_TEXT &&
 96                     !S.trim(self._.lastNode[0].nodeValue) &&
 97                     self._.lastNode.parent()._4e_isBlockBoundary()) {
 98                     var testRange = new KERange(range.document);
 99                     testRange.moveToPosition(self._.lastNode, KER.POSITION_AFTER_END);
100                     if (testRange.checkEndOfBlock()) {
101                         var path = new ElementPath(testRange.endContainer);
102                         var lastBlock = path.block || path.blockLimit;
103                         self._.lastNode = lastBlock._4e_nextSourceNode(TRUE);
104                     }
105                 }
106 
107                 // Probably the document end is reached, we need a marker node.
108                 if (!self._.lastNode) {
109                     self._.lastNode = self._.docEndMarker = new Node(range.document.createTextNode(''));
110                     DOM.insertAfter(self._.lastNode[0], lastNode[0]);
111                 }
112 
113                 // Let's reuse self variable.
114                 range = NULL;
115             }
116 
117             var currentNode = self._.nextNode;
118             lastNode = self._.lastNode;
119 
120             self._.nextNode = NULL;
121             while (currentNode) {
122                 // closeRange indicates that a paragraph boundary has been found,
123                 // so the range can be closed.
124                 var closeRange = FALSE;
125 
126                 // includeNode indicates that the current node is good to be part
127                 // of the range. By default, any non-element node is ok for it.
128                 var includeNode = ( currentNode[0].nodeType != KEN.NODE_ELEMENT ),
129                     continueFromSibling = FALSE;
130 
131                 // If it is an element node, let's check if it can be part of the
132                 // range.
133                 if (!includeNode) {
134                     var nodeName = currentNode.nodeName();
135 
136                     if (currentNode._4e_isBlockBoundary(self.forceBrBreak && { br:1 })) {
137                         // <br> boundaries must be part of the range. It will
138                         // happen only if ForceBrBreak.
139                         if (nodeName == 'br')
140                             includeNode = TRUE;
141                         else if (!range && !currentNode[0].childNodes.length && nodeName != 'hr') {
142                             // If we have found an empty block, and haven't started
143                             // the range yet, it means we must return self block.
144                             block = currentNode;
145                             isLast = currentNode.equals(lastNode);
146                             break;
147                         }
148 
149                         // The range must finish right before the boundary,
150                         // including possibly skipped empty spaces. (#1603)
151                         if (range) {
152                             range.setEndAt(currentNode, KER.POSITION_BEFORE_START);
153 
154                             // The found boundary must be set as the next one at self
155                             // point. (#1717)
156                             if (nodeName != 'br')
157                                 self._.nextNode = currentNode;
158                         }
159 
160                         closeRange = TRUE;
161                     } else {
162                         // If we have child nodes, let's check them.
163                         if (currentNode[0].firstChild) {
164                             // If we don't have a range yet, let's start it.
165                             if (!range) {
166                                 range = new KERange(self.range.document);
167                                 range.setStartAt(currentNode, KER.POSITION_BEFORE_START);
168                             }
169 
170                             currentNode = new Node(currentNode[0].firstChild);
171                             continue;
172                         }
173                         includeNode = TRUE;
174                     }
175                 }
176                 else if (currentNode[0].nodeType == KEN.NODE_TEXT) {
177                     // Ignore normal whitespaces (i.e. not including   or
178                     // other unicode whitespaces) before/after a block node.
179                     if (beginWhitespaceRegex.test(currentNode[0].nodeValue))
180                         includeNode = FALSE;
181                 }
182 
183                 // The current node is good to be part of the range and we are
184                 // starting a new range, initialize it first.
185                 if (includeNode && !range) {
186                     range = new KERange(self.range.document);
187                     range.setStartAt(currentNode, KER.POSITION_BEFORE_START);
188                 }
189 
190                 // The last node has been found.
191                 isLast = ( !closeRange || includeNode ) && currentNode.equals(lastNode);
192 
193                 // If we are in an element boundary, let's check if it is time
194                 // to close the range, otherwise we include the parent within it.
195                 if (range && !closeRange) {
196                     while (!currentNode[0].nextSibling && !isLast) {
197                         var parentNode = currentNode.parent();
198 
199                         if (parentNode._4e_isBlockBoundary(self.forceBrBreak && { br:1 })) {
200                             closeRange = TRUE;
201                             isLast = isLast || parentNode.equals(lastNode);
202                             break;
203                         }
204 
205                         currentNode = parentNode;
206                         includeNode = TRUE;
207                         isLast = currentNode.equals(lastNode);
208                         continueFromSibling = TRUE;
209                     }
210                 }
211 
212                 // Now finally include the node.
213                 if (includeNode)
214                     range.setEndAt(currentNode, KER.POSITION_AFTER_END);
215 
216                 currentNode = currentNode._4e_nextSourceNode(continueFromSibling, NULL, lastNode);
217                 isLast = !currentNode;
218 
219                 // We have found a block boundary. Let's close the range and move out of the
220                 // loop.
221                 if (isLast || ( closeRange && range ))
222                     break;
223             }
224 
225             // Now, based on the processed range, look for (or create) the block to be returned.
226             if (!block) {
227                 // If no range has been found, self is the end.
228                 if (!range) {
229                     self._.docEndMarker && self._.docEndMarker._4e_remove();
230                     self._.nextNode = NULL;
231                     return NULL;
232                 }
233 
234                 var startPath = new ElementPath(range.startContainer);
235                 var startBlockLimit = startPath.blockLimit,
236                     checkLimits = { div:1, th:1, td:1 };
237                 block = startPath.block;
238 
239                 if ((!block || !block[0])
240                     && !self.enforceRealBlocks
241                     && checkLimits[ startBlockLimit.nodeName() ]
242                     && range.checkStartOfBlock()
243                     && range.checkEndOfBlock())
244                     block = startBlockLimit;
245                 else if (!block || ( self.enforceRealBlocks && block.nodeName() == 'li' )) {
246                     // Create the fixed block.
247                     block = new Node(self.range.document.createElement(blockTag || 'p'));
248                     // Move the contents of the temporary range to the fixed block.
249                     block[0].appendChild(range.extractContents());
250                     block._4e_trim();
251                     // Insert the fixed block into the DOM.
252                     range.insertNode(block);
253                     removePreviousBr = removeLastBr = TRUE;
254                 }
255                 else if (block.nodeName() != 'li') {
256                     // If the range doesn't includes the entire contents of the
257                     // block, we must split it, isolating the range in a dedicated
258                     // block.
259                     if (!range.checkStartOfBlock() || !range.checkEndOfBlock()) {
260                         // The resulting block will be a clone of the current one.
261                         block = block.clone(FALSE);
262 
263                         // Extract the range contents, moving it to the new block.
264                         block[0].appendChild(range.extractContents());
265                         block._4e_trim();
266 
267                         // Split the block. At self point, the range will be in the
268                         // right position for our intents.
269                         var splitInfo = range.splitBlock();
270 
271                         removePreviousBr = !splitInfo.wasStartOfBlock;
272                         removeLastBr = !splitInfo.wasEndOfBlock;
273 
274                         // Insert the new block into the DOM.
275                         range.insertNode(block);
276                     }
277                 }
278                 else if (!isLast) {
279                     // LIs are returned as is, with all their children (due to the
280                     // nested lists). But, the next node is the node right after
281                     // the current range, which could be an <li> child (nested
282                     // lists) or the next sibling <li>.
283 
284                     self._.nextNode = ( block.equals(lastNode) ? NULL :
285                         range.getBoundaryNodes().endNode._4e_nextSourceNode(TRUE, NULL, lastNode) );
286                 }
287             }
288 
289             if (removePreviousBr) {
290                 var previousSibling = new Node(block[0].previousSibling);
291                 if (previousSibling[0] && previousSibling[0].nodeType == KEN.NODE_ELEMENT) {
292                     if (previousSibling.nodeName() == 'br')
293                         previousSibling._4e_remove();
294                     else if (previousSibling[0].lastChild && DOM.nodeName(previousSibling[0].lastChild) == 'br')
295                         DOM._4e_remove(previousSibling[0].lastChild);
296                 }
297             }
298 
299             if (removeLastBr) {
300                 // Ignore bookmark nodes.(#3783)
301                 var bookmarkGuard = Walker.bookmark(FALSE, TRUE);
302 
303                 var lastChild = new Node(block[0].lastChild);
304                 if (lastChild[0] && lastChild[0].nodeType == KEN.NODE_ELEMENT && lastChild.nodeName() == 'br') {
305                     // Take care not to remove the block expanding <br> in non-IE browsers.
306                     if (UA['ie']
307                         || lastChild.prev(bookmarkGuard)
308                         || lastChild.next(bookmarkGuard))
309                         lastChild.remove();
310                 }
311             }
312 
313             // Get a reference for the next element. self is important because the
314             // above block can be removed or changed, so we can rely on it for the
315             // next interation.
316             if (!self._.nextNode) {
317                 self._.nextNode = ( isLast || block.equals(lastNode) ) ? NULL :
318                     block._4e_nextSourceNode(TRUE, NULL, lastNode);
319             }
320 
321             return block;
322         }
323     });
324 
325     KERange.prototype.createIterator = function () {
326         return new Iterator(this);
327     };
328 }, {
329     requires:['./base', './range', './elementPath', './walker']
330 });
