 75                 function stylesFilter(styles, whitelist) {
 76                     return function (styleText, element) {
 77                         var rules = [];
 78                         // html-encoded quote might be introduced by 'font-family'
 79                         // from MS-Word which confused the following regexp. e.g.
 80                         //'font-family: "Lucida, Console"'
 81                         String(styleText)
 82                             .replace(/"/g, '"')
 83                             .replace(/\s*([^ :;]+)\s*:\s*([^;]+)\s*(?=;|$)/g,
 84                             function (match, name, value) {
 85                                 name = name.toLowerCase();
 86                                 name == 'font-family' && ( value = value.replace(/["']/g, '') );
 87 
 88                                 var namePattern,
 89                                     valuePattern,
 90                                     newValue,
 91                                     newName;
 92                                 for (var i = 0; i < styles.length; i++) {
 93                                     if (styles[ i ]) {
 94                                         namePattern = styles[ i ][ 0 ];
 95                                         valuePattern = styles[ i ][ 1 ];
 96                                         newValue = styles[ i ][ 2 ];
 97                                         newName = styles[ i ][ 3 ];
 98 
 99                                         if (name.match(namePattern) &&
100                                             ( !valuePattern || value.match(valuePattern) )) {
101                                             name = newName || name;
102                                             whitelist && ( newValue = newValue || value );
103 
104                                             if (typeof newValue == 'function')
105                                                 newValue = newValue(value, element, name);
106 
107                                             // Return an couple indicate both name and value
108                                             // changed.
109                                             if (newValue && newValue.push) {
110                                                 name = newValue[ 0 ];
111                                                 newValue = newValue[ 1 ];
112                                             }
113 
114                                             if (typeof newValue == 'string')
115                                                 rules.push([ name, newValue ]);
116                                             return;
117                                         }
118                                     }
119                                 }
120 
121                                 !whitelist && rules.push([ name, value ]);
122 
123                             });
124 
125                         for (var i = 0; i < rules.length; i++)
126                             rules[ i ] = rules[ i ].join(':');
127 
128                         return rules.length ?
129                             ( rules.join(';') + ';' ) : false;
130                     };
131                 }

338                 // Return the last non-space child node of the block (#4344).
339                 function lastNoneSpaceChild(block) {
340                     var childNodes = block.childNodes,
341                         lastIndex = childNodes.length,
342                         last = childNodes[ lastIndex - 1 ];

348                 function trimFillers(block, fromSource) {
349                     // If the current node is a block, and if we're converting from source or
350                     // we're not in IE then search for and remove any tailing BR node.
351                     // Also, any   at the end of blocks are fillers, remove them as well.
352                     // (#2886)
353                     var lastChild = lastNoneSpaceChild(block);
354                     if (lastChild) {
355                         if (( fromSource || !UA['ie'] ) &&
356                             lastChild.nodeType == 1 &&
357                             lastChild.nodeName == 'br') {
358                             block.removeChild(lastChild);
359                         }
360                         else if (lastChild.nodeType == 3 &&
361                             tailNbspRegex.test(lastChild.nodeValue)) {
362                             block.removeChild(lastChild);
363                         }
364                     }
365                 }

367                 function blockNeedsExtension(block) {

370                     return !lastChild
371                         || lastChild.nodeType == 1 &&
372                         lastChild.nodeName == 'br'

374                         // to move cursor at the end of the form. (#4791)
375                         || block.nodeName == 'form' &&
376                         lastChild.nodeName == 'input';

450             var protectElementRegex = /<(a|area|img|input)\b([^>]*)>/gi,
451                 protectAttributeRegex = /\b(href|src|name)\s*=\s*(?:(?:"[^"]*")|(?:'[^']*')|(?:[^ "'>]+))/gi;
452             // ie 6-7 会将 关于 url 的 content value 替换为 dom value
453             // #a -> http://xxx/#a
454             // ../x.html -> http://xx/x.html
455             function protectAttributes(html) {
456                 return html.replace(protectElementRegex, function (element, tag, attributes) {
457                     return '<' + tag + attributes.replace(protectAttributeRegex, function (fullAttr, attrName) {
458                         // We should not rewrite the existed protected attributes,
459                         // e.g. clipboard content from editor. (#5218)
460                         if (attributes.indexOf('_ke_saved_' + attrName) == -1)
461                             return ' _ke_saved_' + fullAttr + ' ' + fullAttr;
462 
463                         return fullAttr;
464                     }) + '>';
465                 });
466             }

470             function protectRealComments(html) {
471                 return html.replace(/<!--(?!{ke_protected})[\s\S]+?-->/g, function (match) {
472                     return '<!--' + protectedSourceMarker +
473                         '{C}' +
474                         encodeURIComponent(match).replace(/--/g, '%2D%2D') +
475                         '-->';
476                 });
477             }
478 
479             function unprotectRealComments(html) {
480                 return html.replace(/<!--\{ke_protected\}\{C\}([\s\S]+?)-->/g, function (match, data) {
481                     return decodeURIComponent(data);
482                 });
483             }
