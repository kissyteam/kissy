  9         SRC_NODE = 'srcNode',

 13 
 14     function capitalFirst(s) {
 15         return s.charAt(0).toUpperCase() + s.substring(1);
 16     }
 17 
 18     /**
 19      * UIBase for class-based component
 20      * @class
 21      * @namespace
 22      * @name UIBase
 23      * @extends Base
 24      */
 25     function UIBase(config) {
 26         // 读取用户设置的属性值并设置到自身
 27         Base.apply(this, arguments);
 28         // 根据 srcNode 设置属性值
 29         // 按照类层次执行初始函数，主类执行 initializer 函数，扩展类执行构造器函数
 30         initHierarchy(this, config);
 31         // 是否自动渲染
 32         config && config.autoRender && this.render();
 33 
 34         /**
 35          * @name UIBase#afterRenderUI
 36          * @description fired when root node is ready
 37          * @event
 38          * @param e
 39          */
 40     }
 41 
 42     /**
 43      * 模拟多继承
 44      * init attr using constructors ATTRS meta info
 45      */
 46     function initHierarchy(host, config) {
 47 
 48         var c = host.constructor;
 49 
 50         while (c) {
 51 
 52             // 从 markup 生成相应的属性项
 53             if (config &&
 54                 config[SRC_NODE] &&
 55                 c.HTML_PARSER) {
 56                 if ((config[SRC_NODE] = Node.one(config[SRC_NODE]))) {
 57                     applyParser.call(host, config[SRC_NODE], c.HTML_PARSER);
 58                 }
 59             }
 60 
 61             c = c.superclass && c.superclass.constructor;
 62         }
 63 
 64         callMethodByHierarchy(host, "initializer", "constructor");
 65 
 66     }
 67 
 68     function callMethodByHierarchy(host, mainMethod, extMethod) {
 69         var c = host.constructor,
 70             extChains = [],
 71             ext,
 72             main,
 73             exts,
 74             t;
 75 
 76         // define
 77         while (c) {
 78 
 79             // 收集扩展类
 80             t = [];
 81             if (exts = c.__ks_exts) {
 82                 for (var i = 0; i < exts.length; i++) {
 83                     ext = exts[i];
 84                     if (ext) {
 85                         if (extMethod != "constructor") {
 86                             //只调用真正自己构造器原型的定义，继承原型链上的不要管
 87                             if (ext.prototype.hasOwnProperty(extMethod)) {
 88                                 ext = ext.prototype[extMethod];
 89                             } else {
 90                                 ext = null;
 91                             }
 92                         }
 93                         ext && t.push(ext);
 94                     }
 95                 }
 96             }
 97 
 98             // 收集主类
 99             // 只调用真正自己构造器原型的定义，继承原型链上的不要管 !important
100             // 所以不用自己在 renderUI 中调用 superclass.renderUI 了，UIBase 构造器自动搜寻
101             // 以及 initializer 等同理
102             if (c.prototype.hasOwnProperty(mainMethod) && (main = c.prototype[mainMethod])) {
103                 t.push(main);
104             }
105 
106             // 原地 reverse
107             if (t.length) {
108                 extChains.push.apply(extChains, t.reverse());
109             }
110 
111             c = c.superclass && c.superclass.constructor;
112         }
113 
114         // 初始化函数
115         // 顺序：父类的所有扩展类函数 -> 父类对应函数 -> 子类的所有扩展函数 -> 子类对应函数
116         for (i = extChains.length - 1; i >= 0; i--) {
117             extChains[i] && extChains[i].call(host);
118         }
119     }
120 
121     /**
122      * 销毁组件
123      * 顺序： 子类 destructor -> 子类扩展 destructor -> 父类 destructor -> 父类扩展 destructor
124      */
125     function destroyHierarchy(host) {
126         var c = host.constructor,
127             exts,
128             d,
129             i;
130 
131         while (c) {
132             // 只触发该类真正的析构器，和父亲没关系，所以不要在子类析构器中调用 superclass
133             if (c.prototype.hasOwnProperty("destructor")) {
134                 c.prototype.destructor.apply(host);
135             }
136 
137             if ((exts = c.__ks_exts)) {
138                 for (i = exts.length - 1; i >= 0; i--) {
139                     d = exts[i] && exts[i].prototype.__destructor;
140                     d && d.apply(host);
141                 }
142             }
143 
144             c = c.superclass && c.superclass.constructor;
145         }
146     }
147 
148     function applyParser(srcNode, parser) {
149         var host = this, p, v;
150 
151         // 从 parser 中，默默设置属性，不触发事件
152         for (p in parser) {
153             if (parser.hasOwnProperty(p)) {
154                 v = parser[p];
155 
156                 // 函数
157                 if (S.isFunction(v)) {
158                     host.__set(p, v.call(host, srcNode));
159                 }
160                 // 单选选择器
161                 else if (S.isString(v)) {
162                     host.__set(p, srcNode.one(v));
163                 }
164                 // 多选选择器
165                 else if (S.isArray(v) && v[0]) {
166                     host.__set(p, srcNode.all(v[0]))
167                 }
168             }
169         }
170     }

174     UIBase.ATTRS = {
175         // 是否已经渲染完毕
176         rendered:{
177             value:false
178         },
179         // dom 节点是否已经创建完毕
180         created:{
181             value:false
182         }
183     };

185     S.extend(UIBase, Base,
186         /**
187          * @lends UIBase.prototype
188          */
189         {
190 
191             /**
192              * 建立节点，先不放在 dom 树中，为了性能!
193              */
194             create:function () {
195                 var self = this;
196                 // 是否生成过节点
197                 if (!self.get("created")) {
198                     self._createDom();
199                     self.fire('createDom');
200                     callMethodByHierarchy(self, "createDom", "__createDom");
201                     self.fire('afterCreateDom');
202                     self.__set("created", true);
203                 }
204             },
205 
206             /**
207              * 渲染组件到 dom 结构
208              */
209             render:function () {
210                 var self = this;
211                 // 是否已经渲染过
212                 if (!self.get("rendered")) {
213                     self.create();
214                     self._renderUI();
215                     // 实际上是 beforeRenderUI
216                     self.fire('renderUI');
217                     callMethodByHierarchy(self, "renderUI", "__renderUI");
218                     self.fire('afterRenderUI');
219                     self._bindUI();
220                     // 实际上是 beforeBindUI
221                     self.fire('bindUI');
222                     callMethodByHierarchy(self, "bindUI", "__bindUI");
223                     self.fire('afterBindUI');
224                     self._syncUI();
225                     // 实际上是 beforeSyncUI
226                     self.fire('syncUI');
227                     callMethodByHierarchy(self, "syncUI", "__syncUI");
228                     self.fire('afterSyncUI');
229                     self.__set("rendered", true);
230                 }
231                 return self;
232             },
233 
234             /**
235              * 创建 dom 节点，但不放在 document 中
236              */
237             _createDom:noop,
238 
239             /**
240              * 节点已经创建完毕，可以放在 document 中了
241              */
242             _renderUI:noop,
243 
244             /**
245              * @protected
246              * @function
247              */
248             renderUI:noop,
249 
250             /**
251              * 根据属性变化设置 UI
252              */
253             _bindUI:function () {
254                 var self = this,
255                     attrs = self['__attrs'],
256                     attr, m;
257 
258                 for (attr in attrs) {
259                     if (attrs.hasOwnProperty(attr)) {
260                         m = UI_SET + capitalFirst(attr);
261                         if (self[m]) {
262                             // 自动绑定事件到对应函数
263                             (function (attr, m) {
264                                 self.on('after' + capitalFirst(attr) + 'Change', function (ev) {
265                                     self[m](ev.newVal, ev);
266                                 });
267                             })(attr, m);
268                         }
269                     }
270                 }
271             },
272 
273             /**
274              * @protected
275              * @function
276              */
277             bindUI:noop,
278 
279             /**
280              * 根据当前（初始化）状态来设置 UI
281              */
282             _syncUI:function () {
283                 var self = this,
284                     v,
285                     f,
286                     attrs = self['__attrs'];
287                 for (var a in attrs) {
288                     if (attrs.hasOwnProperty(a)) {
289                         var m = UI_SET + capitalFirst(a);
290                         //存在方法，并且用户设置了初始值或者存在默认值，就同步状态
291                         if ((f = self[m])
292                             // 用户如果设置了显式不同步，就不同步，比如一些值从 html 中读取，不需要同步再次设置
293                             && attrs[a].sync !== false
294                             && (v = self.get(a)) !== undefined) {
295                             f.call(self, v);
296                         }
297                     }
298                 }
299             },
300 
301             /**
302              * protected
303              * @function
304              */
305             syncUI:noop,
306 
307 
308             /**
309              * 销毁组件
310              */
311             destroy:function () {
312                 var self = this;
313                 destroyHierarchy(self);
314                 self.fire('destroy');
315                 self.detach();
316             }
317         },
318         /**
319          * @lends UIBase
320          */
321         {
322             /**
323              * 根据基类以及扩展类得到新类
324              * @param {Function|Function[]} base 基类
325              * @param {Function[]} exts 扩展类
326              * @param {Object} px 原型 mix 对象
327              * @param {Object} sx 静态 mix 对象
328              * @returns {UIBase} 组合 后 的 新类
329              */
330             create:function (base, exts, px, sx) {
331                 if (S.isArray(base)) {
332                     sx = px;
333                     px = exts;
334                     exts = /*@type Function[]*/base;
335                     base = UIBase;
336                 }
337                 base = base || UIBase;
338                 if (S.isObject(exts)) {
339                     sx = px;
340                     px = exts;
341                     exts = [];
342                 }
343 
344                 function C() {
345                     UIBase.apply(this, arguments);
346                 }
347 
348                 S.extend(C, base, px, sx);
349 
350                 if (exts) {
351 
352                     C.__ks_exts = exts;
353 
354                     var desc = {
355                         // ATTRS:
356                         // HMTL_PARSER:
357                     }, constructors = exts.concat(C);
358 
359                     // [ex1,ex2],扩展类后面的优先，ex2 定义的覆盖 ex1 定义的
360                     // 主类最优先
361                     S.each(constructors, function (ext) {
362                         if (ext) {
363                             // 合并 ATTRS/HTML_PARSER 到主类
364                             S.each([ATTRS, HTML_PARSER], function (K) {
365                                 if (ext[K]) {
366                                     desc[K] = desc[K] || {};
367                                     // 不覆盖主类上的定义，因为继承层次上扩展类比主类层次高
368                                     // 但是值是对象的话会深度合并
369                                     // 注意：最好值是简单对象，自定义 new 出来的对象就会有问题!
370                                     S.mix(desc[K], ext[K], true, undefined, true);
371                                 }
372                             });
373                         }
374                     });
375 
376                     S.each(desc, function (v, k) {
377                         C[k] = v;
378                     });
379 
380                     var prototype = {};
381 
382                     // 主类最优先
383                     S.each(constructors, function (ext) {
384                         if (ext) {
385                             var proto = ext.prototype;
386                             // 合并功能代码到主类，不覆盖
387                             for (var p in proto) {
388                                 // 不覆盖主类，但是主类的父类还是覆盖吧
389                                 if (proto.hasOwnProperty(p)) {
390                                     prototype[p] = proto[p];
391                                 }
392                             }
393                         }
394                     });
395 
396                     S.each(prototype, function (v, k) {
397                         C.prototype[k] = v;
398                     });
399                 }
400                 return C;
401             }
402         });
403 
404     return UIBase;
