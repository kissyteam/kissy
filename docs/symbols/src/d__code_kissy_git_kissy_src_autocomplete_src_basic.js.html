1 /**
  2  * Provide basic api for AutoComplete
  3  */
  4 KISSY.add("autocomplete/basic", function (S, UIBase, AutoComplete, AutoCompleteMenu, LocalDataSource, RemoteDataSource) {
  5 
  6     /**
  7      * Provide basic api for AutoComplete Component
  8      * @class
  9      * @name Basic
 10      * @memberOf AutoComplete
 11      * @extends AutoComplete
 12      */
 13     return UIBase.create(AutoComplete, [], {
 14 
 15         __CLASS:"AutoComplete.Basic",
 16 
 17         initializer:function () {
 18             var self = this,
 19                 dataSource,
 20                 autoCompleteMenu,
 21                 data;
 22             if (!self.get("dataSource")) {
 23                 if (data = self.get("data")) {
 24                     dataSource = new LocalDataSource({
 25                         data:data,
 26                         dataSourceCfg:self.get("dataSourceCfg")
 27                     });
 28                 } else {
 29                     dataSource = new RemoteDataSource({
 30                         xhrCfg:self.get("xhrCfg"),
 31                         dataSourceCfg:self.get("dataSourceCfg")
 32                     });
 33                 }
 34                 self.__set('dataSource', dataSource);
 35             }
 36 
 37             if (!self.get("menu")) {
 38                 autoCompleteMenu = new AutoCompleteMenu({
 39                     prefixCls:self.get("prefixCls")
 40                 });
 41                 self.__set("menu", autoCompleteMenu);
 42             }
 43         }
 44     }, {
 45         ATTRS:/**
 46          * @lends AutoComplete.Basic
 47          */
 48         {
 49 
 50             /**
 51              * Whether destroy menu when this destroys.Default true
 52              * @type Boolean
 53              */
 54             destroyMenu:{
 55                 value:true
 56             },
 57 
 58             /**
 59              * Array of static data. data and xhrCfg are mutually exclusive.
 60              * @type Array
 61              */
 62             data:{},
 63             /**
 64              * xhrCfg IO configuration.same as {@link} IO. data and xhrCfg are mutually exclusive.
 65              * @type Object
 66              */
 67             xhrCfg:{
 68                 value:{}
 69             },
 70 
 71             /**
 72              * Extra config for remote dataSource.<br/>
 73              * {String} dataSourceCfg.cache :
 74              * Whether server response data is cached<br/>
 75              * {Function} dataSource.parse :
 76              * Serve as a parse function to parse server
 77              * response to return a valid array of data for autoComplete.
 78              * Used for xhrCfg.<br/>
 79              * {String} dataSourceCfg.paramName :
 80              * Used as parameter name to send autoS=Complete input's value to server<br/>
 81              * {Boolean} dataSourceCfg.allowEmpty <br/>
 82              * static data:whether return all data when input is empty.default:true<br/>
 83              * whether send empty to server when input val is empty.default:false
 84              * @type Object
 85              */
 86             dataSourceCfg:{
 87                 value:{}
 88             }
 89         }
 90     });
 91 
 92 }, {
 93     requires:['uibase', './input', './menu', './localDataSource', './remoteDataSource']
 94 });
