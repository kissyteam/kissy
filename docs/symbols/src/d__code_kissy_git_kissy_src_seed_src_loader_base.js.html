187     function defaultComponentJsName(m) {
188         var name = m.name,
189             suffix = ".js",
190             min = "-min",
191             match;
192         if (match = name.match(/(.+)(\.css)$/i)) {
193             suffix = match[2];
194             name = match[1];
195         }
196         if (m.getPackageInfo().isDebug()) {
197             min = "";
198         }
199         return name + min + suffix;
200     }

202     function getPackageInfo(self, mod) {
203         var modName = mod.name,
204             Env = self.Env,
205             packages = Env.packages || {},
206             pName = "",
207             p,
208             packageDesc;
209 
210         for (p in packages) {
211             if (packages.hasOwnProperty(p)) {
212                 if (S.startsWith(modName, p) &&
213                     p.length > pName.length) {
214                     pName = p;
215                 }
216             }
217         }
218 
219         packageDesc = packages[pName] ||
220             Env.defaultPackage ||
221             (Env.defaultPackage = new Loader.Package({SS:self}));
222 
223         return packageDesc;
224     }
