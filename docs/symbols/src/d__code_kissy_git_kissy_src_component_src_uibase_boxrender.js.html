1 /**
  2  * @fileOverview UIBase.Box
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add('component/uibase/boxrender', function (S) {
  6 
  7     var $ = S.all, doc = S.Env.host.document;
  8 
  9     function BoxRender() {
 10     }
 11 
 12     BoxRender.ATTRS = {
 13         el:{
 14             //容器元素
 15             setter:function (v) {
 16                 return $(v);
 17             }
 18         },
 19 
 20         // 构建时批量生成，不需要执行单个
 21         elCls:{
 22         },
 23 
 24         elStyle:{
 25         },
 26 
 27         width:{
 28         },
 29 
 30         height:{
 31         },
 32 
 33         elTagName:{
 34             // 生成标签名字
 35             value:"div"
 36         },
 37 
 38         elAttrs:{
 39         },
 40 
 41         content:{
 42         },
 43 
 44         elBefore:{
 45             // better named to renderBefore, too late !
 46         },
 47 
 48         render:{},
 49 
 50         visible:{
 51             value:true
 52         },
 53 
 54         visibleMode:{
 55             value:"display"
 56         },
 57         // content 设置的内容节点,默认根节点
 58         contentEl:{
 59             valueFn:function () {
 60                 return this.get("el");
 61             }
 62         }
 63     };
 64 
 65     BoxRender.HTML_PARSER = {
 66         content:function (el) {
 67             // 从 contentElCls 的标志中读取
 68             var contentElCls = this.get("contentElCls");
 69             return (contentElCls ? el.one("." + contentElCls) : el).html();
 70         }
 71     };
 72 
 73     BoxRender.prototype =
 74     /**
 75      * @lends Component.UIBase.Box.Render#
 76      */
 77     {
 78 
 79         __renderUI:function () {
 80             var self = this;
 81             // 新建的节点才需要摆放定位
 82             if (!self.get("srcNode")) {
 83                 var render = self.get("render"),
 84                     el = self.get("el"),
 85                     renderBefore = self.get("elBefore");
 86                 if (renderBefore) {
 87                     el.insertBefore(renderBefore, undefined);
 88                 } else if (render) {
 89                     el.appendTo(render, undefined);
 90                 } else {
 91                     el.appendTo(doc.body, undefined);
 92                 }
 93             }
 94         },
 95 
 96         /**
 97          * 只负责建立节点，如果是 decorate 过来的，甚至内容会丢失
 98          * 通过 render 来重建原有的内容
 99          */
100         __createDom:function () {
101             var self = this;
102             if (!self.get("srcNode")) {
103                 var el,
104                     contentEl = self.get("contentEl");
105 
106                 el = $("<" + self.get("elTagName") + ">");
107 
108                 if (contentEl) {
109                     el.append(contentEl);
110                 }
111 
112                 self.__set("el", el);
113 
114                 if (!contentEl) {
115                     // 没取到,这里设下值, uiSet 时可以 set("content")  取到
116                     self.__set("contentEl", el);
117                 }
118             }
119         },
120 
121         _uiSetElAttrs:function (attrs) {
122             this.get("el").attr(attrs);
123         },
124 
125         _uiSetElCls:function (cls) {
126             this.get("el").addClass(cls);
127         },
128 
129         _uiSetElStyle:function (style) {
130             this.get("el").css(style);
131         },
132 
133         _uiSetWidth:function (w) {
134             this.get("el").width(w);
135         },
136 
137         _uiSetHeight:function (h) {
138             var self = this;
139             self.get("el").height(h);
140         },
141 
142         _uiSetContent:function (c) {
143             var self = this, el;
144             // srcNode 时不重新渲染 content
145             // 防止内部有改变，而 content 则是老的 html 内容
146             if (self.get("srcNode") && !self.get("rendered")) {
147             } else {
148                 el = self.get("contentEl");
149                 if (typeof c == "string") {
150                     el.html(c);
151                 } else if (c) {
152                     el.empty().append(c);
153                 }
154             }
155         },
156 
157         _uiSetVisible:function (isVisible) {
158             var el = this.get("el"),
159                 visibleMode = this.get("visibleMode");
160             if (visibleMode == "visibility") {
161                 el.css("visibility", isVisible ? "visible" : "hidden");
162             } else {
163                 el.css("display", isVisible ? "" : "none");
164             }
165         },
166 
167         __destructor:function () {
168             var el = this.get("el");
169             if (el) {
170                 el.remove();
171             }
172         }
173     };
174 
175     return BoxRender;
176 }, {
177     requires:['node']
178 });
