1 /**
  2  * @fileOverview close extension for kissy dialog
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/closerender", function (S, Node) {
  6 
  7     var CLS_PREFIX = 'ext-';
  8 
  9     function Close() {
 10     }
 11 
 12     Close.ATTRS = {
 13         closable:{             // 是否需要关闭按钮
 14             value:true,
 15             sync:false
 16         },
 17         closeBtn:{
 18         }
 19     };
 20 
 21     Close.HTML_PARSER = {
 22         closeBtn:function (el) {
 23             return el.one("." + this.get("prefixCls") + CLS_PREFIX + 'close');
 24         }
 25     };
 26 
 27     Close.prototype = {
 28         _uiSetClosable:function (v) {
 29             this.get("closeBtn")[v ? "show" : "hide"]();
 30         },
 31         __createDom:function () {
 32             var self = this,
 33                 closeBtn = self.get("closeBtn"),
 34                 closable = self.get("closable"),
 35                 el = self.get("el");
 36 
 37             if (!closeBtn) {
 38                 closeBtn = new Node("<a " +
 39                     "tabindex='0' " +
 40                     (closable ? "" : "style='display:none'") +
 41                     "href='javascript:void(\"关闭\")' " +
 42                     "role='button' " +
 43                     "class='" + this.get("prefixCls") + CLS_PREFIX + "close" + "'>" +
 44                     "<span class='" +
 45                     self.get("prefixCls") + CLS_PREFIX + "close-x" +
 46                     "'>关闭<" + "/span>" +
 47                     "<" + "/a>").appendTo(el);
 48                 self.__set("closeBtn", closeBtn);
 49             } else {
 50                 closeBtn[closable ? "show" : "hide"]();
 51             }
 52         },
 53 
 54         __destructor:function () {
 55             var self = this,
 56                 closeBtn = self.get("closeBtn");
 57             closeBtn && closeBtn.detach();
 58         }
 59     };
 60 
 61     return Close;
 62 
 63 }, {
 64     requires:["node"]
 65 });
