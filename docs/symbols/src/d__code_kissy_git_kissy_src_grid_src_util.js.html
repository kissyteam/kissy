  9         CLS_MASK_MSG = CLS_MASK + '-msg';
 10     /**
 11      * This class specifies some util tools of grid
 12      * @name Util
 13      * @class
 14      * @memberOf Grid
 15      */
 16     var util =
 17     /** @lends Grid.Util */
 18     {
 19         /**
 20          * @description mask the dom element
 21          * @param {String|HTMLElement} element the element such as selector,Dom or Node will be masked
 22          * @param {String} [msg] when mask one element ,you can show some message to user
 23          * @param {String} [msgCls] when show message, you can set it's style by this css class
 24          * @example
 25          *    Grid.Util.maskElement('#domId');
 26          */
 27         maskElement:function (element, msg, msgCls) {
 28             var maskedEl = S.one(element),
 29                 maskedNode = maskedEl.getDOMNode(),
 30                 maskDiv = S.one('.' + CLS_MASK, maskedNode),
 31                 tpl = null,
 32                 msgDiv = null,
 33                 top = null,
 34                 left = null;
 35             if (!maskDiv) {
 36                 maskDiv = S.one(DOM.create('<div class="' + CLS_MASK + '"></div>')).appendTo(maskedNode);
 37                 maskedEl.addClass('x-masked-relative x-masked');
 38                 if (UA.ie === 6) {
 39                     maskDiv.height(maskedEl.height());
 40                 }
 41                 if (msg) {
 42                     tpl = ['<div class="' + CLS_MASK_MSG + '"><div>', msg, '</div></div>'].join('');
 43                     msgDiv = S.one(DOM.create(tpl)).appendTo(maskedNode);
 44                     if (msgCls) {
 45                         msgDiv.addClass(msgCls);
 46                     }
 47                     try {
 48                         top = (maskedEl.height() - msgDiv.height()) / 2;
 49                         left = (maskedEl.width() - msgDiv.width()) / 2;
 50 
 51                         msgDiv.css({ left:left, top:top });
 52                     } catch (ex) {
 53                         S.log('mask error occurred');
 54                     }
 55                 }
 56             }
 57             return maskDiv;
 58         },
 59         /**
 60          * @description unmask the dom element
 61          * @param {String|HTMLElement} element the element such as selector,Dom or Node will  unmask
 62          * @example
 63          *    S.LP.unmaskElement('#domId');
 64          */
 65         unmaskElement:function (element) {
 66             var maskedEl = S.one(element),
 67                 msgEl = maskedEl.children('.' + CLS_MASK_MSG),
 68                 maskDiv = maskedEl.children('.' + CLS_MASK);
 69             if (msgEl) {
 70                 msgEl.remove();
 71             }
 72             if (maskDiv) {
 73                 maskDiv.remove();
 74             }
 75             maskedEl.removeClass('x-masked-relative x-masked');
 76 
 77         }
 78     };

 80     function formatTimeUnit(v) {
 81         if (v < 10) {
 82             return '0' + v;
 83         }
 84         return v;
 85     }
 86 
 87     /**
 88      * This class specifies some formatter for grid's cell renderer
 89      * @name Format
 90      * @class
 91      * @memberOf Grid.Util
 92      */
 93     util.Format =
 94     /** @lends Grid.Util.Format */
 95     {
 96         /**
 97          @description 日期格式化函数
 98          @param {Number|Date} d 格式话的日期，一般为1970 年 1 月 1 日至今的毫秒数
 99          @return {String} 格式化后的日期格式为 2011-10-31
100          @example
101          * 一般用法：<br>
102          * S.LP.Format.dateRenderer(1320049890544);输出：2011-10-31 <br>
103          * 表格中用于渲染列：<br>
104          * {title:"出库日期",dataIndex:"date",renderer:Grid.Util.Format.dateRenderer}
105          */
106         dateRenderer:function (d) {
107             if (!d) {
108                 return '';
109             }
110             if (S.isString(d)) {
111                 return d;
112             }
113             var date = null;
114             try {
115                 date = new Date(d);
116             } catch (e) {
117                 return '';
118             }
119             if (!date || !date.getFullYear) {
120                 return '';
121             }
122             return date.getFullYear() + '-' + formatTimeUnit(date.getMonth() + 1) + '-' + formatTimeUnit(date.getDate());//S.Date.format(d,'yyyy-mm-dd');
123         },
124         /**
125          @description 日期时间格式化函数
126          @param {Number|Date} d 格式话的日期，一般为1970 年 1 月 1 日至今的毫秒数
127          @return {String} 格式化后的日期格式时间为 2011-10-31 16 : 41 : 02
128          */
129         datetimeRenderer:function (d) {
130             if (!d) {
131                 return '';
132             }
133             if (S.isString(d)) {
134                 return d;
135             }
136             var date = null;
137             try {
138                 date = new Date(d);
139             } catch (e) {
140                 return '';
141             }
142             if (!date || !date.getFullYear) {
143                 return '';
144             }
145             return date.getFullYear() + '-' + formatTimeUnit(date.getMonth() + 1) + '-' + formatTimeUnit(date.getDate()) + ' ' + formatTimeUnit(date.getHours()) + ':' + formatTimeUnit(date.getMinutes()) + ':' + formatTimeUnit(date.getSeconds());
146         },
147         /**
148          @description 文本截取函数，当文本超出一定数字时，会截取文本，添加...
149          @param {Number} length 截取多少字符
150          @return {function} 返回处理函数 返回截取后的字符串，如果本身小于指定的数字，返回原字符串。如果大于，则返回截断后的字符串，并附加...
151          */
152         cutTextRenderer:function (length) {
153             return function (value) {
154                 value = value || '';
155                 if (value.toString().length > length) {
156                     return value.toString().substring(0, length) + '...';
157                 }
158                 return value;
159             };
160         },
161         /**
162          * @description 枚举格式化函数
163          * @param {Object} enumObj 键值对的枚举对象 {"1":"大","2":"小"}
164          * @return {Function} 返回指定枚举的格式化函数
165          * @example
166          * //Grid 的列定义
167          *  {title:"状态",dataIndex:"status",renderer:Grid.Util.Format.enumRenderer({"1":"入库","2":"出库"})}
168          */
169         enumRenderer:function (enumObj) {
170             return function (value) {
171                 return enumObj[value] || '';
172             };
173         },
174         /*
175          * @description 将多个值转换成一个字符串
176          * @param {Object} enumObj 键值对的枚举对象 {"1":"大","2":"小"}
177          * @return {Function} 返回指定枚举的格式化函数
178          * @example
179          * //Grid 的列定义
180          *  {title:"状态",dataIndex:"status",renderer:Grid.Util.Format.multipleItemsRenderer({"1":"入库","2":"出库","3":"退货"})}
181          *  //数据源是[1,2] 时，则返回 "入库,出库"
182          *
183          */
184         multipleItemsRenderer:function (enumObj) {
185             var enumFun = Grid.Util.Format.enumRenderer(enumObj);
186             return function (values) {
187                 var result = [];
188                 if (!values) {
189                     return '';
190                 }
191                 if (!S.isArray(values)) {
192                     values = values.toString().split(',');
193                 }
194                 S.each(values, function (value) {
195                     result.push(enumFun(value));
196                 });
197 
198                 return result.join(',');
199             };
200         },
201         /*
202          * @description 将财务数据分转换成元
203          * @param {Number|String} enumObj 键值对的枚举对象 {"1":"大","2":"小"}
204          * @return {Number} 返回将分转换成元的数字
205          */
206         moneyCentRenderer:function (v) {
207             if (S.isString(v)) {
208                 v = parseFloat(v);
209             }
210             if (S.isNumber(v)) {
211                 return (v * 0.01).toFixed(2);
212             }
213             return v;
214         }
215     };

217     /**
218      * 屏蔽指定元素，并显示加载信息
219      * @memberOf Grid.Util
220      * @class 加载屏蔽类
221      * @property {String|DOM|Node} el 要屏蔽的元素，选择器、Dom元素或Node元素
222      * @param {String|DOM|Node} element 要屏蔽的元素，选择器、Dom元素或Node元素
223      * @param {Object} config 配置信息<br>
224      * 1) msg :加载时显示的加载信息<br>
225      * 2) msgCls : 加载时显示信息的样式
226      */
227     function LoadMask(element, config) {
228         var _self = this;
229 
230         _self.el = element;
231         LoadMask.superclass.constructor.call(_self, config);
232         _self._init();
233     }
234 
235     S.extend(LoadMask, S.Base);
236     //对象原型
237     S.augment(LoadMask,
238         /** @lends Grid.Util.LoadMask.prototype */
239         {
240             /**
241              * 加载时显示的加载信息
242              * @field
243              * @default Loading...
244              */
245             msg:'Loading...',
246             /**
247              * 加载时显示的加载信息的样式
248              * @field
249              * @default x-mask-loading
250              */
251             msgCls:'x-mask-loading',
252             /**
253              * 加载控件是否禁用
254              * @type {Boolean}
255              * @field
256              * @default false
257              */
258             disabled:false,
259             _init:function () {
260                 var _self = this;
261                 _self.msg = _self.get('msg') || _self.msg;
262             },
263             /**
264              * @description 设置控件不可用
265              */
266             disable:function () {
267                 this.disabled = true;
268             },
269             /**
270              * @private 设置控件可用
271              */
272             enable:function () {
273                 this.disabled = false;
274             },
275 
276             /**
277              * @private 加载已经完毕，解除屏蔽
278              */
279             onLoad:function () {
280                 util.unmaskElement(this.el);
281             },
282 
283             /**
284              * @private 开始加载，屏蔽当前元素
285              */
286             onBeforeLoad:function () {
287                 if (!this.disabled) {
288                     util.maskElement(this.el, this.msg, this.msgCls);
289 
290                 }
291             },
292             /**
293              * 显示加载条，并遮盖元素
294              */
295             show:function () {
296                 this.onBeforeLoad();
297             },
298 
299             /**
300              * 隐藏加载条，并解除遮盖元素
301              */
302             hide:function () {
303                 this.onLoad();
304             },
305 
306             /*
307              * 清理资源
308              */
309             destroy:function () {
310                 this.el = null;
311             }
312         });
313 
314     util.LoadMask = LoadMask;
315 
316 
317     return util;
