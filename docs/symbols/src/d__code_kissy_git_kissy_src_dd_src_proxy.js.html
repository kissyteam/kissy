1 /**
  2  * @fileOverview generate proxy drag object,
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("dd/proxy", function (S, Node, Base) {
  6     var DESTRUCTOR_ID = "__proxy_destructors",
  7         stamp = S.stamp,
  8         MARKER = S.guid("__dd_proxy"),
  9         PROXY_ATTR = "__proxy";
 10 
 11     /**
 12      * provide abilities for draggable tp create a proxy drag node,
 13      * instead of dragging the original node.
 14      * @memberOf DD
 15      * @class
 16      */
 17     function Proxy() {
 18         var self = this;
 19         Proxy.superclass.constructor.apply(self, arguments);
 20         self[DESTRUCTOR_ID] = {};
 21     }
 22 
 23     Proxy.ATTRS =
 24     /**
 25      * @lends DD.Proxy#
 26      */
 27     {
 28         /**
 29          * how to get the proxy node. default:clone the node itself deeply.
 30          * @type {Function}
 31          */
 32         node:{
 33             value:function (drag) {
 34                 return new Node(drag.get("node").clone(true));
 35             }
 36         },
 37         /**
 38          * destroy the proxy node at the end of this drag. default:false
 39          * @type {boolean}
 40          */
 41         destroyOnEnd:{
 42             value:false
 43         },
 44 
 45         /**
 46          * move the original node at the end of the drag. default:true
 47          * @type {boolean}
 48          */
 49         moveOnEnd:{
 50             value:true
 51         }
 52     };
 53 
 54     S.extend(Proxy, Base,
 55         /**
 56          * @lends DD.Proxy#
 57          */
 58         {
 59             /**
 60              * make this draggable object can be proxied.
 61              * @param {DD.Draggable} drag
 62              */
 63             attachDrag:function (drag) {
 64 
 65                 var self = this,
 66                     tag = stamp(drag, 1, MARKER);
 67 
 68                 if (tag && self[DESTRUCTOR_ID][tag]) {
 69                     return;
 70                 }
 71 
 72                 function start() {
 73                     var node = self.get("node"),
 74                         dragNode = drag.get("node");
 75                     // cache proxy node
 76                     if (!self[PROXY_ATTR]) {
 77                         if (S.isFunction(node)) {
 78                             node = node(drag);
 79                             node.addClass("ks-dd-proxy");
 80                             node.css("position", "absolute");
 81                             self[PROXY_ATTR] = node;
 82                         }
 83                     } else {
 84                         node = self[PROXY_ATTR];
 85                     }
 86                     dragNode.parent()
 87                         .append(node);
 88                     node.show();
 89                     node.offset(dragNode.offset());
 90                     drag.__set("dragNode", dragNode);
 91                     drag.__set("node", node);
 92                 }
 93 
 94                 function end() {
 95                     var node = self[PROXY_ATTR];
 96                     if (self.get("moveOnEnd")) {
 97                         drag.get("dragNode").offset(node.offset());
 98                     }
 99                     if (self.get("destroyOnEnd")) {
100                         node.remove();
101                         self[PROXY_ATTR] = 0;
102                     } else {
103                         node.hide();
104                     }
105                     drag.__set("node", drag.get("dragNode"));
106                 }
107 
108                 drag.on("dragstart", start);
109                 drag.on("dragend", end);
110 
111                 tag = stamp(drag, 0, MARKER);
112 
113                 self[DESTRUCTOR_ID][tag] = {
114                     drag:drag,
115                     fn:function () {
116                         drag.detach("dragstart", start);
117                         drag.detach("dragend", end);
118                     }
119                 };
120             },
121             /**
122              * make this draggable object unproxied
123              * @param {DD.Draggable} drag
124              */
125             detachDrag:function (drag) {
126                 var self = this,
127                     tag = stamp(drag, 1, MARKER),
128                     destructors = self[DESTRUCTOR_ID];
129                 if (tag && destructors[tag]) {
130                     destructors[tag].fn();
131                     delete destructors[tag];
132                 }
133             },
134 
135             /**
136              * make all draggable object associated with this proxy object unproxied
137              */
138             destroy:function () {
139                 var self = this,
140                     node = self.get("node"),
141                     destructors = self[DESTRUCTOR_ID];
142                 if (node && !S.isFunction(node)) {
143                     node.remove();
144                 }
145                 for (var d in destructors) {
146                     this.unAttach(destructors[d].drag);
147                 }
148             }
149         });
150 
151     // for compatibility
152     var ProxyPrototype = Proxy.prototype;
153     ProxyPrototype.attach = ProxyPrototype.attachDrag;
154     ProxyPrototype.unAttach = ProxyPrototype.detachDrag;
155 
156     return Proxy;
157 }, {
158     requires:['node', 'base']
159 });
