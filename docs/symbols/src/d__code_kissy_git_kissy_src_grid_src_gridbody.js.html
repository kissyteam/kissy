  7     var CLS_GRID_ROW = 'ks-grid-row',
  8         CLS_ROW_ODD = 'ks-grid-row-odd',

 11         CLS_ROW_SELECTED = 'ks-grid-row-selected',
 12         CLS_ROW_HOVER = 'ks-grid-row-hover',
 13         CLS_GRID_CELL = 'ks-grid-cell',
 14         CLS_GRID_CELL_INNER = 'ks-grid-cell-inner',
 15         CLS_TD_PREFIX = 'grid-td-',
 16         CLS_CELL_TEXT = 'ks-grid-cell-text',
 17         CLS_CELL_EMPTY = 'ks-grid-cell-empty',
 18         CLS_SCROLL_WITH = '17',
 19         ATTR_COLUMN_FIELD = 'data-column-field',
 20         DATA_ELEMENT = 'row-element';

 22     var GridBodyRender = Component.Render.extend({
 23         /**
 24          * @see {Component.Controller#renderUI}
 25          */
 26         renderUI:function () {
 27             var _self = this,
 28                 el = _self.get('el'),
 29                 tpl = _self._getTpl(),
 30                 tbody,
 31                 headerRowEl;
 32             new S.Node(tpl).appendTo(el);
 33             tbody = el.one('tbody');
 34             _self.set('tbodyEl', tbody, {silent:1});
 35             headerRowEl = _self._createHeaderRow();
 36             _self.set('headerRowEl', headerRowEl);
 37 
 38         },
 39         //clear data in table
 40         clearData:function () {
 41             var _self = this,
 42                 tbodyEl = _self.get('tbodyEl');
 43             tbodyEl.children('.' + CLS_GRID_ROW).remove();
 44         },
 45         clearRowSelected:function () {
 46             var _self = this,
 47                 tbodyEl = _self.get('tbodyEl');
 48             tbodyEl.all('.' + CLS_GRID_ROW).removeClass(CLS_ROW_SELECTED);
 49         },
 50         /**
 51          *
 52          * @internal only used by Grid.GridBody
 53          */
 54         findCell:function (id, rowEl) {
 55             var cls = CLS_TD_PREFIX + id;
 56             return rowEl.one('.' + cls);
 57         },
 58         /**
 59          * find the row dom in this view
 60          * @internal only used by Grid.GridBody
 61          */
 62         findRow:function (record) {
 63             if (!record) {
 64                 return null;
 65             }
 66             var _self = this,
 67                 tbodyEl = _self.get('tbodyEl'),
 68                 rows = tbodyEl.children('.' + CLS_GRID_ROW),
 69                 result = null;
 70             rows.each(function (rowEl) {
 71                 if (rowEl.data(DATA_ELEMENT) === record) {
 72                     result = rowEl;
 73                     return false;
 74                 }
 75             });
 76             return result;
 77         },
 78         /**
 79          * get selected rows
 80          */
 81         getSelectedRows:function () {
 82             var _self = this,
 83                 tbodyEl = _self.get('tbodyEl');
 84             return tbodyEl.all('.' + CLS_ROW_SELECTED);
 85         },
 86         //get all rows
 87         getAllRows:function () {
 88             var _self = this,
 89                 tbodyEl = _self.get('tbodyEl');
 90             return tbodyEl.all('.' + CLS_GRID_ROW);
 91         },
 92         /**
 93          * identify  whether the row was selected
 94          */
 95         isRowSelected:function (row) {
 96             var rowEl = S.one(row);
 97             return rowEl.hasClass(CLS_ROW_SELECTED);
 98         },
 99         /**
100          * identify  whether the row has been hovering
101          */
102         isRowHover:function (row) {
103             var rowEl = S.one(row);
104             return rowEl.hasClass(CLS_ROW_HOVER);
105         },
106         /**
107          * remove one record
108          * @internal only used by Grid.GridBody
109          */
110         removeRow:function (record) {
111             var _self = this,
112                 rowEl = _self.findRow(record);
113             if (rowEl) {
114                 rowEl.remove();
115             }
116             return rowEl;
117         },
118         resetHeaderRow:function () {
119             var _self = this,
120                 headerRowEl = _self.get('headerRowEl'),
121                 tbodyEl = _self.get('tbodyEl');
122             headerRowEl.remove();
123             headerRowEl = _self._createHeaderRow();
124             headerRowEl.prependTo(tbodyEl);
125             _self.set('headerRowEl', headerRowEl);
126         },
127         /**
128          * make the row selected or cancel it.
129          */
130         setRowSelected:function (row, selected) {
131             var rowEl = S.one(row);
132             if (selected) {
133                 rowEl.addClass(CLS_ROW_SELECTED);
134             } else {
135                 rowEl.removeClass(CLS_ROW_SELECTED);
136             }
137         },
138         /**
139          * set hover status or cancel it
140          */
141         setRowHover:function (row, hover) {
142             var rowEl = S.one(row);
143             if (hover) {
144                 rowEl.addClass(CLS_ROW_HOVER);
145             } else {
146                 rowEl.removeClass(CLS_ROW_HOVER);
147             }
148         },
149         //show or hide column
150         setColumnVisible:function (column) {
151             var _self = this,
152                 hide = !column.get('visible'),
153                 colId = column.get('id'),
154                 tbodyEl = _self.get('tbodyEl'),
155                 cells = tbodyEl.all('.' + CLS_TD_PREFIX + colId);
156             if (hide) {
157                 cells.hide();
158             } else {
159                 cells.show();
160             }
161         },
162         /**
163          * when header's column width changed, column in this component changed followed
164          */
165         setColumnsWidth:function (column) {
166             var _self = this,
167                 headerRowEl = _self.get('headerRowEl'),
168                 cell = _self.findCell(column.get('id'), headerRowEl);
169             if (cell) {
170                 cell.width(column.get('width'));
171             }
172             _self.setTableWidth();
173         },
174         //set table width
175         setTableWidth:function (columnsWidth) {
176             var _self = this,
177                 width = _self.get('width'),
178                 height = _self.get('height'),
179                 tableEl = _self.get('tbodyEl').parent(),
180                 forceFit = _self.get('forceFit'),
181                 headerRowEl = _self.get('headerRowEl');
182             columnsWidth = columnsWidth || _self._getColumnsWidth();
183             if (!width) {
184                 return;
185             }
186             if (width >= columnsWidth) {
187                 columnsWidth = width;
188                 if (height) {
189                     columnsWidth = width - CLS_SCROLL_WITH;
190                 }
191             }
192             tableEl.width(columnsWidth);
193         },
194         /**
195          * update the row dom in this view
196          * @internal only used by Grid.GridBody
197          */
198         updateRow:function (record) {
199             var _self = this,
200                 rowEl = _self.findRow(record),
201                 columns = _self._getColumns();
202             if (rowEl) {
203                 S.each(columns, function (column) {
204                     _self._updateCell(column, record, rowEl);
205                 });
206             }
207         },
208         //update cell dom
209         _updateCell:function (column, record, rowEl) {
210             var _self = this,
211                 cellEl = _self.findCell(column.get('id'), rowEl),
212                 text = _self._getCellText(column, record);
213             cellEl.one('.' + CLS_GRID_CELL_INNER).html(text);
214 
215         },
216         //create row element and append to tbody
217         _createRow:function (record, index) {
218             var _self = this,
219                 columns = _self._getColumns(),
220                 tbodyEl = _self.get('tbodyEl'),
221                 rowTpl = _self.get('rowTpl'),
222                 oddCls = index % 2 === 0 ? CLS_ROW_ODD : CLS_ROW_EVEN,
223                 cellsTpl = [],
224                 rowEl;
225 
226             S.each(columns, function (column) {
227                 var dataIndex = column.get('dataIndex');
228                 cellsTpl.push(_self._getCellTpl(column, dataIndex, record));
229             });
230             cellsTpl.push(_self._getEmptyCellTpl());
231             rowTpl = Template(rowTpl).render({cellsTpl:cellsTpl.join(''), oddCls:oddCls});
232             rowEl = new S.Node(rowTpl).appendTo(tbodyEl);
233             //append record to the dom
234             rowEl.data(DATA_ELEMENT, record);
235             if (index === 0) {
236                 rowEl.addClass(CLS_ROW_FIRST);
237             }
238             return rowEl;
239         },
240         //create the first row that it don't has any data,only to set columns' width
241         _createHeaderRow:function () {
242             var _self = this,
243                 columns = _self._getColumns(),
244                 tbodyEl = _self.get('tbodyEl'),
245                 rowTpl = _self.get('headerRowTpl'),
246                 rowEl,
247                 cellsTpl = [];
248 
249             S.each(columns, function (column) {
250                 cellsTpl.push(_self._getHeaderCellTpl(column));
251             });
252 
253             //if this component set width,add a empty column to fit row width
254             cellsTpl.push(_self._getEmptyCellTpl());
255             rowTpl = Template(rowTpl).render({cellsTpl:cellsTpl.join('')});
256             rowEl = S.Node(rowTpl).appendTo(tbodyEl);
257             return rowEl;
258         },
259         // 
260         _getColumns:function () {
261             var _self = this;
262             return _self.get('columns');
263         },
264         //get the sum of the columns' width
265         _getColumnsWidth:function () {
266             var _self = this,
267                 columns,
268                 totalWidth = 0;
269             columns = _self.get('columns');
270             S.each(columns, function (column) {
271                 if (column.get('visible')) {
272                     totalWidth += column.get("el").outerWidth();
273                 }
274             });
275             return totalWidth;
276         },
277         //get cell text by record and column
278         _getCellText:function (column, record) {
279             var _self = this,
280                 textTpl = column.get('cellTpl') || _self.get('cellTextTpl'),
281                 dataIndex = column.get('dataIndex'),
282                 renderer = column.get('renderer'),
283                 text = renderer ? renderer(record[dataIndex], record) : record[dataIndex];
284             return Template(textTpl).render({text:text, tips:_self._getTips(column, dataIndex, record)});
285         },
286         //get cell template by config and record
287         _getCellTpl:function (column, dataIndex, record) {
288             var _self = this,
289                 cellText = _self._getCellText(column, record),
290                 cellTpl = _self.get('cellTpl');
291             return Template(cellTpl)
292                 .render({
293                     id:column.get('id'),
294                     dataIndex:dataIndex,
295                     cellText:cellText,
296                     hide:!column.get('visible')
297                 });
298         },
299         //get cell tips
300         _getTips:function (column, dataIndex, record) {
301             var showTip = column.get('showTip'),
302                 value = '';
303             if (showTip) {
304                 value = record[dataIndex];
305                 if (S.isFunction(showTip)) {
306                     value = showTip(value, record);
307                 }
308             }
309             return value;
310         },
311         _getHeaderCellTpl:function (column) {
312             var _self = this,
313                 headerCellTpl = _self.get('headerCellTpl');
314             return Template(headerCellTpl).render({
315                 id:column.get('id'),
316                 width:column.get('width'),
317                 hide:!column.get('visible')
318             });
319         },
320         _getEmptyCellTpl:function () {
321             return '<td class="' + CLS_CELL_EMPTY + '"></td>';
322         },
323         //get the template of column
324         _getTpl:function () {
325             var _self = this,
326                 attrs = _self.getAttrVals(),
327                 tpl = _self.get('tpl');
328             return Template(tpl).render(attrs);
329         }
330     }, {
331         ATTRS:{
332             tbodyEl:{},
333             headerRowEl:{}
334         }

337     /**
338      * This class specifies the definition for the body of a {@link Grid}.
339      * In general, this class will not be instanced directly, instead a viewConfig option is passed to the grid

346         /**

349         {
350 
351             /**
352              * @see Component.Controller#bindUI
353              */
354             bindUI:function () {
355                 var _self = this;
356                 _self._publishEvent();
357                 _self._bindScrollEvent();
358                 _self._bindRowEvent();
359             },
360             /**
361              * clear data in this component
362              * @example:
363              * grid.clearData();
364              */
365             clearData:function () {
366                 var _self = this;
367                 _self.get('view').clearData();
368             },
369             /**
370              * clear rows' selection
371              */
372             clearSelection:function () {
373                 var _self = this,
374                     selectedRows = _self.get('view').getSelectedRows();
375                 selectedRows.each(function (row) {
376                     _self.onRowSelected(row, false);
377                 });
378             },
379             /**
380              * find the cell dom by record and column id
381              * @param {String|Number} id the column id
382              * @param {Object} record the record that showed in this component,if can not find ,return null
383              * @return  {Node}
384              */
385             findCell:function (id, record) {
386                 var _self = this,
387                     rowEl = null;
388                 if (record instanceof S.Node) {
389                     rowEl = record;
390                 } else {
391                     rowEl = _self.findRow(record);
392                 }
393                 if (rowEl) {
394                     return _self.get('view').findCell(id, rowEl);
395                 }
396                 return null;
397             },
398             /**
399              * find the dom by the record in this component
400              * @param {Object} record the record used to find row dom
401              * @return Node
402              */
403             findRow:function (record) {
404                 var _self = this;
405                 return _self.get('view').findRow(record);
406             },
407             /**
408              * show or hide the column
409              * @param {Grid.Column} column the column changed visible status.
410              */
411             setColumnVisible:function (column) {
412                 this.get('view').setColumnVisible(column);
413             },
414             /**
415              * @private
416              * @see Grid.Bindable#onLoad
417              */
418             onLoad:function () {
419                 var _self = this,
420                     store = _self.get('store'),
421                     records = store.getResult();
422                 _self.showData(records);
423             },
424             /**
425              * @private
426              * @see Grid.Bindable#onAdd
427              */
428             onAdd:function (e) {
429                 var _self = this,
430                     data = e.data,
431                     store = _self.get('store'),
432                     count = store.getCount();
433                 _self._addData(data, count);
434             },
435             /**
436              * @private
437              */
438             onRemove:function (e) {
439                 var _self = this,
440                     data = e.data,
441                     removedRow = null;
442                 S.each(data, function (record) {
443                     removedRow = _self.get('view').removeRow(record);
444                     if (removedRow) {
445                         _self.fire('rowremoved', {record:record, row:removedRow[0]});
446                     }
447                 });
448             },
449             /**
450              * @private
451              */
452             onUpdate:function (e) {
453                 var _self = this,
454                     record = e.record;
455                 _self.get('view').updateRow(record);
456             },
457             /**
458              * @private
459              */
460             onLocalSort:function () {
461                 var _self = this;
462                 _self.onLoad();
463             },
464             /**
465              * @private
466              */
467             onRowSelected:function (rowEl, selected) {
468                 var _self = this,
469                     view = _self.get('view'),
470                 //isSelected = view.isRowSelected(rowEl),
471                     event;
472                 //if(isSelected != selected){
473                 event = selected ? 'rowselected' : 'rowunselected';
474                 view.setRowSelected(rowEl, selected);
475                 _self.fire(event, {row:rowEl[0], record:rowEl.data(DATA_ELEMENT)});
476                 //}
477 
478             },
479             /**
480              * set the records selected by the key-value
481              * @param {Array|Object} records the records which will be set to selected
482              */
483             setSelection:function (records) {
484                 var _self = this,
485                     view = _self.get('view');
486 
487                 if (!records) {
488                     return;
489                 }
490                 if (!S.isArray(records)) {
491                     records = [records];
492                 }
493                 S.each(records, function (record) {
494                     var rowEl = _self.findRow(record);
495                     if (rowEl && !view.isRowSelected(rowEl)) {
496                         _self.onRowSelected(rowEl, true);
497                     }
498                 });
499             },
500             /**
501              * show data in this controller
502              * @param {Array} data show the given data in table
503              */
504             showData:function (data) {
505                 var _self = this;
506                 _self.clearData();
507                 _self._addData(data);
508                 _self.fire('aftershow');
509             },
510             /**
511              * when some columns changed,must reset body's column
512              */
513             resetColumns:function () {
514                 var _self = this,
515                     store = _self.get('store');
516                 //remove the rows of this table
517                 //_self.clearData();
518                 //recreate the header row
519                 _self.get('view').resetHeaderRow();
520                 //show data
521                 if (store) {
522                     _self.onLoad();
523                 }
524                 //
525             },
526             /**
527              * change the column's width
528              * @param {Grid.Column} column a column in config
529              */
530             resetColumnsWidth:function (column) {
531                 this.get('view').setColumnsWidth(column);
532             },
533             /**
534              * set all rows selected
535              */
536             setAllSelection:function () {
537                 var _self = this,
538                     rows = _self.get('view').getAllRows();
539                 rows.each(function (row) {
540                     _self.onRowSelected(row, true);
541                 });
542             },

553             },

559                     var rowEl = _self.get('view')._createRow(record, position + index);
560                     _self.fire('rowcreated', {record:record, data:record, row:rowEl[0]});

562             },
563             //when body scrolled,the other component of grid also scrolled

566                     el = _self.get('el');
567                 el.on('scroll', function () {
568                     var left = el.scrollLeft(),
569                         top = el.scrollTop();
570                     _self.fire('scroll', {scrollLeft:left, scrollTop:top});
571                 });
572             },
573             //bind rows event of table
574             _bindRowEvent:function () {
575                 var _self = this,
576                     tbodyEl = _self.get('tbodyEl');
577 
578                 tbodyEl.delegate('click', '.' + CLS_GRID_ROW, function (e) {
579                     _self._rowClickEvent(e);

581             },

585                     parent = _self.get('parent'),
586                     events = _self.get('events');
587                 if (!parent) {
588                     return;
589                 }
590                 S.each(events, function (event) {
591                     _self.publish(event, {
592                         bubbles:1
593                     });
594                 });
595             },
596             _rowClickEvent:function (event) {
597                 var _self = this,
598                     multiSelect = _self.get('multiSelect'),
599                     sender = S.one(event.currentTarget),
600                     view = _self.get('view'),
601                     record = sender.data(DATA_ELEMENT),
602                     cell = S.one(event.target).closest("." + CLS_GRID_CELL),
603                     selected = view.isRowSelected(sender);
604                 //when in multiple select model,toggle the row's select status
605                 if (cell) {
606                     _self.fire('cellclick', {record:record, row:sender[0], cell:cell[0], field:cell.attr(ATTR_COLUMN_FIELD), domTarget:event.target});
607                 }
608                 _self.fire('rowclick', {record:record, row:sender[0]});
609                 if (multiSelect) {
610                     _self.onRowSelected(sender, !selected);
611                 } else {
612                     if (!selected) {
613                         _self.clearSelection();
614                         _self.onRowSelected(sender, true);
615                     }
616                 }
617             },

626                 _self.setTableWidth();
627                 //}
628             },
629             //when set this component's height ,the table's width is also changed

633                 _self.setTableWidth();
634                 //}
