1 /**
  2  * @fileOverview mask extension for kissy
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("uibase/mask", function () {
  6 
  7     /**
  8      * make component can show with mask
  9      * @class
 10      * @memberOf UIBase
 11      */
 12     function Mask() {
 13     }
 14 
 15     Mask.ATTRS =
 16     /**
 17      * @lends UIBase.Mask.prototype
 18      */
 19     {
 20         /**
 21          * whether show mask layer when component shows
 22          * @type boolean
 23          */
 24         mask:{
 25             value:false
 26         }
 27     };
 28 
 29     Mask.prototype = {
 30         _uiSetMask:function (v) {
 31             var self = this;
 32             if (v) {
 33                 self.on("show", self.get("view")._maskExtShow, self.get("view"));
 34                 self.on("hide", self.get("view")._maskExtHide, self.get("view"));
 35             } else {
 36                 self.detach("show", self.get("view")._maskExtShow, self.get("view"));
 37                 self.detach("hide", self.get("view")._maskExtHide, self.get("view"));
 38             }
 39         }
 40     };
 41 
 42 
 43     return Mask;
 44 }, {requires:["ua"]});
