546             /**
547              * Insert a element into current editor.
548              * @param {NodeList} element
549              */
550             insertElement:function (element) {
551 
552                 var self = this;
553 
554                 if (self.get("mode") !== WYSIWYG_MODE) {
555                     return;
556                 }
557 
558                 self.focus();
559 
560                 var clone,
561                     elementName = element['nodeName'](),
562                     xhtml_dtd = Editor.XHTML_DTD,
563                     isBlock = xhtml_dtd['$block'][ elementName ],
564                     KER = Editor.RANGE,
565                     selection = self.getSelection(),
566                     ranges = selection && selection.getRanges(),
567                     range,
568                     notWhitespaceEval,
569                     i,
570                     next,
571                     nextName,
572                     lastElement;
573 
574                 if (!ranges || ranges.length == 0) {
575                     return;
576                 }
577 
578                 self.execCommand("save");
579 
580                 for (i = ranges.length - 1; i >= 0; i--) {
581                     range = ranges[ i ];
582                     // Remove the original contents.
583 
584                     clone = !i && element || element['clone'](TRUE);
585                     range.insertNodeByDtd(clone);
586                     // Save the last element reference so we can make the
587                     // selection later.
588                     if (!lastElement) {
589                         lastElement = clone;
590                     }
591                 }
592 
593                 if (!lastElement) {
594                     return;
595                 }
596 
597                 range.moveToPosition(lastElement, KER.POSITION_AFTER_END);
598                 // If we're inserting a block element immediately followed by
599                 // another block element, the selection must move there. (#3100,#5436)
600                 if (isBlock) {
601                     notWhitespaceEval = Editor.Walker.whitespaces(true);
602                     next = lastElement.next(notWhitespaceEval, 1);
603                     nextName = next && next[0].nodeType == DOM.ELEMENT_NODE
604                         && next.nodeName();
605                     // Check if it's a block element that accepts text.
606                     if (nextName &&
607                         xhtml_dtd.$block[ nextName ] &&
608                         xhtml_dtd[ nextName ]['#']) {
609                         range.moveToElementEditablePosition(next);
610                     }
611                 }
612                 selection.selectRanges([ range ]);
613                 self.focus();
614                 // http://code.google.com/p/kissy/issues/detail?can=1&start=100&id=121
615                 clone && clone.scrollIntoView(undefined, false);
616                 saveLater(self);
617                 return clone;
618             },

620             /**
621              * Insert html string into current editor.
622              * @param data {string}
623              */
624             insertHtml:function (data, dataFilter) {
625                 var self = this,
626                     htmlDataProcessor;
627 
628                 if (self.get("mode") !== WYSIWYG_MODE) {
629                     return;
630                 }
631 
632                 if (htmlDataProcessor = self.htmlDataProcessor) {
633                     data = htmlDataProcessor.toDataFormat(data, null, dataFilter);
634                 }
635 
636                 self.focus();
637                 self.execCommand("save");
638 
639                 var editorDoc = self.get("document")[0],
640                     saveInterval = 0;
641                 // ie9 仍然需要这样！
642                 // ie9 标准 selection 有问题，连续插入不能定位光标到插入内容后面
643                 if (IS_IE) {
644                     var $sel = editorDoc.selection;
645                     if ($sel.type == 'Control') {
646                         $sel.clear();
647                     }
648                     try {
649                         $sel.createRange().pasteHTML(data);
650                     } catch (e) {
651                         S.log("insertHtml error in ie");
652                     }
653                 } else {
654                     // ie9 仍然没有
655                     // 1.webkit insert html 有问题！会把标签去掉，算了直接用 insertElement.
656                     // 10.0 修复？？
657                     // firefox 初始编辑器无焦点报异常
658                     try {
659                         editorDoc.execCommand('inserthtml', FALSE, data);
660                     } catch (e) {
661                         setTimeout(function () {
662                             // still not ok in ff!
663                             // 手动选择 body 的第一个节点
664                             if (self.getSelection().getRanges().length == 0) {
665                                 var r = new Editor.Range(editorDoc),
666                                     node = DOM.first(editorDoc.body, function (el) {
667                                         return el.nodeType == 1 && DOM.nodeName(el) != "br";
668                                     });
669                                 if (!node) {
670                                     node = new Node(editorDoc.createElement("p"));
671                                     node._4e_appendBogus(undefined);
672                                     editorDoc.body.appendChild(node[0]);
673                                 }
674                                 r.setStartAt(node, Editor.RANGE.POSITION_AFTER_START);
675                                 r.select();
676                             }
677                             editorDoc.execCommand('inserthtml', FALSE, data);
678                         }, saveInterval = 100);
679                     }
680                 }
681                 // bug by zjw2004112@163.com :
682                 // 有的浏览器 ： chrome , ie67 貌似不会自动滚动到粘贴后的位置
683                 setTimeout(function () {
684                     self.getSelection().scrollIntoView();
685                 }, saveInterval);
686                 saveLater(self, saveInterval);
687             }
688         });

690     /**
691      * 初始化iframe内容以及浏览器间兼容性处理，
692      * 必须等待iframe内的脚本向父窗口通知
693      *
694      * @param id {string}
695      */
696     Editor["_initIFrame"] = function (id) {
697 
698         var self = focusManager.getInstance(id),
699             doc = self.get("document")[0],
700         // Remove bootstrap script from the DOM.
701             script = doc.getElementById("ke_active_script");
702 
703         DOM.remove(script);
704 
705         fixByBindIframeDoc(self);
706 
707         var body = doc.body;
708 
709         /**
710          * from kissy editor 1.0
711          *
712          * // 注1：在 tinymce 里，designMode = "on" 放在 try catch 里。
713          //     原因是在 firefox 下，当iframe 在 display: none 的容器里，会导致错误。
714          //     但经过我测试，firefox 3+ 以上已无此现象。
715          // 注2： ie 用 contentEditable = true.
716          //     原因是在 ie 下，IE needs to use contentEditable or
717          // it will display non secure items for HTTPS
718          // Ref:
719          //   - Differences between designMode and contentEditable
720          //     http://74.125.153.132/search?q=cache:5LveNs1yHyMJ:nagoon97.wordpress.com/2008/04/20/differences-between-designmode-and-contenteditable/+ie+contentEditable+designMode+different&cd=6&hl=en&ct=clnk
721          */
722 
723         // 这里对主流浏览器全部使用 contenteditable
724         // 那么不同于 kissy editor 1.0
725         // 在body范围外右键，不会出现 复制，粘贴等菜单
726         // 因为这时右键作用在document而不是body
727         // 1.0 document.designMode='on' 是编辑模式
728         // 2.0 body.contentEditable=true body外不是编辑模式
729         if (IS_IE) {
730             // Don't display the focus border.
731             body['hideFocus'] = TRUE;
732             // Disable and re-enable the body to avoid IE from
733             // taking the editing focus at startup. (#141 / #523)
734             body.disabled = TRUE;
735             body['contentEditable'] = TRUE;
736             body.removeAttribute('disabled');
737         } else {
738             // Avoid opening design mode in a frame window thread,
739             // which will cause host page scrolling.(#4397)
740             setTimeout(function () {
741                 // Prefer 'contentEditable' instead of 'designMode'. (#3593)
742                 if (UA['gecko'] || UA['opera']) {
743                     body['contentEditable'] = TRUE;
744                 }
745                 else if (UA['webkit'])
746                     body.parentNode['contentEditable'] = TRUE;
747                 else
748                     doc['designMode'] = 'on';
749             }, 0);
750         }
751 
752         // IE standard compliant in editing frame doesn't focus the editor when

851     function blinkCursor(doc, retry) {
852         var body = doc.body;
853         tryThese(
854             function () {
855                 doc['designMode'] = 'on';
856                 //异步引起时序问题，尽可能小间隔
857                 setTimeout(function () {
858                     doc['designMode'] = 'off';
859                     body.focus();
860                     // Try it again once..
861                     if (!arguments.callee.retry) {
862                         arguments.callee.retry = TRUE;
863                         //arguments.callee();
864                     }
865                 }, 50);
866             },
867             function () {
868                 // The above call is known to fail when parent DOM
869                 // tree layout changes may break design mode. (#5782)
870                 // Refresh the 'contentEditable' is a cue to this.
871                 doc['designMode'] = 'off';
872                 DOM.attr(body, 'contentEditable', FALSE);
873                 DOM.attr(body, 'contentEditable', TRUE);
874                 // Try it again once..
875                 !retry && blinkCursor(doc, 1);
876             }
877         );
878     }

880     function fixByBindIframeDoc(self) {
881         var iframe = self.get("iframe"),
882             textarea = self.get("textarea")[0],
883             win = self.get("window")[0],
884             doc = self.get("document")[0];
885 
886         // Gecko need a key event to 'wake up' the editing
887         // ability when document is empty.(#3864)
888         // activateEditing 删掉，初始引起屏幕滚动了
889         // Webkit: avoid from editing form control elements content.
890         if (UA['webkit']) {
891             Event.on(doc, "click", function (ev) {
892                 var control = new Node(ev.target);
893                 if (S.inArray(control.nodeName(), ['input', 'select'])) {
894                     ev.preventDefault();
895                 }
896             });
897             // Prevent from editig textfield/textarea value.
898             Event.on(doc, "mouseup", function (ev) {
899                 var control = new Node(ev.target);
900                 if (S.inArray(control.nodeName(), ['input', 'textarea'])) {
901                     ev.preventDefault();
902                 }
903             });
904         }
905 
906 
907         // Create an invisible element to grab focus.
908         if (UA['gecko'] || IS_IE || UA['opera']) {
909             var focusGrabber;
910             focusGrabber = new Node(
911                 // Use 'span' instead of anything else to fly under the screen-reader radar. (#5049)
912                 '<span ' +
913                     'tabindex="-1" ' +
914                     'style="position:absolute; left:-10000"' +
915                     ' role="presentation"' +
916                     '></span>').insertAfter(textarea);
917             focusGrabber.on('focus', function () {
918                 self.focus();
919             });
920             self.activateGecko = function () {
921                 if (UA['gecko'] && self.__iframeFocus)
922                     focusGrabber[0].focus();
923             };
924             self.on('destroy', function () {
925                 focusGrabber.detach();
926                 focusGrabber.remove();
927             });
928         }
929 
930 
931         Event.on(win, 'focus', function () {
932             /**
933              * yiminghe特别注意：firefox光标丢失bug
934              * blink后光标出现在最后，这就需要实现保存range
935              * focus后再恢复range
936              */
937             if (UA['gecko']) {
938                 blinkCursor(doc, FALSE);
939             }
940             else if (UA['opera']) {
941                 doc.body.focus();
942             }
943             // focus 后强制刷新自己状态
944             self.notifySelectionChange();
945         });
946 
947 
948         if (UA['gecko']) {
949             /**
950              * firefox 焦点丢失后，再点编辑器区域焦点会移不过来，要点两下
951              */
952             Event.on(doc, "mousedown", function () {
953                 if (!self.__iframeFocus) {
954                     blinkCursor(doc, FALSE);
955                 }
956             });
957         }
958 
959         if (IS_IE) {
960             // Override keystrokes which should have deletion behavior
961             // on control types in IE . (#4047)
962             /**
963              * 选择img，出现缩放框后不能直接删除
964              */
965             Event.on(doc, 'keydown', function (evt) {
966                 var keyCode = evt.keyCode;
967                 // Backspace OR Delete.
968                 if (keyCode in { 8:1, 46:1 }) {
969                     var sel = self.getSelection(),
970                         control = sel.getSelectedElement();
971                     if (control) {
972                         // Make undo snapshot.
973                         self.execCommand('save');
974                         // Delete any element that 'hasLayout' (e.g. hr,table) in IE8 will
975                         // break up the selection, safely manage it here. (#4795)
976                         var bookmark = sel.getRanges()[ 0 ].createBookmark();
977                         // Remove the control manually.
978                         control.remove();
979                         sel.selectBookmarks([ bookmark ]);
980                         self.execCommand('save');
981                         evt.preventDefault();
982                     }
983                 }
984             });
985 
986             // PageUp/PageDown scrolling is broken in document
987             // with standard doctype, manually fix it. (#4736)
988             // ie8 主窗口滚动？？
989             if (doc.compatMode == 'CSS1Compat') {
990                 var pageUpDownKeys = { 33:1, 34:1 };
991                 Event.on(doc, 'keydown', function (evt) {
992                     if (evt.keyCode in pageUpDownKeys) {
993                         setTimeout(function () {
994                             self.getSelection().scrollIntoView();
995                         }, 0);
996                     }
997                 });
998             }
999         }
1000 
1001         // Gecko/Webkit need some help when selecting control type elements. (#3448)
1002         if (UA['webkit']) {
1003             Event.on(doc, "mousedown", function (ev) {
1004                 var control = new Node(ev.target);
1005                 if (S.inArray(control.nodeName(), ['img', 'hr', 'input', 'textarea', 'select'])) {
1006                     self.getSelection().selectElement(control);
1007                 }
1008             });
1009         }
1010 
1011 
1012         if (UA['gecko']) {
1013             Event.on(doc, "dragstart", function (ev) {
1014                 var control = new Node(ev.target);
1015                 if (control.nodeName() === 'img' && /ke_/.test(control[0].className)) {
1016                     // firefox禁止拖放
1017                     ev.preventDefault();
1018                 }
1019             });
1020         }
1021         //注意：必须放在这个位置，等iframe加载好再开始运行
1022         //加入焦点管理，和其他实例联系起来
1023         focusManager.add(self);
1024 
1025     }

1027     function prepareIFrameHtml(id, customStyle, customLink, data) {
1028         var links = "",
1029             i,
1030             innerCssFile = Utils.debugUrl("theme/editor-iframe.css");
1031 
1032         for (i = 0; i < customLink.length; i++) {
1033             links += S.substitute('<link href="' + '{href}" rel="stylesheet" />', {
1034                 href:customLink[i]
1035             });
1036         }
1037 
1038         return S.substitute(IFRAME_HTML_TPL, {
1039             // kissy-editor #12
1040             // IE8 doesn't support carets behind images(empty content after image's block)
1041             // setting ie7 compatible mode would force IE8+ to run in IE7 compat mode.
1042             doctype:DOC.documentMode === 8 ?
1043                 '<meta http-equiv="X-UA-Compatible" content="IE=7" />' :
1044                 "",
1045             title:"${title}",
1046             href:innerCssFile,
1047             style:customStyle,
1048             // firefox 必须里面有东西，否则编辑前不能删除!
1049             data:data || " ",
1050             script:id ?
1051                 // The script that launches the bootstrap logic on 'domReady', so the document
1052                 // is fully editable even before the editing iframe is fully loaded (#4455).
1053                 // 确保iframe确实载入成功,过早的话 document.domain 会出现无法访问
1054                 ('<script id="ke_active_script">' +
1055                     ( DOM.isCustomDomain() ? ( 'document.domain="' + DOC.domain + '";' ) : '' ) +
1056                     'parent.KISSY.Editor._initIFrame("' + id + '");' +
1057                     '</script>') :
1058                 ''
1059 
1060         });
1061     }

1073     function setUpIFrame(self, data) {
1074         var iframe = self.get("iframe"),
1075             html = prepareIFrameHtml(self._UUID,
1076                 self.get('customStyle'),
1077                 self.get('customLink'), data),
1078             iframeDom = iframe[0],
1079             win = iframeDom.contentWindow,
1080             doc;
1081         iframe.__loaded = 1;
1082         try {
1083             // In IE, with custom document.domain, it may happen that
1084             // the iframe is not yet available, resulting in "Access
1085             // Denied" for the following property access.
1086             //ie 设置domain 有问题：yui也有
1087             //http://yuilibrary.com/projects/yui2/ticket/2052000
1088             //http://waelchatila.com/2007/10/31/1193851500000.html
1089             //http://nagoon97.wordpress.com/tag/designmode/
1090             doc = win.document;
1091         } catch (e) {
1092             // Trick to solve this issue, forcing the iframe to get ready
1093             // by simply setting its "src" property.
1094             //noinspection SillyAssignmentJS
1095             iframeDom.src = iframeDom.src;
1096             // In IE6 though, the above is not enough, so we must pause the
1097             // execution for a while, giving it time to think.
1098             if (IS_IE < 7) {
1099                 setTimeout(run, 10);
1100                 return;
1101             }
1102         }
1103         run();
1104         function run() {
1105             doc = win.document;
1106             self.__set("document", new Node(doc));
1107             self.__set("window", new Node(win));
1108             iframe.detach();
1109             // Don't leave any history log in IE. (#5657)
1110             doc['open']("text/html", "replace");
1111             doc.write(html);
1112             doc.close();
1113         }
1114     }

1116     function createIframe(self, afterData) {
1117         var iframe = new Node(IFRAME_TPL),
1118             textarea = self.get("textarea");
1119         if (textarea.hasAttr("tabindex")) {
1120             iframe.attr("tabIndex", UA['webkit'] ? -1 : textarea.attr("tabIndex"));
1121         }
1122         self.get("iframeWrapEl").prepend(iframe);
1123         self.set("iframe", iframe);
1124         self.__docReady = 0;
1125         // With FF, it's better to load the data on iframe.load. (#3894,#4058)
1126         if (UA['gecko'] && !iframe.__loaded) {
1127             iframe.on('load', function () {
1128                 setUpIFrame(self, afterData);
1129             }, self);
1130         } else {
1131             // webkit(chrome) load等不来！
1132             setUpIFrame(self, afterData);
1133         }
