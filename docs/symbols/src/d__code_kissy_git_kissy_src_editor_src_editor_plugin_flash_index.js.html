1 /**
  2  * Add flash plugin.
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/flash/index", function (S, Editor, FlashBaseClass, flashUtils) {
  6 
  7     var CLS_FLASH = 'ke_flash',
  8         TYPE_FLASH = 'flash';
  9 
 10     return {
 11         init:function (editor) {
 12             var dataProcessor = editor.htmlDataProcessor,
 13                 dataFilter = dataProcessor.dataFilter;
 14 
 15             dataFilter.addRules({
 16                     tags:{
 17                         'object':function (element) {
 18                             var classId = element.getAttribute("classid"), i;
 19                             if (!classId) {
 20                                 var childNodes = element.childNodes;
 21                                 // Look for the inner <embed>
 22                                 for (i = 0; i < childNodes.length; i++) {
 23                                     if (childNodes[i].nodeName == 'embed') {
 24                                         if (!flashUtils.isFlashEmbed(childNodes[i][ i ])) {
 25                                             return dataProcessor
 26                                                 .createFakeParserElement(element,
 27                                                 CLS_FLASH, TYPE_FLASH, true);
 28                                         } else {
 29                                             return null;
 30                                         }
 31                                     }
 32                                 }
 33                                 return null;
 34                             }
 35                             return dataProcessor.createFakeParserElement(element,
 36                                 CLS_FLASH, TYPE_FLASH, true);
 37                         },
 38                         'embed':function (element) {
 39                             if (flashUtils.isFlashEmbed(element)) {
 40                                 return dataProcessor
 41                                     .createFakeParserElement(element, CLS_FLASH, TYPE_FLASH, true);
 42                             } else {
 43                                 return null;
 44                             }
 45                         }
 46                     }},
 47                 5);
 48 
 49 
 50             var pluginConfig = editor.get("pluginConfig").flash || {},
 51                 flashControl = new FlashBaseClass({
 52                     editor:editor,
 53                     cls:CLS_FLASH,
 54                     type:TYPE_FLASH,
 55                     contextMenuHandlers:{
 56                         Flash属性:function () {
 57                             var selectedEl = this.selectedEl;
 58                             if (selectedEl) {
 59                                 flashControl.show(selectedEl);
 60                             }
 61                         }
 62                     }
 63                 });
 64 
 65             if (pluginConfig.btn !== false) {
 66                 editor.addButton({
 67                     contentCls:"ke-toolbar-flash",
 68                     title:"插入Flash",
 69                     mode:Editor.WYSIWYG_MODE
 70                 }, {
 71                     offClick:function () {
 72                         flashControl.show();
 73                     }
 74                 });
 75             }
 76         }
 77     };
 78 
 79 }, {
 80     requires:['editor', '../flashCommon/baseClass', '../flashCommon/utils']
 81 });
