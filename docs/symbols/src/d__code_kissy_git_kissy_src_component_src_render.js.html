1 /**
  2  * @fileOverview render base class for kissy
  3  * @author yiminghe@gmail.com
  4  * @see http://martinfowler.com/eaaDev/uiArchs.html
  5  */
  6 KISSY.add("component/render", function (S, UIBase, UIStore) {
  7 
  8     /**
  9      * Base Render class for KISSY Component.
 10      * @class
 11      * @memberOf Component
 12      * @name Render
 13      * @extends UIBase
 14      */
 15     return UIBase.create([UIBase.Box.Render],
 16         /**
 17          * @lends Component.Render#
 18          */
 19         {
 20 
 21             /**
 22              * Get all css class name to be applied to the root element of this component for given state.
 23              * the css class names are prefixed with component name.
 24              * @param {String} [state] This component's state info.
 25              */
 26             getComponentCssClassWithState:function (state) {
 27                 var self = this, componentCls = this.__componentClasses;
 28                 state = state || "";
 29                 return self.getCssClassWithPrefix(componentCls.split(/\s+/).join(state + " ") + state);
 30             },
 31 
 32             /**
 33              * Get full class name (with prefix) for current component
 34              * @param classes {String} class names without prefixCls. Separated by space.
 35              * @function
 36              * @return {String} class name with prefixCls
 37              * @private
 38              */
 39             getCssClassWithPrefix:UIStore.getCssClassWithPrefix,
 40 
 41             createDom:function () {
 42                 var self = this;
 43                 self.get("el").addClass(self.getComponentCssClassWithState());
 44             },
 45 
 46             /**
 47              * Returns the dom element which is responsible for listening keyboard events.
 48              */
 49             getKeyEventTarget:function () {
 50                 return this.get("el");
 51             },
 52 
 53             /**
 54              * Return the dom element into which child component to be rendered.
 55              */
 56             getContentElement:function () {
 57                 return this.get("contentEl") || this.get("el");
 58             },
 59 
 60             /**
 61              * @protected
 62              */
 63             _uiSetFocusable:function (v) {
 64                 var el = this.getKeyEventTarget(),
 65                     tabindex = el.attr("tabindex");
 66                 if (tabindex >= 0 && !v) {
 67                     el.attr("tabindex", -1);
 68                 } else if (!(tabindex >= 0) && v) {
 69                     el.attr("tabindex", 0);
 70                 }
 71             },
 72 
 73             /**
 74              * @protected
 75              */
 76             _uiSetHighlighted:function (v) {
 77                 var self = this,
 78                     componentCls = self.getComponentCssClassWithState("-hover"),
 79                     el = self.get("el");
 80                 el[v ? 'addClass' : 'removeClass'](componentCls);
 81             },
 82 
 83             /**
 84              * @protected
 85              */
 86             _uiSetDisabled:function (v) {
 87                 var self = this,
 88                     componentCls = self.getComponentCssClassWithState("-disabled"),
 89                     el = self.get("el");
 90                 el[v ? 'addClass' : 'removeClass'](componentCls)
 91                     //不能被 tab focus 到
 92                     //support aria
 93                     .attr({
 94                         tabindex:v ? -1 : 0,
 95                         aria-disabled:v
 96                     });
 97 
 98             },
 99             /**
100              * @protected
101              */
102             _uiSetActive:function (v) {
103                 var self = this,
104                     componentCls = self.getComponentCssClassWithState("-active");
105                 self.get("el")[v ? 'addClass' : 'removeClass'](componentCls)
106                     .attr("aria-pressed", !!v);
107             },
108             /**
109              * @protected
110              */
111             _uiSetFocused:function (v) {
112                 var self = this,
113                     el = self.get("el"),
114                     componentCls = self.getComponentCssClassWithState("-focused");
115                 el[v ? 'addClass' : 'removeClass'](componentCls);
116             }
117 
118         }, {//  screen state
119             ATTRS:/**
120              * @lends Component.Render#
121              */
122             {
123                 /**
124                  * see {@link Component.Controller#prefixCls}
125                  */
126                 prefixCls:{},
127                 /**
128                  * see {@link Component.Controller#focusable}
129                  */
130                 focusable:{},
131                 /**
132                  * see {@link Component.Controller#focused}
133                  */
134                 focused:{},
135                 /**
136                  * see {@link Component.Controller#active}
137                  */
138                 active:{},
139                 /**
140                  * see {@link Component.Controller#disabled}
141                  */
142                 disabled:{},
143                 /**
144                  * see {@link Component.Controller#highlighted}
145                  */
146                 highlighted:{}
147             }
148         }, "Component_Render");
149 }, {
150     requires:['uibase', './uistore']
151 });
