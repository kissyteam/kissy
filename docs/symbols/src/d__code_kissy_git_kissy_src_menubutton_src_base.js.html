1 /**
  2  * @fileOverview combination of menu and button ,similar to native select
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menubutton/base", function (S, UIBase, Node, Button, MenuButtonRender, Menu, Component, undefined) {
  6 
  7     var win = S.Env.host;
  8 
  9     function getMenu(self, init) {
 10         var m = self.get("menu");
 11         if (S.isFunction(m)) {
 12             if (init) {
 13                 m = m.call(self);
 14                 self.__set("menu", m);
 15             } else {
 16                 return null;
 17             }
 18         }
 19         if (m && m.get("parent") !== self) {
 20             m.__set("parent", self);
 21             self.__bindMenu();
 22         }
 23         return m;
 24     }
 25 
 26     function _reposition() {
 27         var self = this,
 28             menu = getMenu(self);
 29         if (menu &&
 30             menu.get("visible")) {
 31             menu.set("align", S.merge({
 32                 node:self.get("el")
 33             }, ALIGN, self.get("menuAlign")));
 34         }
 35     }
 36 
 37     var $ = Node.all,
 38         KeyCodes = Node.KeyCodes,
 39         ALIGN = {
 40             points:["bl", "tl"],
 41             overflow:{
 42                 failX:1,
 43                 failY:1,
 44                 adjustX:1,
 45                 adjustY:1
 46             }
 47         },
 48         /**
 49          * @name MenuButton
 50          * @constructor
 51          * @extends Button
 52          */
 53             MenuButton =
 54             UIBase.create(Button, [Component.DecorateChild], {
 55 
 56                 _getMenu:function (init) {
 57                     return getMenu(this, init);
 58                 },
 59 
 60                 initializer:function () {
 61                     this._reposition = S.buffer(_reposition, 50, this);
 62                 },
 63 
 64                 /**
 65                  * private
 66                  */
 67                 _hideMenu:function () {
 68                     var menu = getMenu(this);
 69                     if (menu) {
 70                         menu.hide();
 71                     }
 72                 },
 73 
 74                 /**
 75                  * private
 76                  */
 77                 _showMenu:function () {
 78                     var self = this,
 79                         el = self.get("el"),
 80                         menu = getMenu(self, 1);
 81                     if (menu && !menu.get("visible")) {
 82                         menu.set("align", S.merge({
 83                             node:el
 84                         }, ALIGN, self.get("menuAlign")));
 85                         menu.show();
 86                         el.attr("aria-haspopup", menu.get("el").attr("id"));
 87                     }
 88                 },
 89 
 90                 _uiSetCollapsed:function (v) {
 91                     if (v) {
 92                         this._hideMenu();
 93                     } else {
 94                         this._showMenu();
 95                     }
 96                 },
 97 
 98                 /**
 99                  * 产生菜单时对菜单监听，只监听一次
100                  * @protected
101                  */
102                 __bindMenu:function () {
103                     var self = this,
104                         menu = getMenu(self);
105                     if (menu) {
106                         menu.on("afterActiveItemChange", function (ev) {
107                             self.set("activeItem", ev.newVal);
108                         });
109 
110                         menu.on("click", self._handleMenuClick, self);
111 
112                         //窗口改变大小，重新调整
113                         $(win).on("resize", self._reposition, self);
114                         /*
115                          bind 与 getMenu 都可能调用，时序不定
116                          */
117                         self.__bindMenu = S.noop;
118                     }
119                 },
120 
121                 /**
122                  * @protected
123                  */
124                 _handleMenuClick:function (e) {
125                     var self = this;
126                     self.fire("click", {
127                         target:e.target
128                     });
129                 },
130 
131                 /**
132                  * @private
133                  */
134                 bindUI:function () {
135                     this.__bindMenu();
136                 },
137 
138                 /**
139                  * @inheritDoc
140                  */
141                 _handleKeyEventInternal:function (e) {
142                     var self = this,
143                         menu = getMenu(self);
144 
145                     // space 只在 keyup 时处理
146                     if (e.keyCode == KeyCodes.SPACE) {
147                         // Prevent page scrolling in Chrome.
148                         e.preventDefault();
149                         if (e.type != "keyup") {
150                             return undefined;
151                         }
152                     } else if (e.type != "keydown") {
153                         return undefined;
154                     }
155                     //转发给 menu 处理
156                     if (menu && menu.get("visible")) {
157                         var handledByMenu = menu._handleKeydown(e);
158                         // esc
159                         if (e.keyCode == KeyCodes.ESC) {
160                             self.set("collapsed", true);
161                             return true;
162                         }
163                         return handledByMenu;
164                     }
165 
166                     // Menu is closed, and the user hit the down/up/space key; open menu.
167                     if (e.keyCode == KeyCodes.SPACE ||
168                         e.keyCode == KeyCodes.DOWN ||
169                         e.keyCode == KeyCodes.UP) {
170                         self.set("collapsed", false);
171                         return true;
172                     }
173                     return undefined;
174                 },
175 
176                 /**
177                  * handle click or enter key
178                  */
179                 _performInternal:function () {
180                     var self = this;
181                     self.set("collapsed", !self.get("collapsed"));
182 
183                 },
184 
185                 /**
186                  * @inheritDoc
187                  */
188                 _handleBlur:function (e) {
189                     var self = this;
190                     MenuButton.superclass._handleBlur.call(self, e);
191                     // such as : click the document
192                     self.set("collapsed", true);
193                 },
194 
195                 constructMenu:function () {
196                     var self = this,
197                         m = new Menu.PopupMenu(S.mix({
198                             prefixCls:self.get("prefixCls")
199                         }, self.get("menuCfg")));
200                     self.__set("menu", m);
201                     self.__bindMenu();
202                     return m;
203                 },
204 
205                 /**
206                  * if no menu , then construct
207                  * @private
208                  */
209                 getMenu:function () {
210                     var self = this,
211                         m = getMenu(self);
212                     if (!m) {
213                         m = self.constructMenu();
214                     }
215                     return m;
216                 },
217 
218                 /**
219                  * Adds a new menu item at the end of the menu.
220                  * @param item Menu item to add to the menu.
221                  */
222                 addItem:function (item, index) {
223                     this.getMenu().addChild(item, index);
224                 },
225 
226                 removeItem:function (c, destroy) {
227                     /**
228                      * @type Controller
229                      */
230                     var menu = getMenu(this);
231                     if (menu) {
232                         menu.removeChild(c, destroy);
233                     }
234                 },
235 
236                 removeItems:function (destroy) {
237                     var menu = getMenu(this);
238                     menu && menu.removeChildren(destroy);
239                 },
240 
241                 getItemAt:function (index) {
242                     var menu = getMenu(this);
243                     return menu && menu.getChildAt(index);
244                 },
245 
246                 // 禁用时关闭已显示菜单
247                 _uiSetDisabled:function (v) {
248                     var self = this;
249                     MenuButton.superclass._uiSetDisabled.apply(self, S.makeArray(arguments));
250                     !v && self.set("collapsed", true);
251                 },
252 
253                 /**
254                  * @private
255                  */
256                 decorateChildrenInternal:function (ui, el, cls) {
257                     // 不能用 diaplay:none , menu 的隐藏是靠 visibility
258                     // eg: menu.show(); menu.hide();
259                     el.css("visibility", "hidden");
260                     var self = this,
261                         docBody = S.one(el[0].ownerDocument.body);
262                     docBody.prepend(el);
263                     var menu = new ui(S.mix({
264                         srcNode:el,
265                         prefixCls:cls
266                     }, self.get("menuCfg")));
267                     self.__set("menu", menu);
268                 },
269 
270                 /**
271                  * @private
272                  */
273                 destructor:function () {
274                     var self = this, menu = getMenu(self);
275                     $(win).detach("resize", self._reposition, self);
276                     menu && menu.destroy();
277                 }
278 
279             }, {
280                 ATTRS:{
281                     activeItem:{
282                         view:true
283                     },
284                     menuAlign:{
285                         value:{}
286                     },
287                     menuCfg:{},
288                     decorateChildCls:{
289                         value:"popupmenu"
290                     },
291                     // 不关心选中元素 , 由 select 负责
292                     // selectedItem
293                     menu:{},
294                     collapsed:{
295                         value:true,
296                         view:true
297                     }
298                 },
299                 DefaultRender:MenuButtonRender
300             });
301 
302     Component.UIStore.setUIByClass("menu-button", {
303         priority:Component.UIStore.PRIORITY.LEVEL2,
304         ui:MenuButton
305     });
306 
307     if (1 > 2) {
308         MenuButton.getItemAt();
309     }
310 
311     return MenuButton;
312 }, {
313     requires:["uibase", "node", "button", "./menubuttonrender", "menu", "component"]
314 });
