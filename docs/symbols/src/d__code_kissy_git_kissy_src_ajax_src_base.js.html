  2  * @fileOverview a scalable client io framework
  3  * @author  yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax/base", function (S, JSON, Event, XhrObject, undefined) {
  6 
  7         var rlocalProtocol = /^(?:about|app|app\-storage|.+\-extension|file|widget):$/,
  8             rspace = /\s+/,
  9             rurl = /^([\w\+\.\-]+:)(?:\/\/([^\/?#:]*)(?::(\d+))?)?/,
 10             mirror = function (s) {
 11                 return s;
 12             },
 13             HTTP_PORT = 80,
 14             HTTPS_PORT = 443,
 15             rnoContent = /^(?:GET|HEAD)$/,
 16             curLocation,
 17             curLocationParts;
 18 
 19 
 20         try {
 21             curLocation = location.href;
 22         } catch (e) {
 23             S.log("ajax/base get curLocation error : ");
 24             S.log(e);
 25             // Use the href attribute of an A element
 26             // since IE will modify it given document.location
 27             curLocation = document.createElement("a");
 28             curLocation.href = "";
 29             curLocation = curLocation.href;
 30         }
 31 
 32         curLocationParts = rurl.exec(curLocation);
 33 
 34         var isLocal = rlocalProtocol.test(curLocationParts[1]),
 35             transports = {},
 36             defaultConfig = {
 37                 // isLocal:isLocal,
 38                 type:"GET",
 39                 // only support utf-8 when post, encoding can not be changed actually
 40                 contentType:"application/x-www-form-urlencoded; charset=UTF-8",
 41                 async:true,
 42                 // whether add []
 43                 serializeArray:true,
 44                 // whether param data
 45                 processData:true,
 46 
 47                 /*
 48                  url:"",
 49                  context:null,
 50                  // 单位秒!!
 51                  timeout: 0,
 52                  data: null,
 53                  // 可取json | jsonp | script | xml | html | text | null | undefined
 54                  dataType: null,
 55                  username: null,
 56                  password: null,
 57                  cache: null,
 58                  mimeType:null,
 59                  xdr:{
 60                  subDomain:{
 61                  proxy:'http://xx.t.com/proxy.html'
 62                  },
 63                  src:''
 64                  },
 65                  headers: {},
 66                  xhrFields:{},
 67                  // jsonp script charset
 68                  scriptCharset:null,
 69                  crossdomain:false,
 70                  forceScript:false,
 71                  */
 72 
 73                 accepts:{
 74                     xml:"application/xml, text/xml",
 75                     html:"text/html",
 76                     text:"text/plain",
 77                     json:"application/json, text/javascript",
 78                     *:"*/*"
 79                 },
 80                 converters:{
 81                     text:{
 82                         json:JSON.parse,
 83                         html:mirror,
 84                         text:mirror,
 85                         xml:S.parseXML
 86                     }
 87                 },
 88                 contents:{
 89                     xml:/xml/,
 90                     html:/html/,
 91                     json:/json/
 92                 }
 93             };
 94 
 95         defaultConfig.converters.html = defaultConfig.converters.text;
 96 
 97         function setUpConfig(c) {
 98             // deep mix,exclude context!
 99             var context = c.context;
100             delete c.context;
101             c = S.mix(S.clone(defaultConfig), c || {}, undefined, undefined, true);
102             c.context = context;
103 
104             if (!S.isBoolean(c.crossDomain)) {
105                 var parts = rurl.exec(c.url.toLowerCase());
106                 c.crossDomain = !!( parts &&
107                     ( parts[ 1 ] != curLocationParts[ 1 ] || parts[ 2 ] != curLocationParts[ 2 ] ||
108                         ( parts[ 3 ] || ( parts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) )
109                             !=
110                             ( curLocationParts[ 3 ] || ( curLocationParts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) ) )
111                     );
112             }
113 
114             if (c.processData && c.data && !S.isString(c.data)) {
115                 // 必须 encodeURIComponent 编码 utf-8
116                 c.data = S.param(c.data, undefined, undefined, c.serializeArray);
117             }
118 
119             // fix #90 ie7 about "//x.htm"
120             c.url = c.url.replace(/^\/\//, curLocationParts[1] + "//");
121             c.type = c.type.toUpperCase();
122             c.hasContent = !rnoContent.test(c.type);
123 
124             if (!c.hasContent) {
125                 if (c.data) {
126                     c.url += ( /\?/.test(c.url) ? "&" : "?" ) + c.data;
127                     delete c.data;
128                 }
129                 if (c.cache === false) {
130                     c.url += ( /\?/.test(c.url) ? "&" : "?" ) + "_ksTS=" + (S.now() + "_" + S.guid());
131                 }
132             }
133 
134             // 数据类型处理链，一步步将前面的数据类型转化成最后一个
135             c.dataType = S.trim(c.dataType || "*").split(rspace);
136 
137             c.context = c.context || c;
138             return c;
139         }
140 
141         function fire(eventType, xhrObject) {
142             /**
143              * @name io#complete
144              * @description 请求完成（成功或失败）后触发
145              * @event
146              * @param {Event.Object} e
147              * @param {Object} e.ajaxConfig 当前请求的配置
148              * @param {io.XhrObject} e.xhr 当前请求对象
149              */
150 
151             /**
152              * @name io#success
153              * @description 请求成功后触发
154              * @event
155              * @param {Event.Object} e
156              * @param {Object} e.ajaxConfig 当前请求的配置
157              * @param {io.XhrObject} e.xhr 当前请求对象
158              */
159 
160             /**
161              * @name io#error
162              * @description 请求失败后触发
163              * @event
164              * @param {Event.Object} e
165              * @param {Object} e.ajaxConfig 当前请求的配置
166              * @param {io.XhrObject} e.xhr 当前请求对象
167              */
168             io.fire(eventType, { ajaxConfig:xhrObject.config, xhr:xhrObject});
169         }
170 
171         /**
172          * @name io
173          * @description kissy io framework
174          * @namespace io framework
175          * @function
176          * @param {Object} c 发送请求配置选项
177          * @param {String} c.url 请求地址
178          */
179         function io(c) {
180             if (!c.url) {
181                 return undefined;
182             }
183             c = setUpConfig(c);
184             var xhrObject = new XhrObject(c);
185 
186 
187             /**
188              * @name io#start
189              * @description 生成请求对象前触发
190              * @event
191              * @param {Event.Object} e
192              * @param {Object} e.ajaxConfig 当前请求的配置
193              * @param {io.XhrObject} e.xhr 当前请求对象
194              */
195 
196             fire("start", xhrObject);
197             var transportContructor = transports[c.dataType[0]] || transports["*"],
198                 transport = new transportContructor(xhrObject);
199             xhrObject.transport = transport;
200 
201             if (c.contentType) {
202                 xhrObject.setRequestHeader("Content-Type", c.contentType);
203             }
204             var dataType = c.dataType[0],
205                 accepts = c.accepts;
206             // Set the Accepts header for the server, depending on the dataType
207             xhrObject.setRequestHeader(
208                 "Accept",
209                 dataType && accepts[dataType] ?
210                     accepts[ dataType ] + (dataType === "*" ? "" : ", */*; q=0.01"  ) :
211                     accepts[ "*" ]
212             );
213 
214             // Check for headers option
215             for (var i in c.headers) {
216                 xhrObject.setRequestHeader(i, c.headers[ i ]);
217             }
218 
219             function genHandler(handleStr) {
220                 return function (v) {
221                     if (xhrObject.timeoutTimer) {
222                         clearTimeout(xhrObject.timeoutTimer);
223                         xhrObject.timeoutTimer = 0;
224                     }
225                     var h = c[handleStr];
226                     h && h.apply(c.context, v);
227                     fire(handleStr, xhrObject);
228                 };
229             }
230 
231             xhrObject.then(genHandler("success"), genHandler("error"));
232 
233             xhrObject.fin(genHandler("complete"));
234 
235             xhrObject.readyState = 1;
236 
237             /**
238              * @name io#send
239              * @description 发送请求前触发
240              * @event
241              * @param {Event.Object} e
242              * @param {Object} e.ajaxConfig 当前请求的配置
243              * @param {io.XhrObject} xhr 当前请求对象
244              */
245 
246             fire("send", xhrObject);
247 
248             // Timeout
249             if (c.async && c.timeout > 0) {
250                 xhrObject.timeoutTimer = setTimeout(function () {
251                     xhrObject.abort("timeout");
252                 }, c.timeout * 1000);
253             }
254 
255             try {
256                 // flag as sending
257                 xhrObject.state = 1;
258                 transport.send();
259             } catch (e) {
260                 // Propagate exception as error if not done
261                 if (xhrObject.status < 2) {
262                     xhrObject._xhrReady(-1, e);
263                     // Simply rethrow otherwise
264                 } else {
265                     S.error(e);
266                 }
267             }
268 
269             return xhrObject;
270         }
271 
272         S.mix(io, Event.Target);
273 
274         S.mix(io, {
275             isLocal:isLocal,
276             setupConfig:function (setting) {
277                 S.mix(defaultConfig, setting, undefined, undefined, true);
278             },
279             setupTransport:function (name, fn) {
280                 transports[name] = fn;
281             },
282             getTransport:function (name) {
283                 return transports[name];
284             },
285             getConfig:function () {
286                 return defaultConfig;
287             }
288         });
289 
290         return io;
291     },
292     {
293         requires:["json", "event", "./XhrObject"]
294     });
