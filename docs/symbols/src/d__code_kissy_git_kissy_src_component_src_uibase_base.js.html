  7     var UI_SET = '_uiSet',
  8         SRC_NODE = 'srcNode',

 10         HTML_PARSER = 'HTML_PARSER',

 14     /**
 15      * UIBase for class-based component.
 16      * @class
 17      * @memberOf Component
 18      * @extends Base
 19      */
 20     function UIBase(config) {
 21         var self = this;
 22 
 23         // 读取用户设置的属性值并设置到自身
 24         Base.apply(self, arguments);
 25         // 根据 srcNode 设置属性值
 26         // 按照类层次执行初始函数，主类执行 initializer 函数，扩展类执行构造器函数
 27         initHierarchy(self, config);
 28         // 是否自动渲染
 29         config && config.autoRender && self.render();
 30 
 31         /**
 32          * @name Component.UIBase#afterRenderUI
 33          * @description fired when root node is ready
 34          * @event
 35          * @param e
 36          */
 37 
 38 
 39         /**
 40          * @name Component.UIBase#afterBindUI
 41          * @description fired when component 's internal event is binded.
 42          * @event
 43          * @param e
 44          */
 45     }

 47     /**
 48      * 模拟多继承
 49      * init attr using constructors ATTRS meta info
 50      */
 51     function initHierarchy(host, config) {
 52 
 53         var c = host.constructor;
 54 
 55         while (c) {
 56 
 57             // 从 markup 生成相应的属性项
 58             if (config &&
 59                 config[SRC_NODE] &&
 60                 c.HTML_PARSER) {
 61                 if ((config[SRC_NODE] = Node.one(config[SRC_NODE]))) {
 62                     applyParser.call(host, config, c.HTML_PARSER);
 63                 }
 64             }
 65 
 66             c = c.superclass && c.superclass.constructor;
 67         }
 68 
 69         callMethodByHierarchy(host, "initializer", "constructor");
 70 
 71     }
 72 
 73     function callMethodByHierarchy(host, mainMethod, extMethod) {
 74         var c = host.constructor,
 75             extChains = [],
 76             ext,
 77             main,
 78             exts,
 79             t;
 80 
 81         // define
 82         while (c) {
 83 
 84             // 收集扩展类
 85             t = [];
 86             if (exts = c.__ks_exts) {
 87                 for (var i = 0; i < exts.length; i++) {
 88                     ext = exts[i];
 89                     if (ext) {
 90                         if (extMethod != "constructor") {
 91                             //只调用真正自己构造器原型的定义，继承原型链上的不要管
 92                             if (ext.prototype.hasOwnProperty(extMethod)) {
 93                                 ext = ext.prototype[extMethod];
 94                             } else {
 95                                 ext = null;
 96                             }
 97                         }
 98                         ext && t.push(ext);
 99                     }
100                 }
101             }
102 
103             // 收集主类
104             // 只调用真正自己构造器原型的定义，继承原型链上的不要管 !important
105             // 所以不用自己在 renderUI 中调用 superclass.renderUI 了，UIBase 构造器自动搜寻
106             // 以及 initializer 等同理
107             if (c.prototype.hasOwnProperty(mainMethod) && (main = c.prototype[mainMethod])) {
108                 t.push(main);
109             }
110 
111             // 原地 reverse
112             if (t.length) {
113                 extChains.push.apply(extChains, t.reverse());
114             }
115 
116             c = c.superclass && c.superclass.constructor;
117         }
118 
119         // 初始化函数
120         // 顺序：父类的所有扩展类函数 -> 父类对应函数 -> 子类的所有扩展函数 -> 子类对应函数
121         for (i = extChains.length - 1; i >= 0; i--) {
122             extChains[i] && extChains[i].call(host);
123         }
124     }

126     /**
127      * 销毁组件
128      * 顺序： 子类 destructor -> 子类扩展 destructor -> 父类 destructor -> 父类扩展 destructor
129      */
130     function destroyHierarchy(host) {
131         var c = host.constructor,
132             extensions,
133             d,
134             i;
135 
136         while (c) {
137             // 只触发该类真正的析构器，和父亲没关系，所以不要在子类析构器中调用 superclass
138             if (c.prototype.hasOwnProperty("destructor")) {
139                 c.prototype.destructor.apply(host);
140             }
141 
142             if ((extensions = c.__ks_exts)) {
143                 for (i = extensions.length - 1; i >= 0; i--) {
144                     d = extensions[i] && extensions[i].prototype.__destructor;
145                     d && d.apply(host);
146                 }
147             }
148 
149             c = c.superclass && c.superclass.constructor;
150         }
151     }

153     function applyParser(config, parser) {
154         var host = this, p, v, srcNode = config[SRC_NODE];
155 
156         // 从 parser 中，默默设置属性，不触发事件
157         for (p in parser) {
158             if (parser.hasOwnProperty(p) &&
159                 // 用户设置过那么这里不从 dom 节点取
160                 config[p] === undefined) {
161                 v = parser[p];
162                 // 函数
163                 if (S.isFunction(v)) {
164                     host.__set(p, v.call(host, srcNode));
165                 }
166                 // 单选选择器
167                 else if (S.isString(v)) {
168                     host.__set(p, srcNode.one(v));
169                 }
170                 // 多选选择器
171                 else if (S.isArray(v) && v[0]) {
172                     host.__set(p, srcNode.all(v[0]))
173                 }
174             }
175         }
176     }

178     /**
179      * 根据属性变化设置 UI
180      */
181     function bindUI(self) {
182         var attrs = self.getAttrs(),
183             attr, m;
184 
185         for (attr in attrs) {
186             if (attrs.hasOwnProperty(attr)) {
187                 m = UI_SET + ucfirst(attr);
188                 if (self[m]) {
189                     // 自动绑定事件到对应函数
190                     (function (attr, m) {
191                         self.on('after' + ucfirst(attr) + 'Change', function (ev) {
192                             // fix! 防止冒泡过来的
193                             if (ev.target === self) {
194                                 self[m](ev.newVal, ev);
195                             }
196                         });
197                     })(attr, m);
198                 }
199             }
200         }
201     }
202 
203     /**
204      * 根据当前（初始化）状态来设置 UI
205      */
206     function syncUI(self) {
207         var v,
208             f,
209             attrs = self.getAttrs();
210         for (var a in attrs) {
211             if (attrs.hasOwnProperty(a)) {
212                 var m = UI_SET + ucfirst(a);
213                 //存在方法，并且用户设置了初始值或者存在默认值，就同步状态
214                 if ((f = self[m])
215                     // 用户如果设置了显式不同步，就不同步，比如一些值从 html 中读取，不需要同步再次设置
216                     && attrs[a].sync !== false
217                     && (v = self.get(a)) !== undefined) {
218                     f.call(self, v);
219                 }
220             }
221         }
222     }

238     UIBase.ATTRS =

242     {
243         /**
244          * Whether this component is rendered.
245          * @type Boolean
246          */
247         rendered:{
248             value:false
249         },
250         /**
251          * Whether this component 's dom structure is created.
252          * @type Boolean

255             value:false

259     S.extend(UIBase, Base,
260         /**
261          * @lends Component.UIBase.prototype
262          */
263         {
264 
265             /**
266              * Create dom structure of this component.
267              */
268             create:function () {
269                 var self = this;
270                 // 是否生成过节点
271                 if (!self.get("created")) {
272                     self.fire('beforeCreateDom');
273                     callMethodByHierarchy(self, "createDom", "__createDom");
274                     self.__set("created", true);
275                     self.fire('afterCreateDom');
276                 }
277                 return self;
278             },
279 
280             /**
281              * Put dom structure of this component to document and bind event.
282              */
283             render:function () {
284                 var self = this;
285                 // 是否已经渲染过
286                 if (!self.get("rendered")) {
287                     self.create(undefined);
288                     self.fire('beforeRenderUI');
289                     callMethodByHierarchy(self, "renderUI", "__renderUI");
290                     self.fire('afterRenderUI');
291                     self.fire('beforeBindUI');
292                     bindUI(self);
293                     callMethodByHierarchy(self, "bindUI", "__bindUI");
294                     self.fire('afterBindUI');
295                     self.fire('beforeSyncUI');
296                     syncUI(self);
297                     callMethodByHierarchy(self, "syncUI", "__syncUI");
298                     self.fire('afterSyncUI');
299                     self.__set("rendered", true);
300                 }
301                 return self;
302             },
303 
304             /**
305              * For overridden. DOM creation logic of subclass component.
306              * @protected
307              * @function

309             createDom:noop,

334              * Destroy this component.

336             destroy:function () {
337                 var self = this;
338                 destroyHierarchy(self);
339                 self.fire('destroy');
340                 self.detach();
341                 return self;

344         /**
345          * @lends Component.UIBase
346          */
347         {
348             /**
349              * Create a new class which extends UIBase.
350              * @param {Function|Function[]} base Parent class constructor.
351              * @param {Function[]} extensions Class constructors for extending.
352              * @param {Object} px Object to be mixed into new class 's prototype.
353              * @param {Object} sx Object to be mixed into new class.
354              * @returns {UIBase} A new class which extends UIBase.
355              */
356             create:function (base, extensions, px, sx) {
357                 var args = S.makeArray(arguments), t;
358                 if (S.isArray(base)) {
359                     sx = px;
360                     px = extensions;
361                     extensions = /*@type Function[]*/base;
362                     base = UIBase;
363                 }
364                 base = base || UIBase;
365                 if (S.isObject(extensions)) {
366                     sx = px;
367                     px = extensions;
368                     extensions = [];
369                 }
370 
371                 var name = "UIBaseDerived";
372 
373                 if (S.isString(t = args[args.length - 1])) {
374                     name = t;
375                 }
376 
377                 function C() {
378                     UIBase.apply(this, arguments);
379                 }
380 
381                 // debug mode , give the right name for constructor
382                 // refer : http://limu.iteye.com/blog/1136712
383                 S.log("UIBase.create : " + name, eval("C=function " + name.replace(/[-.]/g, "_") + "(){ UIBase.apply(this, arguments);}"));
384 
385                 S.extend(C, base, px, sx);
386 
387                 if (extensions) {
388                     C.__ks_exts = extensions;
389 
390                     var desc = {
391                         // ATTRS:
392                         // HTML_PARSER:
393                     }, constructors = extensions['concat'](C);
394 
395                     // [ex1,ex2]，扩展类后面的优先，ex2 定义的覆盖 ex1 定义的
396                     // 主类最优先
397                     S.each(constructors, function (ext) {
398                         if (ext) {
399                             // 合并 ATTRS/HTML_PARSER 到主类
400                             S.each([ATTRS, HTML_PARSER], function (K) {
401                                 if (ext[K]) {
402                                     desc[K] = desc[K] || {};
403                                     // 不覆盖主类上的定义，因为继承层次上扩展类比主类层次高
404                                     // 但是值是对象的话会深度合并
405                                     // 注意：最好值是简单对象，自定义 new 出来的对象就会有问题(用 function return 出来)!
406                                     S.mix(desc[K], ext[K], {
407                                         deep:true
408                                     });
409                                 }
410                             });
411                         }
412                     });
413 
414                     S.each(desc, function (v, k) {
415                         C[k] = v;
416                     });
417 
418                     var prototype = {};
419 
420                     // 主类最优先
421                     S.each(constructors, function (ext) {
422                         if (ext) {
423                             var proto = ext.prototype;
424                             // 合并功能代码到主类，不覆盖
425                             for (var p in proto) {
426                                 // 不覆盖主类，但是主类的父类还是覆盖吧
427                                 if (proto.hasOwnProperty(p)) {
428                                     prototype[p] = proto[p];
429                                 }
430                             }
431                         }
432                     });
433 
434                     S.each(prototype, function (v, k) {
435                         C.prototype[k] = v;
436                     });
437                 }
438                 return C;
439             }
440         });
