1 /**
  2  * @fileOverview only one droppable instance for multiple droppable nodes
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("dd/droppable-delegate", function (S, DDM, Droppable, DOM, Node) {
  6 
  7 
  8     function dragStart() {
  9         var self = this,
 10             container = self.get("container"),
 11             selector = self.get("selector");
 12         self.__allNodes = container.all(selector);
 13     }
 14 
 15     /**
 16      * make multiple nodes droppable under a container using only one droppable instance
 17      * @memberOf DD
 18      * @class
 19      */
 20     function DroppableDelegate() {
 21         var self = this;
 22         DroppableDelegate.superclass.constructor.apply(self, arguments);
 23         // 提高性能，拖放开始时缓存代理节点
 24         DDM.on("dragstart", dragStart, self);
 25     }
 26 
 27     S.extend(DroppableDelegate, Droppable, {
 28 
 29             /**
 30              * 根据鼠标位置得到真正的可放目标，暂时不考虑 mode，只考虑鼠标
 31              * @param ev
 32              * @private
 33              * @override
 34              */
 35             getNodeFromTarget:function (ev, dragNode, proxyNode) {
 36                 var pointer = {
 37                     left:ev.pageX,
 38                     top:ev.pageY
 39                 },
 40                     self = this,
 41                     allNodes = self.__allNodes,
 42                     ret = 0,
 43                     vArea = Number.MAX_VALUE;
 44 
 45 
 46                 if (allNodes) {
 47                     allNodes.each(function (n) {
 48                         var domNode = n[0];
 49                         // 排除当前拖放的元素以及代理节点
 50                         if (domNode === proxyNode || domNode === dragNode) {
 51                             return;
 52                         }
 53                         if (DDM.inRegion(DDM.region(n), pointer)) {
 54                             // 找到面积最小的那个
 55                             var a = DDM.area(DDM.region(n));
 56                             if (a < vArea) {
 57                                 vArea = a;
 58                                 ret = n;
 59                             }
 60                         }
 61                     });
 62                 }
 63 
 64                 if (ret) {
 65                     self.__set("lastNode", self.get("node"));
 66                     self.__set("node", ret);
 67                 }
 68 
 69                 return ret;
 70             },
 71 
 72             _handleOut:function () {
 73                 var self = this;
 74                 DroppableDelegate.superclass._handleOut.apply(self, arguments);
 75                 self.__set("node", 0);
 76                 self.__set("lastNode", 0);
 77             },
 78 
 79             _handleOver:function (ev) {
 80                 var self = this,
 81                     node = self.get("node"),
 82                     superOut = DroppableDelegate.superclass._handleOut,
 83                     superOver = DroppableDelegate.superclass._handleOver,
 84                     superEnter = DroppableDelegate.superclass._handleEnter,
 85                     lastNode = self.get("lastNode");
 86 
 87                 if (lastNode[0] !== node[0]) {
 88                     /**
 89                      * 同一个 drop 对象内委托的两个可 drop 节点相邻，先通知上次的离开
 90                      */
 91                     self.__set("node", lastNode);
 92                     superOut.apply(self, arguments);
 93                     /**
 94                      * 再通知这次的进入
 95                      */
 96                     self.__set("node", node);
 97                     superEnter.call(self, ev);
 98                 } else {
 99                     superOver.call(self, ev);
100                 }
101             },
102 
103             _end:function () {
104                 var self = this;
105                 DroppableDelegate.superclass._end.apply(self, arguments);
106                 self.__set("node", 0);
107             }
108         },
109         {
110             ATTRS:/**
111              * @lends DD.DroppableDelegate#
112              */
113             {
114 
115 
116                 // 继承自 Drappable ，当前正在委托的放节点目标
117 
118 
119                 /**
120                  * 上一个成为放目标的委托节点
121                  * @private
122                  */
123                 lastNode:{
124                 },
125 
126                 /**
127                  * a selector query to get the children of container to make droppable elements from.
128                  * usually as for tag.cls.
129                  * @type String
130                  */
131                 selector:{
132                 },
133 
134                 /**
135                  * a selector query to get the container to listen for mousedown events on.
136                  * All "draggable selector" should be a child of this container
137                  * @type {String|HTMLElement}
138                  */
139                 container:{
140                     setter:function (v) {
141                         return Node.one(v);
142                     }
143                 }
144             }
145         });
146 
147     return DroppableDelegate;
148 }, {
149     requires:['./ddm', './droppable', 'dom', 'node']
150 });
