SWFAudio Test using soundmanager2

使用举例:采用soundmanager2延迟初始化，消除全局污染
简单使用可用 swfaudiolite.js，都具有 play（开始/继续）,stop（停止）,pause（暂停）

代码：


KISSY.ready(function(){
	
var S=KISSY;
	/**
		统一使用延迟初始化，其他方式有问题
		指定本地的swf地址
	*/
	var sm = new S.SoundManager("soundmanager2.swf");
	sm.useFlashBlock = true;
	//开始初始化
	sm.beginDelayedInit();
	sm.onready(function(){
		//具体使用方法参见：http://www.schillmania.com/projects/soundmanager2/
		
		/*播放音乐方式1：
			@param {String} music id
			@param {String} music path
		*/
		//sm.play("test","waka.mp3");
		
		
		
		/*
			播放音乐方式2：
			createSound方法，可以播放前缓存
			@param autoLoad 播放前缓存
		*/
		
		var s=sm.createSound({
                id: "music_1",
                url: "waka.mp3",
                autoPlay:true,  //注意使用 autoPlay 立即播放 ，
                autoLoad: true //开始下载
            });
  


	
	
	
	//停止
		var _stop=S.one("#_stop");
		_stop.on("click",function(){
			s.stop();
		});
		
		//开始/继续当前
		var _start=S.one("#_start");
		_start.on("click",function(){
			s.play();
		});
		
		//暂停
		var _pause=S.one("#_pause");
		_pause.on("click",function(){
			s.pause();
		});
		
		
				
	});

});
